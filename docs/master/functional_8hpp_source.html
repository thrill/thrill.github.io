123 //! apply the Functor to each item in a std::vector<T> and return a new
124 //! std::vector<U> with different type.
125 template <typename Type, typename Functor>
126 inline auto MapVector(const std::vector<Type>&input, const Functor &f)
127 ->std::vector<typename std::result_of<Functor(Type)>::type>
128 {
129  std::vector<typename std::result_of<Functor(Type)>::type> output;
130  output.reserve(input.size());
131  for (typename std::vector<Type>::const_iterator it = input.begin();
132  it != input.end(); ++it) {
133  output.emplace_back(f(*it));
134  }
135  return output;
136 }

145 //! Compute the component-wise sum of two std::array<T,N> of same sizes.
146 template <typename Type, size_t N, typename Operation>
147 class ComponentSum<std::array<Type, N>, Operation>
148  : public std::binary_function<
149  std::array<Type, N>, std::array<Type, N>, std::array<Type, N> >
150 {
151 public:

153  explicit ComponentSum(const Operation& op = Operation()) : op_(op) { }
154  ArrayType operator () (const ArrayType& a, const ArrayType& b) const {
155  ArrayType out;

164 //! Compute the component-wise sum of two std::vector<T> of same sizes.
165 template <typename Type, typename Operation>
166 class ComponentSum<std::vector<Type>, Operation>
167  : public std::binary_function<
168  std::vector<Type>, std::vector<Type>, std::vector<Type> >
169 {
170 public:
171  using VectorType = std::vector<Type>;
172  explicit ComponentSum(const Operation& op = Operation()) : op_(op) { }
173  VectorType operator () (const VectorType& a, const VectorType& b) const {
174  assert(a.size() == b.size());
175  VectorType out;
176  out.reserve(std::min(a.size(), b.size()));
177  for (size_t i = 0; i < min(a.size(), b.size()); ++i)
178  out.emplace_back(op_(a[i], b[i]));
179  return out;
180  }
181 
182 private:
183  Operation op_;
184 };

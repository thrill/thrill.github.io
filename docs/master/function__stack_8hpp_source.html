31 /*!
32  * Base case for the chaining of lambda functions. The last lambda function
33  * receives an input element, no emitter and should have no return type. It
34  * should therefore store the input parameter externally.
35  *
36  * \param lambda Lambda function that represents the chain end.
37  */
38 template <typename Lambda>
39 auto RunEmitter(const Lambda& lambda) {
40  // lambda is captured by non-const copy so that we can use functors with
41  // non-const operator(), i.e. stateful functors (e.g. for sampling)
42  return [ =, lambda = lambda](const auto& input) mutable->void {
43  lambda(input);
44  };
45 }
46 
47 /*!
48  * Recursive case for the chaining of lambda functions. The given lambda
49  * function receives an input element, an emitter and should have no return
50  * type. The emitter will be built using the chain of remaining lambda
51  * functions.
52  *
53  * \param lambda Current lambda function to be chained.
54  *
55  * \param rest Remaining lambda functions.
56  */
57 template <typename Lambda, typename ... MoreLambdas>
58 auto RunEmitter(const Lambda& lambda, const MoreLambdas& ... rest) {
59  // lambda is captured by non-const copy so that we can use functors with
60  // non-const operator(), i.e. stateful functors (e.g. for sampling)
61  return [ =, lambda = lambda](const auto& input) mutable->void {
62  lambda(input, RunEmitter(rest ...));
63  };
64 }

66 /*!
67  * A FunctionStack is a chain of lambda functions that can be folded to a single
68  * lambda functions. The FunctionStack basically consists of a tuple that
69  * contains lambda functions of varying types. All lambda functions within the
70  * chain receive a single input value and a emitter function. The emitter
71  * function is used for chaining lambdas together. The single exception to this
72  * is the last lambda function, which receives no emitter.
73  *
74  * \tparam Lambdas Types of the different lambda functions.
75  */
76 template <typename Input_, typename ... Lambdas>
77 class FunctionStack
78 {
79 public:
80  using Input = Input_;
81 
82  FunctionStack() = default;
83 
84  /*!
85  * Initialize the function chain with a given tuple of functions.
86  *
87  * \param stack Tuple of lambda functions.
88  */
89  explicit FunctionStack(const std::tuple<Lambdas ...>& stack)
90  : stack_(stack) { }
91 
92  /*!
93  * Add a lambda function to the end of the chain.
94  *
95  * \tparam Function Type of the lambda functions.
96  *
97  * \param append_func Lambda function that should be added to the chain.
98  *
99  * \return New chain containing the previous and new lambda function(s).
100  */
101  template <typename Function>
102  auto push(const Function& append_func) const {
103  // append to function stack's type the new function: we prepend it to
104  // the type line because later we will
105  std::tuple<Lambdas ..., Function> new_stack
106  = std::tuple_cat(stack_, std::make_tuple(append_func));
107 
108  return FunctionStack<Input, Lambdas ..., Function>(new_stack);
109  }
110 
111  /*!
112  * Build a single lambda function by "folding" the chain. Folding means
113  * that the chain is processed from back to front and each emitter is
114  * composed using previous lambda functions.
115  *
116  * \return Single "folded" lambda function representing the chain.
117  */
118  auto fold() const {
119  const size_t Size = sizeof ... (Lambdas);
120  return EmitSequence(common::make_index_sequence<Size>{ });
121  }
122 
123  //! Is true if the FunctionStack is empty.
124  static constexpr bool empty = (sizeof ... (Lambdas) == 0);
125 
126 private:
127  //! Tuple of varying type that stores all lambda functions.
128  std::tuple<Lambdas ...> stack_;
129 
130  /*!
131  * Auxilary function for "folding" the chain. This is needed to send all
132  * lambda functions as parameters to the function that folds them together.
133  *
134  * \return Single "folded" lambda function representing the chain.
135  */
136  template <size_t ... Is>
137  auto EmitSequence(common::index_sequence<Is ...>) const {
138  return RunEmitter(std::get<Is>(stack_) ...);
139  }
140 };
141 
142 template <typename Input, typename Lambda>
143 static inline auto MakeFunctionStack(const Lambda& lambda) {
144  return FunctionStack<Input, Lambda>(std::make_tuple(lambda));
145 }

31 /*!
32  * This is an expected O(1) LRU cache which contains a set of (key,value)
33  * elements. Elements can be put() into LRU cache, and later retrieves using
34  * get(). Insertion and retrieval will remark the elements as most recently
35  * used, pushing all other back in priority. The LRU cache itself does not limit
36  * the number of items, because it has no eviction mechanism. Instead, the user
37  * program must check size() after an insert and may extract the least recently
38  * used element.
39  */
40 template <typename Key, typename Value>
41 class LruCache
42 {
43 public:
44  using KeyValuePair = typename std::pair<Key, Value>;
45 
46  using List = typename std::list<KeyValuePair>;
47  using ListIterator = typename List::iterator;
48 
49  using Map = typename std::unordered_map<Key, ListIterator>;
50 
51  //! put or replace/touch item in LRU cache
52  void put(const Key& key, const Value& value) {
53  // first try to find an existing key
54  typename Map::iterator it = map_.find(key);
55  if (it != map_.end()) {
56  list_.erase(it->second);
57  map_.erase(it);
58  }
59 
60  // insert key into linked list at the front (most recently used)
61  list_.push_front(KeyValuePair(key, value));
62  // store iterator to linked list entry in map
63  map_[key] = list_.begin();
64  }
65 
66  //! get and touch value from LRU cache for key.
67  const Value& get(const Key& key) {
68  typename Map::iterator it = map_.find(key);
69  if (it == map_.end()) {
70  throw std::range_error("There is no such key in cache");
71  }
72  else {
73  list_.splice(list_.begin(), list_, it->second);
74  return it->second->second;
75  }
76  }
77 
78  //! test if key exists in LRU cache
79  bool exists(const Key& key) const {
80  return map_.find(key) != map_.end();
81  }
82 
83  //! return number of items in LRU cache
84  size_t size() const {
85  return map_.size();
86  }
87 
88  //! return the least recently used key value pair
89  KeyValuePair pop() {
90  assert(size());
91  typename List::iterator last = list_.end();
92  --last;
93  KeyValuePair out = *last;
94  map_.erase(last->first);
95  list_.pop_back();
96  return out;
97  }
98 
99 private:
100  //! list of entries in least-recently used order.
101  List list_;
102  //! map for accelerated access to keys
103  Map map_;
104 };

106 /*!
107  * This is an expected O(1) LRU cache which contains a set of key-only
108  * elements. Elements can be put() into LRU cache, and tested for existence
109  * using exists(). Insertion and touch() will remark the elements as most
110  * recently used, pushing all other back in priority. The LRU cache itself does
111  * not limit the number of items, because it has no eviction mechanism. Instead,
112  * the user program must check size() after an insert and may extract the least
113  * recently used element.
114  */
115 template <typename Key, typename Alloc = std::allocator<Key> >
116 class LruCacheSet
117 {
118 public:
119  using List = typename std::list<Key, Alloc>;
120  using ListIterator = typename List::iterator;
121 
122  using Map = typename std::unordered_map<
123  Key, ListIterator, std::hash<Key>, std::equal_to<Key>,
124  typename Alloc::template rebind<
125  std::pair<const Key, ListIterator> >::other>;
126 
127  explicit LruCacheSet(const Alloc& alloc = Alloc())
128  : list_(alloc), map_(alloc) { }

133  typename Map::iterator it = map_.find(key);
134  if (it != map_.end()) {
135  list_.erase(it->second);
136  map_.erase(it);
137  }

141  // store iterator to linked list entry in map
142  map_[key] = list_.begin();

145  //! touch value from LRU cache for key.

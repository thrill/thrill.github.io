26 //! Add x + y but truncate result upwards such that it fits into original
27 //! datatype
28 template <typename IntegerType, unsigned bits = (8* sizeof(IntegerType))>
29 static inline
30 IntegerType AddTruncToType(const IntegerType& a, const IntegerType& b) {
31  size_t s = static_cast<size_t>(a) + static_cast<size_t>(b);
32  if (s >= (size_t(1) << bits))
33  s = (size_t(1) << bits) - 1;
34  return static_cast<IntegerType>(s);

101 /*!
102  * Calculate a Perfect Matching (1-Factor) on a Complete Graph. Used by
103  * collective network algorithms.
104  *
105  * \param r round [0..n-1) of matching
106  * \param p rank of this processor 0..n-1
107  * \param n number of processors (graph size)
108  * \return peer processor in this round
109  */
110 static inline size_t CalcOneFactorPeer(size_t r, size_t p, size_t n) {
111  assert(r < CalcOneFactorSize(n));
112  assert(p < n);
113 
114  if (n % 2 == 0) {
115  // p is even
116  size_t idle = (r * n / 2) % (n - 1);
117  if (p == n - 1)
118  return idle;
119  else if (p == idle)
120  return n - 1;
121  else
122  return (r - p + n - 1) % (n - 1);
123  }
124  else {
125  // p is odd
126  return (r - p + n) % n;
127  }
128 }

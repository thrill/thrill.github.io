93  //! Register a buffered read callback and a default exception callback.
94  void AddRead(int fd, const Callback& read_cb) {
95  CheckSize(fd);
96  if (!watch_[fd].read_cb.size()) {
97  select_.SetRead(fd);
98  select_.SetException(fd);
99  }
100  watch_[fd].active = true;

139  //! Cancel all callbacks on a given fd.
140  void Cancel(net::Connection& c) final {
141  assert(dynamic_cast<Connection*>(&c));
142  Connection& tc = static_cast<Connection&>(c);
143  int fd = tc.GetSocket().fd();
144  CheckSize(fd);
145 
146  if (watch_[fd].read_cb.size() == 0 &&
147  watch_[fd].write_cb.size() == 0)
148  LOG << "SelectDispatcher::Cancel() fd=" << fd
149  << " called with no callbacks registered.";
150 
151  select_.ClearRead(fd);
152  select_.ClearWrite(fd);
153  select_.ClearException(fd);
154 
155  Watch& w = watch_[fd];
156  w.read_cb.clear();
157  w.write_cb.clear();
158  w.except_cb = Callback();
159  w.active = false;
160  }

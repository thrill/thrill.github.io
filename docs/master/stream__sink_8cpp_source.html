66 void StreamSink::AppendPinnedBlock(const PinnedBlock& block, bool is_last_block) {
67  if (block.size() == 0) return;
68 
69  sem_.wait();
70 
71  LOG << "StreamSink::AppendPinnedBlock()"
72  << " block=" << block
73  << " is_last_block=" << is_last_block;
74 
75  sLOG << "sending block" << tlx::hexdump(block.ToString());
76 
77  StreamMultiplexerHeader header(magic_, block);
78  header.stream_id = id_;
79  header.sender_worker = (host_rank_ * workers_per_host()) + local_worker_id_;
80  header.receiver_local_worker = peer_local_worker_;
81  header.is_last_block = is_last_block;
82 
83  net::BufferBuilder bb;
84  header.Serialize(bb);
85 
86  net::Buffer buffer = bb.ToBuffer();
87  assert(buffer.size() == MultiplexerHeader::total_size);
88 
89  item_counter_ += block.num_items();
90  byte_counter_ += buffer.size() + block.size();
91  ++block_counter_;
92 
93  stream_.multiplexer_.dispatcher_.AsyncWrite(
94  *connection_,
95  // send out Buffer and Block, guaranteed to be successive
96  std::move(buffer), PinnedBlock(block),
97  [this](net::Connection&) { sem_.signal(); });
98 
99  if (is_last_block) {
100  assert(!closed_);
101  closed_ = true;
102 
103  // wait for the last Blocks to be transmitted (take away semaphore
104  // tokens)
105  for (size_t i = 0; i < num_queue_; ++i)
106  sem_.wait();
107 
108  LOG << "StreamSink::AppendPinnedBlock()"
109  << " sent 'piggy-backed close stream' id=" << id_
110  << " from=" << my_worker_rank()
111  << " (host=" << host_rank_ << ")"
112  << " to=" << peer_worker_rank()
113  << " (host=" << peer_rank_ << ")";
114 
115  Finalize();
116  }
117 }

115 //! Randomized cycling disk allocation scheme functor.
116 //! \remarks model of \b allocation_strategy concept
117 struct RandomCyclic : public Striping {
118 private:
119  std::vector<size_t> perm_;
120 
121  void init() {
122  for (size_t i = 0; i < diff_; i++)
123  perm_[i] = i;
124 
125  std::random_shuffle(perm_.begin(), perm_.end());
126  }
127 
128 public:
129  RandomCyclic(size_t b, size_t e) : Striping(b, e), perm_(diff_) {
130  init();
131  }
132 
133  RandomCyclic() : Striping(), perm_(diff_) {
134  init();
135  }
136 
137  size_t operator () (size_t i) const {
138  return begin_ + perm_[i % diff_];
139  }
140 
141  static const char * name() {
142  return "randomized cycling striping";
143  }
144 };

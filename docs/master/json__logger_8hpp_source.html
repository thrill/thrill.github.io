164 
165  //! close the line
166  void Close() {
167  if (logger_ && items_ != 0) {
168  assert(items_ % 2 == 0);
169  os_ << '}' << std::endl;
170  items_ = 0;
171  }
172  else if (!logger_ && sub_dict_) {
173  os_ << '}';
174  sub_dict_ = false;
175  }
176  else if (!logger_ && sub_array_) {
177  os_ << ']';
178  sub_array_ = false;
179  }
180  }

214  //! put an items separator (either ',' or ':') and increment counter.
215  void PutSeparator() {
216  if (items_ > 0) {
217  os_ << (items_ % 2 == 0 ? ',' : ':');
218  }
219  items_++;
220  }

222  void PutEscapedChar(char ch) {
223  // from: http://stackoverflow.com/a/7725289
224  switch (ch) {
225  case '\\': os_ << '\\' << '\\';
226  break;
227  case '"': os_ << '\\' << '"';
228  break;
229  case '/': os_ << '\\' << '/';
230  break;
231  case '\b': os_ << '\\' << 'b';
232  break;
233  case '\f': os_ << '\\' << 'f';
234  break;
235  case '\n': os_ << '\\' << 'n';
236  break;
237  case '\r': os_ << '\\' << 'r';
238  break;
239  case '\t': os_ << '\\' << 't';
240  break;
241  default: os_ << ch;
242  break;
243  }
244  }

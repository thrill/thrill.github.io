35 LinuxaioQueue::LinuxaioQueue(int desired_queue_length)
36  : post_thread_state_(NOT_RUNNING), wait_thread_state_(NOT_RUNNING) {
37  if (desired_queue_length == 0) {
38  // default value, 64 entries per queue (i.e. usually per disk) should
39  // be enough
40  max_events_ = 64;
41  }
42  else
43  max_events_ = desired_queue_length;
44 
45  // negotiate maximum number of simultaneous events with the OS
46  context_ = 0;
47  long result;
48  while ((result = syscall(SYS_io_setup, max_events_, &context_)) == -1 &&
49  errno == EAGAIN && max_events_ > 1)
50  {
51  max_events_ <<= 1; // try with half as many events
52  }
53  if (result != 0) {
54  THRILL_THROW_ERRNO(IoError, "linuxaio_queue::linuxaio_queue"
55  " io_setup() nr_events=" << max_events_);
56  }
57 
58  num_free_events_.signal(max_events_);
59 
60  LOG1 << "Set up an linuxaio queue with " << max_events_ << " entries.";
61 
62  StartThread(PostAsync, static_cast<void*>(this), post_thread_, post_thread_state_);
63  StartThread(WaitAsync, static_cast<void*>(this), wait_thread_, wait_thread_state_);
64 }

86 bool LinuxaioQueue::CancelRequest(Request* req) {
87  if (!req)
88  THRILL_THROW_INVALID_ARGUMENT("Empty request canceled disk_queue.");
89  if (post_thread_state_() != RUNNING)
90  LOG1 << "Request canceled in stopped queue.";
91  if (!dynamic_cast<LinuxaioRequest*>(req))
92  LOG1 << "Non-LinuxAIO request submitted to LinuxAIO queue.";
93 
94  Queue::iterator pos;
95  {
96  std::unique_lock<std::mutex> lock(waiting_mtx_);
97 
98  pos = std::find(waiting_requests_.begin(), waiting_requests_.end(), req);
99  if (pos != waiting_requests_.end())
100  {
101  waiting_requests_.erase(pos);
102 
103  // polymorphic_downcast to linuxaio_request,
104  // request is canceled, but was not yet posted.
105  dynamic_cast<LinuxaioRequest*>(req)->completed(false, true);
106 
107  num_waiting_requests_.wait(); // will never block
108  return true;
109  }
110  }
111 
112  std::unique_lock<std::mutex> lock(posted_mtx_);
113 
114  pos = std::find(posted_requests_.begin(), posted_requests_.end(), req);
115  if (pos != posted_requests_.end())
116  {
117  // polymorphic_downcast to linuxaio_request,
118  bool canceled_io_operation = (dynamic_cast<LinuxaioRequest*>(req))->cancel_aio();
119 
120  if (canceled_io_operation)
121  {
122  posted_requests_.erase(pos);
123 
124  // polymorphic_downcast to linuxaio_request,
125 
126  // request is canceled, already posted
127  dynamic_cast<LinuxaioRequest*>(req)->completed(true, true);
128 
129  num_free_events_.signal();
130  num_posted_requests_.wait(); // will never block
131  return true;
132  }
133  }
134 
135  return false;
136 }

139 void LinuxaioQueue::PostRequests() {
140  RequestPtr req;
141  io_event* events = new io_event[max_events_];
142 
143  for ( ; ; ) // as long as thread is running
144  {
145  // might block until next request or message comes in
146  size_t num_currently_waiting_requests = num_waiting_requests_.wait();
147 
148  // terminate if termination has been requested
149  if (post_thread_state_() == TERMINATING && num_currently_waiting_requests == 0)
150  break;
151 
152  std::unique_lock<std::mutex> lock(waiting_mtx_);
153  if (!waiting_requests_.empty())
154  {
155  req = waiting_requests_.front();
156  waiting_requests_.pop_front();
157  lock.unlock();
158 
159  // might block because too many requests are posted
160  num_free_events_.wait();
161 
162  // polymorphic_downcast
163  while (!dynamic_cast<LinuxaioRequest*>(req.get())->post())
164  {
165  // post failed, so first handle events to make queues (more)
166  // empty, then try again.
167 
168  // wait for at least one event to complete, no time limit
169  long num_events = syscall(SYS_io_getevents, context_, 1, max_events_, events, nullptr);
170  if (num_events < 0) {
171  THRILL_THROW_ERRNO(IoError, "linuxaio_queue::post_requests"
172  " io_getevents() nr_events=" << num_events);
173  }
174 
175  HandleEvents(events, num_events, false);
176  }
177 
178  // request is finally posted
179 
180  {
181  std::unique_lock<std::mutex> lock(posted_mtx_);
182  posted_requests_.push_back(req);
183  num_posted_requests_.signal();
184  }
185  }
186  else
187  {
188  lock.unlock();
189 
190  // num_waiting_requests-- was premature, compensate for that
191  num_waiting_requests_.signal();
192  }
193  }
194 
195  delete[] events;
196 }

198 void LinuxaioQueue::HandleEvents(io_event* events, long num_events, bool canceled) {
199  for (int e = 0; e < num_events; ++e)
200  {
201  // size_t is as long as a pointer, and like this, we avoid an icpc warning
202  RequestPtr* r = reinterpret_cast<RequestPtr*>(static_cast<size_t>(events[e].data));
203  r->get()->completed(canceled);
204  delete r; // release auto_ptr reference
205  num_free_events_.signal();
206  num_posted_requests_.wait(); // will never block
207  }
208 }

211 void LinuxaioQueue::WaitRequests() {
212  RequestPtr req;
213  io_event* events = new io_event[max_events_];
214 
215  for ( ; ; ) // as long as thread is running
216  {
217  // might block until next request is posted or message comes in
218  size_t num_currently_posted_requests = num_posted_requests_.wait();
219 
220  // terminate if termination has been requested
221  if (wait_thread_state_() == TERMINATING && num_currently_posted_requests == 0)
222  break;
223 
224  // wait for at least one of them to finish
225  long num_events;
226  for ( ; ; ) {
227  num_events = syscall(SYS_io_getevents, context_, 1, max_events_, events, nullptr);
228  if (num_events < 0) {
229  if (errno == EINTR) {
230  // premature return, e.g. due to signal. Just try again
231  continue;
232  }
233 
234  THRILL_THROW_ERRNO(IoError, "linuxaio_queue::wait_requests"
235  " io_getevents() nr_events=" << max_events_);
236  }
237  break;
238  }
239 
240  // compensate for the one eaten prematurely above
241  num_posted_requests_.signal();
242 
243  HandleEvents(events, num_events, false);
244  }
245 
246  delete[] events;
247 }

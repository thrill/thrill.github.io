111

113  }
114 
115  // dump();

119  bind2nd(FirstFit(), requested_size));
120 
121  if (space == data_->free_space_.end() && requested_size == block_size)
122  {
123  assert(end - begin == 1);

235 void DiskAllocator::AddFreeRegion(int64_t block_pos, int64_t block_size) {
236  // assert(block_size);
237  // dump();
238  LOG << "Deallocating a block with size: " << block_size << " position: " << block_pos;
239  int64_t region_pos = block_pos;
240  int64_t region_size = block_size;
241  SortSeq& free_space = data_->free_space_;
242  if (!free_space.empty())
243  {
244  SortSeq::iterator succ = free_space.upper_bound(region_pos);
245  SortSeq::iterator pred = succ;
246  if (pred != free_space.begin())
247  pred--;
248  if (pred != free_space.end())
249  {
250  if (pred->first <= region_pos && pred->first + pred->second > region_pos)
251  {
252  THRILL_THROW2(BadExternalAlloc,
253  "disk_allocator::check_corruption", "Error: double deallocation of external memory, trying to deallocate region " << region_pos << " + " << region_size << " in empty space [" << pred->first << " + " << pred->second << "]");
254  }
255  }
256  if (succ != free_space.end())
257  {
258  if (region_pos <= succ->first && region_pos + region_size > succ->first)
259  {
260  THRILL_THROW2(BadExternalAlloc, "disk_allocator::check_corruption", "Error: double deallocation of external memory, trying to deallocate region " << region_pos << " + " << region_size << " which overlaps empty space [" << succ->first << " + " << succ->second << "]");
261  }
262  }
263  if (succ == free_space.end())
264  {
265  if (pred == free_space.end())
266  {
267  DeallocationError(block_pos, block_size, pred, succ);
268  assert(pred != free_space.end());
269  }
270  if ((*pred).first + (*pred).second == region_pos)
271  {
272  // coalesce with predecessor
273  region_size += (*pred).second;
274  region_pos = (*pred).first;
275  free_space.erase(pred);
276  }
277  }
278  else
279  {

281  {
282 #if 0

286  assert(pred != succ);

288 #endif

290  if ((*succ).first == region_pos + region_size)

292  // coalesce with successor
293  region_size += (*succ).second;
294  free_space.erase(succ);
295  // -tb: set succ to pred afterwards due to iterator invalidation
296  succ = pred;
297  }

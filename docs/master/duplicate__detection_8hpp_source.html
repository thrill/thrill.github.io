32 /*!
33  * Duplicate detection to identify all elements occuring only on one worker.
34  * This information can be used to locally reduce uniquely-occuring elements.
35  * Therefore this saves communication volume in operations such as api::Reduce()
36  * or api::Join().
37  *
38  * Internally, this duplicate detection uses a golomb encoded distributed single
39  * shot bloom filter to find duplicates and non-duplicates with as low
40  * communication volume as possible. Due to the bloom filter's inherent
41  * properties, this has false duplicates but no false non-duplicates.
42  *
43  * Should only be used when a large amount of uniquely-occuring elements are
44  * expected.
45  */
46 class DuplicateDetection
47 {
48  static constexpr bool debug = false;
49 
50 private:
51  using GolombBitStreamWriter =
52  core::GolombBitStreamWriter<data::CatStream::Writer>;
53 
54  using GolombBitStreamReader =
55  core::GolombBitStreamReader<data::CatStream::Reader>;
56 
57  using GolumbDeltaWriter =
58  core::DeltaStreamWriter<GolombBitStreamWriter, size_t, /* offset */ 1>;
59 
60  using GolumbDeltaReader =
61  core::DeltaStreamReader<GolombBitStreamReader, size_t, /* offset */ 1>;
62 
63  /*!
64  * Sends all hashes in the range
65  * [max_hash / num_workers * p, max_hash / num_workers * (p + 1)) to worker
66  * p. These hashes are encoded with a Golomb encoder in core.
67  *
68  * \param stream_pointer Pointer to data stream
69  * \param hashes Sorted vector of all hashes modulo max_hash
70  * \param golomb_param Golomb parameter
71  * \param num_workers Number of workers in this Thrill process
72  * \param max_hash Modulo for all hashes
73  */
74  void WriteEncodedHashes(const data::CatStreamPtr& stream_pointer,
75  const std::vector<size_t>& hashes,
76  size_t golomb_param,
77  size_t num_workers,
78  size_t max_hash) {
79 
80  std::vector<data::CatStream::Writer> writers =
81  stream_pointer->GetWriters();
82 
83  size_t prev_hash = size_t(-1);
84 
85  for (size_t i = 0, j = 0; i < num_workers; ++i) {
86  common::Range range_i =
87  common::CalculateLocalRange(max_hash, num_workers, i);
88 
89  GolombBitStreamWriter golomb_writer(writers[i], golomb_param);
90  GolumbDeltaWriter delta_writer(
91  golomb_writer,
92  /* initial */ size_t(-1) /* cancels with +1 bias */);
93 
94  for ( /* j is already set from previous workers */
95  ; j < hashes.size() && hashes[j] < range_i.end; ++j) {
96  // Send hash deltas to make the encoded bitset smaller.
97  if (hashes[j] == prev_hash)
98  continue;
99  delta_writer.Put(hashes[j]);
100  prev_hash = hashes[j];
101  }
102  }
103  }
104 
105  /*!
106  * Reads a golomb encoded bitset from a data stream and returns it's
107  * contents in form of a vector of hashes.
108  *
109  * \param stream_pointer Pointer to data stream
110  * \param non_duplicates Target vector for hashes, should be empty beforehand
111  * \param golomb_param Golomb parameter
112  */
113  void ReadEncodedHashesToVector(const data::CatStreamPtr& stream_pointer,
114  std::vector<bool>& non_duplicates,
115  size_t golomb_param) {
116 
117  std::vector<data::CatStream::Reader> readers =
118  stream_pointer->GetReaders();
119 
120  for (data::CatStream::Reader& reader : readers)
121  {
122  GolombBitStreamReader golomb_reader(reader, golomb_param);
123  GolumbDeltaReader delta_reader(
124  golomb_reader, /* initial */ size_t(-1) /* cancels at +1 */);
125 
126  // Builds golomb encoded bitset from data received by the stream.
127  while (delta_reader.HasNext()) {
128  // Golomb code contains deltas, we want the actual values
129  size_t hash = delta_reader.Next<size_t>();
130  assert(hash < non_duplicates.size());
131  non_duplicates[hash] = true;
132  }
133  }
134  }

136 public:
137  /*!
138  * Identifies all hashes which occur on only a single worker.
139  * Returns all local uniques in form of a vector of hashes.
140  *
141  * \param non_duplicates Empty vector, which contains all non-duplicate
142  * hashes after this method
143  * \param hashes Hashes for all elements on this worker.
144  * \param context Thrill context, used for collective communication
145  * \param dia_id Id of the operation, which calls this method. Used
146  * to uniquely identify the data streams used.
147  *
148  * \return Modulo used on all hashes. (Use this modulo on all hashes to
149  * identify possible non-duplicates)
150  */
151  size_t FindNonDuplicates(std::vector<bool>& non_duplicates,
152  std::vector<size_t>& hashes,
153  Context& context,
154  size_t dia_id) {
155 
156  // This bound could often be lowered when we have many duplicates.
157  // This would however require a large amount of added communication.
158  size_t upper_bound_uniques = context.net.AllReduce(hashes.size());
159 
160  // Golomb Parameters taken from original paper (Sanders, Schlag, MÃ¼ller)
161 
162  // Parameter for false positive rate (FPR: 1/fpr_parameter)
163  double fpr_parameter = 8;
164  size_t golomb_param = (size_t)fpr_parameter; // (size_t)(std::log(2) * fpr_parameter);
165  size_t max_hash = upper_bound_uniques * fpr_parameter;
166 
167  for (size_t i = 0; i < hashes.size(); ++i) {
168  hashes[i] = hashes[i] % max_hash;
169  }
170 
171  std::sort(hashes.begin(), hashes.end());
172 
173  data::CatStreamPtr golomb_data_stream = context.GetNewCatStream(dia_id);
174 
175  WriteEncodedHashes(golomb_data_stream,
176  hashes, golomb_param,
177  context.num_workers(),
178  max_hash);
179 
180  // get inbound Golomb/delta-encoded hash stream
181 
182  std::vector<data::CatStream::Reader> readers =
183  golomb_data_stream->GetReaders();
184 
185  std::vector<GolombBitStreamReader> g_readers;
186  std::vector<GolumbDeltaReader> delta_readers;
187  g_readers.reserve(context.num_workers());
188  delta_readers.reserve(context.num_workers());
189 
190  for (auto& reader : readers) {
191  g_readers.emplace_back(reader, golomb_param);
192  delta_readers.emplace_back(
193  g_readers.back(),
194  /* initial */ size_t(-1) /* cancels with +1 bias */);
195  }
196 
197  // multi-way merge hash streams and detect duplicates/notify uniques
198 
199  auto puller = make_multiway_merge_tree<size_t>(
200  delta_readers.begin(), delta_readers.end());
201 
202  // create streams (delta/Golomb encoded) to notify workers of duplicates
203 
204  data::CatStreamPtr duplicates_stream = context.GetNewCatStream(dia_id);
205 
206  std::vector<data::CatStream::Writer> duplicates_writers =
207  duplicates_stream->GetWriters();
208 
209  std::vector<GolombBitStreamWriter> duplicates_gbsw;
210  std::vector<GolumbDeltaWriter> duplicates_dw;
211  duplicates_gbsw.reserve(context.num_workers());
212  duplicates_dw.reserve(context.num_workers());
213 
214  for (size_t i = 0; i < context.num_workers(); ++i) {
215  duplicates_gbsw.emplace_back(duplicates_writers[i], golomb_param);
216  duplicates_dw.emplace_back(
217  duplicates_gbsw.back(),
218  /* initial */ size_t(-1) /* cancels with +1 bias */);
219  }
220 
221  // find all keys only occuring on a single worker and insert to
222  // according bitset
223 
224  if (puller.HasNext())
225  {
226  std::pair<size_t, size_t> this_item;
227  std::pair<size_t, size_t> next_item = puller.NextWithSource();
228 
229  while (puller.HasNext())
230  {
231  this_item = next_item;
232  next_item = puller.NextWithSource();
233 
234  if (this_item.first != next_item.first) {
235  // this_item is a unique
236  sLOG << "!" << this_item.first << "->" << this_item.second;
237  duplicates_dw[this_item.second].Put(this_item.first);
238  }
239  else {
240  // this_item is a duplicate with next_item
241  sLOG << "=" << this_item.first << "-" << this_item.second;
242 
243  // read more items into next_item until the key mismatches
244  while (puller.HasNext() &&
245  (next_item = puller.NextWithSource(),
246  next_item.first == this_item.first))
247  {
248  sLOG << "." << next_item.first << "-" << next_item.second;
249  }
250  }
251  }
252 
253  if (this_item.first != next_item.first) {
254  // last item (next_item) is a unique
255  sLOG << "!" << next_item.first << "->" << next_item.second;
256  duplicates_dw[next_item.second].Put(next_item.first);
257  }
258  }
259 
260  // close duplicate delta writers
261  duplicates_dw.clear();
262  duplicates_gbsw.clear();
263  duplicates_writers.clear();
264 
265  // read inbound duplicate hash bits into non_duplicates hash table
266  assert(non_duplicates.size() == 0);
267  non_duplicates.resize(max_hash);
268  ReadEncodedHashesToVector(
269  duplicates_stream, non_duplicates, golomb_param);
270 
271  return max_hash;
272  }
273 };

40 template <typename StringSet, typename Traits>
41 class StringSetBase
42 {
43 public:
44  //! \name CharIterator Comparisons
45  //! \{
46 
47  //! check equality of two strings a and b at char iterators ai and bi.
48  bool is_equal(const typename Traits::String& a,
49  const typename Traits::CharIterator& ai,
50  const typename Traits::String& b,
51  const typename Traits::CharIterator& bi) const {
52  const StringSet& ss = *static_cast<const StringSet*>(this);
53  return !ss.is_end(a, ai) && !ss.is_end(b, bi) && (*ai == *bi);
54  }
55 
56  //! check if string a is less or equal to string b at iterators ai and bi.
57  bool is_leq(const typename Traits::String& a,
58  const typename Traits::CharIterator& ai,
59  const typename Traits::String& b,
60  const typename Traits::CharIterator& bi) const {
61  const StringSet& ss = *static_cast<const StringSet*>(this);
62  return ss.is_end(a, ai) ||
63  (!ss.is_end(a, ai) && !ss.is_end(b, bi) && *ai <= *bi);
64  }
65 
66  //! \}
67 
68  //! \name Character Extractors
69  //! \{
70 
71  typename Traits::Char
72  get_char(const typename Traits::String& s, size_t depth) const {
73  const StringSet& ss = *static_cast<const StringSet*>(this);
74  return *ss.get_chars(s, depth);
75  }
76 
77  //! Return up to 1 characters of string s at iterator i packed into a
78  //! uint8_t (only works correctly for 8-bit characters)
79  uint8_t get_uint8(
80  const typename Traits::String& s, typename Traits::CharIterator i) const {
81  const StringSet& ss = *static_cast<const StringSet*>(this);
82 
83  if (ss.is_end(s, i)) return 0;
84  return uint8_t(*i);
85  }
86 
87  //! Return up to 2 characters of string s at iterator i packed into a
88  //! uint16_t (only works correctly for 8-bit characters)
89  uint16_t get_uint16(
90  const typename Traits::String& s, typename Traits::CharIterator i) const {
91  const StringSet& ss = *static_cast<const StringSet*>(this);
92 
93  uint16_t v = 0;
94  if (ss.is_end(s, i)) return v;
95  v = (uint16_t(*i) << 8);
96  ++i;
97  if (ss.is_end(s, i)) return v;
98  v |= (uint16_t(*i) << 0);
99  return v;
100  }
101 
102  uint8_t get_uint8(const typename Traits::String& s, size_t depth) const {

104  return get_uint8(s, ss.get_chars(s, depth));
105  }

113

115  StringSet subi(size_t begin, size_t end) const {

120  bool check_order(const typename Traits::String& s1,
121  const typename Traits::String& s2) const {
122  const StringSet& ss = *static_cast<const StringSet*>(this);
123 
124  typename StringSet::CharIterator c1 = ss.get_chars(s1, 0);
125  typename StringSet::CharIterator c2 = ss.get_chars(s2, 0);
126 
127  while (ss.is_equal(s1, c1, s2, c2))
128  ++c1, ++c2;
129 
130  return ss.is_leq(s1, c1, s2, c2);
131  }
132 
133  bool check_order() const {
134  const StringSet& ss = *static_cast<const StringSet*>(this);
135 
136  for (typename Traits::Iterator pi = ss.begin() + 1;
137  pi != ss.end(); ++pi)
138  {
139  if (!check_order(ss[pi - 1], ss[pi])) {
140  LOG1 << "check_order() failed at position " << pi - ss.begin();
141  return false;
142  }
143  }
144  return true;
145  }

42 struct Pool::Arena {
43  //! total size of this Arena
44  size_t total_size;
45  // next and prev pointers for free list.
46  Arena * next_arena, * prev_arena;
47  // left and right pointers for splay tree
48  Arena * left = nullptr, * right = nullptr;
49  //! first sentinel Slot which is never used for payload data
50  Slot head_slot;
51  // following here are actual data slots
52 
53  //! the number of available payload slots (excluding head_slot)
54  uint32_t num_slots() const {
55  return static_cast<uint32_t>(
56  (total_size - sizeof(Arena)) / sizeof(Slot));
57  }
58 
59  Slot * begin() { return &head_slot + 1; }
60  Slot * end() { return &head_slot + 1 + num_slots(); }
61  Slot * slot(size_t i) { return &head_slot + 1 + i; }
62 };

107 
108  if (debug) {
109  std::cout << "AllocateFreeArena()" << std::endl;
110  }
111 
112  // Allocate space for the new block
113  Arena* new_arena =
114  reinterpret_cast<Arena*>(bypass_malloc(arena_size));
115  if (!new_arena) {
116  if (!die_on_failure) return nullptr;
117  fprintf(stderr, "out-of-memory - mem::Pool cannot allocate a new Arena."
118  " size_=%zu\n", size_);
119  abort();
120  }
121  new_arena->total_size = arena_size;
122  new_arena->next_arena = free_arena_;
123  new_arena->prev_arena = nullptr;
124  if (free_arena_)
125  free_arena_->prev_arena = new_arena;
126  free_arena_ = new_arena;
127 
128  new_arena->head_slot.size = new_arena->num_slots();
129  new_arena->head_slot.next = 0;
130 
131  new_arena->slot(0)->size = new_arena->num_slots();
132  new_arena->slot(0)->next = new_arena->num_slots();
133 
134  free_ += new_arena->num_slots();
135 
136  root_arena_ = common::splay(new_arena, root_arena_, ArenaCompare());

165 void* Pool::allocate(size_t bytes) {
166  return malloc(bytes);
167 
168  std::unique_lock<std::mutex> lock(mutex_);
169 
170  if (debug) {
171  std::cout << "allocate() bytes=" << bytes
172  << std::endl;
173  }
174 
175  // round up to whole slot size, and divide by slot size
176  uint32_t n =
177  static_cast<uint32_t>((bytes + sizeof(Slot) - 1) / sizeof(Slot));
178 
179  // check whether n is too large for allocation in our default Arenas, then
180  // allocate a special larger one.
181  if (n * sizeof(Slot) > bytes_per_arena(default_arena_size_)) {
182  if (debug) {
183  std::cout << "Allocate larger Arena of size "
184  << n * sizeof(Slot) << std::endl;
185  }
186  AllocateFreeArena(sizeof(Arena) + n * sizeof(Slot));
187  }
188 
189  Arena* curr_arena = free_arena_;
190 
191  if (curr_arena == nullptr || free_ < n)
192  curr_arena = AllocateFreeArena(default_arena_size_);
193 
194  while (curr_arena != nullptr)
195  {
196  // find an arena with at least n free slots
197  if (curr_arena->head_slot.size >= n)
198  {
199  // iterative over free areas to find a possible fit
200  Slot* prev_slot = &curr_arena->head_slot;
201  Slot* curr_slot = curr_arena->begin() + prev_slot->next;
202 
203  while (curr_slot != curr_arena->end() && curr_slot->size < n) {
204  prev_slot = curr_slot;
205  curr_slot = curr_arena->begin() + curr_slot->next;
206  }
207 
208  // if curr_slot == end then no continuous area was found.
209  if (curr_slot != curr_arena->end())
210  {
211  curr_arena->head_slot.size -= n;
212 
213  prev_slot->next += n;
214  size_ += n;
215  free_ -= n;
216 
217  // allocate more sparse memory
218  while (free_ < min_free_) {
219  if (!AllocateFreeArena(default_arena_size_, false)) break;
220  }
221 
222  if (curr_slot->size > n) {
223  // splits free area, since it is larger than needed
224  Slot* next_slot = curr_arena->begin() + prev_slot->next;
225  assert(next_slot != curr_arena->end());
226 
227  next_slot->size = curr_slot->size - n;
228  next_slot->next = curr_slot->next;
229  }
230  else {
231  // join used areas
232  prev_slot->next = curr_slot->next;
233  }
234 
235  print();
236 
237  if (debug_check_pairing) {
238  size_t i;
239  for (i = 0; i < allocs_.size(); ++i) {
240  if (allocs_[i].first == nullptr) {
241  allocs_[i].first = curr_slot;
242  allocs_[i].second = bytes;
243  break;
244  }
245  }
246  if (i == allocs_.size()) {
247  assert(!"Increase allocs array in Pool().");
248  abort();
249  }
250  }
251 
252  return reinterpret_cast<void*>(curr_slot);
253  }
254  }
255 
256  // advance to next arena in free list order
257  if (debug && 0) {
258  std::cout << "advance next arena"
259  << " curr_arena=" << curr_arena
260  << " next_arena=" << curr_arena->next_arena
261  << std::endl;
262  }
263 
264  curr_arena = curr_arena->next_arena;
265 
266  if (curr_arena == nullptr)
267  curr_arena = AllocateFreeArena(default_arena_size_);
268  }
269  abort();
270 }

272 void Pool::deallocate(void* ptr, size_t bytes) {
273  return free(ptr);
274 
275  if (ptr == nullptr) return;
276 
277  std::unique_lock<std::mutex> lock(mutex_);
278  if (debug) {
279  std::cout << "deallocate() ptr" << ptr << "bytes" << bytes << std::endl;
280  }
281  if (debug_check_pairing) {
282  size_t i;
283  for (i = 0; i < allocs_.size(); ++i) {
284  if (allocs_[i].first != ptr) continue;
285  if (bytes != allocs_[i].second) {
286  assert(!"Mismatching deallocate() size in Pool().");
287  abort();
288  }
289  allocs_[i].first = nullptr;
290  allocs_[i].second = 0;
291  break;
292  }
293  if (i == allocs_.size()) {
294  assert(!"Unknown deallocate() in Pool().");
295  abort();
296  }
297  }
298 
299  // round up to whole slot size, and divide by slot size
300  uint32_t n
301  = static_cast<uint32_t>((bytes + sizeof(Slot) - 1) / sizeof(Slot));
302  assert(n <= size_);
303 
304  // splay arenas to find arena containing ptr
305  root_arena_ = common::splay(ptr, root_arena_, ArenaCompare());
306 
307  if (!(ptr >= root_arena_->begin() && ptr < root_arena_->end())) {
308  assert(!"deallocate() of memory not in any arena.");
309  abort();
310  }
311 
312  Slot* prev_slot = &root_arena_->head_slot;
313  Slot* ptr_slot = reinterpret_cast<Slot*>(ptr);
314 
315  // advance prev_slot until the next jumps over ptr.
316  while (root_arena_->begin() + prev_slot->next < ptr_slot) {
317  prev_slot = root_arena_->begin() + prev_slot->next;
318  }
319 
320  // fill deallocated slot with free information
321  ptr_slot->next = prev_slot->next;
322  ptr_slot->size = n;
323 
324  prev_slot->next = static_cast<uint32_t>(ptr_slot - root_arena_->begin());
325 
326  // defragment free slots, but exempt the head_slot
327  if (prev_slot == &root_arena_->head_slot)
328  prev_slot = root_arena_->begin() + root_arena_->head_slot.next;
329 
330  while (prev_slot->next != root_arena_->num_slots() &&
331  prev_slot->next == prev_slot - root_arena_->begin() + prev_slot->size)
332  {
333  // join free slots
334  Slot* next_slot = root_arena_->begin() + prev_slot->next;
335  prev_slot->size += next_slot->size;
336  prev_slot->next = next_slot->next;
337  }
338 
339  root_arena_->head_slot.size += n;
340  size_ -= n;
341  free_ += n;
342 
343  if (root_arena_->head_slot.size == root_arena_->num_slots() &&
344  free_ >= min_free_ + root_arena_->num_slots())
345  {
346  // remove current arena from tree
347  Arena* root = common::splay_erase_top(root_arena_, ArenaCompare());
348 
349  // splice current arena from free list
350  if (root->prev_arena)
351  root->prev_arena->next_arena = root->next_arena;
352  if (root->next_arena)
353  root->next_arena->prev_arena = root->prev_arena;
354  if (free_arena_ == root)
355  free_arena_ = root->next_arena;
356 
357  free_ -= root->num_slots();
358  bypass_free(root, root->total_size);
359  }
360 
361  print();
362 }

364 void Pool::print() {
365  if (!debug_verify) return;
366 
367  if (debug) {
368  std::cout << "Pool::print()"
369  << " size_=" << size_ << " free_=" << free_ << std::endl;
370  }
371 
372  size_t total_free = 0, total_size = 0;
373 
374  for (Arena* curr_arena = free_arena_; curr_arena != nullptr;
375  curr_arena = curr_arena->next_arena)
376  {
377  std::ostringstream oss;
378 
379  size_t slot = curr_arena->head_slot.next;
380  size_t size = 0, free = 0;
381 
382  // used area at beginning
383  size += slot;
384 
385  while (slot != curr_arena->num_slots()) {
386  if (debug)
387  oss << " slot[" << slot
388  << ",size=" << curr_arena->slot(slot)->size
389  << ",next=" << curr_arena->slot(slot)->next << ']';
390 
391  if (curr_arena->slot(slot)->next <= slot) {
392  std::cout << "invalid chain:" << oss.str() << std::endl;
393  abort();
394  }
395 
396  free += curr_arena->slot(slot)->size;
397  size += curr_arena->slot(slot)->next - slot - curr_arena->slot(slot)->size;
398  slot = curr_arena->slot(slot)->next;
399  }
400 
401  if (debug) {
402  std::cout << "arena[" << curr_arena << "]"
403  << " head_slot.(free)size=" << curr_arena->head_slot.size
404  << " head_slot.next=" << curr_arena->head_slot.next
405  << oss.str()
406  << std::endl;
407  }
408 
409  assert(curr_arena->head_slot.size == free);
410 
411  total_free += free;
412  total_size += size;
413 
414  if (curr_arena->next_arena)
415  assert(curr_arena->next_arena->prev_arena == curr_arena);
416  if (curr_arena->prev_arena)
417  assert(curr_arena->prev_arena->next_arena == curr_arena);
418  }
419  assert(total_size == size_);
420  assert(total_free == free_);
421 }

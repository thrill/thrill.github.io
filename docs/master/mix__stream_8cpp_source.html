28  local_worker_id, dia_id) {
29 
30  sinks_.reserve(num_workers());

32 
33  // construct StreamSink array
34  for (size_t host = 0; host < num_hosts(); ++host) {

72 std::vector<MixStream::Writer> MixStream::GetWriters() {
73  size_t hard_ram_limit = multiplexer_.block_pool_.hard_ram_limit();
74  size_t block_size_base = hard_ram_limit / 16 / multiplexer_.num_workers();
75  size_t block_size = common::RoundDownToPowerOfTwo(block_size_base);
76  if (block_size == 0 || block_size > default_block_size)
77  block_size = default_block_size;
78 
79  {
80  std::unique_lock<std::mutex> lock(multiplexer_.mutex_);
81  multiplexer_.active_streams_++;
82  multiplexer_.max_active_streams_ =
83  std::max(multiplexer_.max_active_streams_,
84  multiplexer_.active_streams_.load());
85  }
86 
87  LOG << "MixStream::GetWriters()"
88  << " hard_ram_limit=" << hard_ram_limit
89  << " block_size_base=" << block_size_base
90  << " block_size=" << block_size
91  << " active_streams=" << multiplexer_.active_streams_
92  << " max_active_streams=" << multiplexer_.max_active_streams_;
93 
94  tx_timespan_.StartEventually();
95 
96  std::vector<Writer> result;
97  result.reserve(num_workers());
98 
99  for (size_t host = 0; host < num_hosts(); ++host) {
100  for (size_t worker = 0; worker < workers_per_host(); ++worker) {
101  if (host == my_host_rank()) {
102  // construct loopback queue writer
103  auto target_queue_ptr =
104  multiplexer_.MixLoopback(id_, local_worker_id_, worker);
105  result.emplace_back(target_queue_ptr, block_size);
106  target_queue_ptr->set_src_mix_stream(this);
107  }
108  else {
109  size_t worker_id = host * workers_per_host() + worker;
110  result.emplace_back(&sinks_[worker_id], block_size);
111  }
112  }
113  }
114 
115  assert(result.size() == num_workers());
116  return result;
117 }

128 void MixStream::Close() {
129  if (is_closed_) return;
130  is_closed_ = true;
131 
132  // close all sinks, this should emit sentinel to all other worker.
133  for (size_t i = 0; i != sinks_.size(); ++i) {
134  if (sinks_[i].closed()) continue;
135  sinks_[i].Close();
136  }
137 
138  // close loop-back queue from this worker to all others on this host.
139  for (size_t worker = 0;
140  worker < multiplexer_.workers_per_host(); ++worker)
141  {
142  auto queue_ptr = multiplexer_.MixLoopback(
143  id_, local_worker_id_, worker);
144 
145  if (!queue_ptr->write_closed())
146  queue_ptr->Close();
147  }
148 
149  // wait for all close packets to arrive.
150  for (size_t i = 0; i < (num_hosts() - 1) * workers_per_host(); ++i) {
151  LOG << "MixStream::Close() wait for closing block"
152  << " local_worker_id_=" << local_worker_id_;
153  sem_closing_blocks_.wait();
154  }
155 
156  multiplexer_.active_streams_--;
157 
158  tx_lifetime_.StopEventually();
159  tx_timespan_.StopEventually();
160  OnAllClosed("MixStream");
161 
162  LOG << "MixStream::Close() finished"
163  << " id_=" << id_
164  << " local_worker_id_=" << local_worker_id_;
165 }

74 std::vector<MixStream::Writer> MixStream::GetWriters() {
75  size_t hard_ram_limit = multiplexer_.block_pool_.hard_ram_limit();
76  size_t block_size_base = hard_ram_limit / 16 / multiplexer_.num_workers();
77  size_t block_size = tlx::round_down_to_power_of_two(block_size_base);
78  if (block_size == 0 || block_size > default_block_size)
79  block_size = default_block_size;
80 
81  {
82  std::unique_lock<std::mutex> lock(multiplexer_.mutex_);
83  multiplexer_.active_streams_++;
84  multiplexer_.max_active_streams_ =
85  std::max(multiplexer_.max_active_streams_,
86  multiplexer_.active_streams_.load());
87  }
88 
89  LOG << "MixStream::GetWriters()"
90  << " hard_ram_limit=" << hard_ram_limit
91  << " block_size_base=" << block_size_base
92  << " block_size=" << block_size
93  << " active_streams=" << multiplexer_.active_streams_
94  << " max_active_streams=" << multiplexer_.max_active_streams_;
95 
96  tx_timespan_.StartEventually();
97 
98  std::vector<Writer> result;
99  result.reserve(num_workers());
100 
101  for (size_t host = 0; host < num_hosts(); ++host) {
102  for (size_t worker = 0; worker < workers_per_host(); ++worker) {
103  if (host == my_host_rank()) {
104  // construct loopback queue writer
105  auto target_queue_ptr =
106  multiplexer_.MixLoopback(id_, local_worker_id_, worker);
107  result.emplace_back(target_queue_ptr, block_size);
108  target_queue_ptr->set_src_mix_stream(this);
109  }
110  else {
111  size_t worker_id = host * workers_per_host() + worker;
112  result.emplace_back(&sinks_[worker_id], block_size);
113  }
114  }
115  }
116 
117  assert(result.size() == num_workers());
118  return result;
119 }

130 void MixStream::Close() {
131  if (is_closed_) return;
132  is_closed_ = true;
133 
134  // close all sinks, this should emit sentinel to all other worker.
135  for (size_t i = 0; i != sinks_.size(); ++i) {
136  if (sinks_[i].closed()) continue;
137  sinks_[i].Close();
138  }
139 
140  // close loop-back queue from this worker to all others on this host.
141  for (size_t worker = 0;
142  worker < multiplexer_.workers_per_host(); ++worker)
143  {
144  auto queue_ptr = multiplexer_.MixLoopback(
145  id_, local_worker_id_, worker);
146 
147  if (!queue_ptr->write_closed())
148  queue_ptr->Close();
149  }
150 
151  // wait for all close packets to arrive.
152  for (size_t i = 0; i < (num_hosts() - 1) * workers_per_host(); ++i) {
153  LOG << "MixStream::Close() wait for closing block"
154  << " local_worker_id_=" << local_worker_id_;
155  sem_closing_blocks_.wait();
156  }
157 
158  multiplexer_.active_streams_--;
159 
160  tx_lifetime_.StopEventually();
161  tx_timespan_.StopEventually();
162  OnAllClosed("MixStream");
163 
164  LOG << "MixStream::Close() finished"
165  << " id_=" << id_
166  << " local_worker_id_=" << local_worker_id_;
167 }

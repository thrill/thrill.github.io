85 void ThreadPool::worker() {
86  // lock mutex, it is released during condition waits
87  std::unique_lock<std::mutex> lock(mutex_);
88 
89  while (true) {
90  // wait on condition variable until job arrives, frees lock
91  if (!terminate_ && jobs_.empty()) {
92  ++idle_;
93  cv_jobs_.wait(
94  lock, [this]() { return terminate_ || !jobs_.empty(); });
95  --idle_;
96  }
97 
98  if (terminate_)
99  break;
100 
101  if (!jobs_.empty()) {
102  // got work. set busy.
103  ++busy_;
104 
105  {
106  // pull job.
107  Job job = std::move(jobs_.front());
108  jobs_.pop_front();
109 
110  // release lock.
111  lock.unlock();
112 
113  // execute job.
114  try {
115  job();
116  }
117  catch (std::exception& e) {
118  std::cerr << "EXCEPTION: " << e.what() << std::endl;
119  }
120  // destroy job by closing scope
121  }
122 
123  // release memory the Job changed
124  std::atomic_thread_fence(std::memory_order_seq_cst);
125 
126  ++done_;
127  --busy_;
128 
129  // relock mutex before signaling condition.
130  lock.lock();
131  cv_finished_.notify_one();
132  }
133  }

42  /*!
43  * Initializes this Manager and initializes all Groups. When this method
44  * returns, the network system is ready to use.
45  *
46  * \param my_rank_ The rank of the worker that owns this Manager.
47  * \param endpoints The ordered list of all endpoints, including the local worker,
48  * where the endpoint at position i corresponds to the worker with id i.
49  */

52 
53  this->my_rank_ = my_rank_;

56  LOG << "Client " << my_rank_ << " starting: " << endpoints[my_rank_];

59  groups_[i] = std::make_unique<Group>(my_rank_, endpoints.size());
60  }
61 
62  // Parse endpoints.
63  std::vector<SocketAddress> address_list
64  = GetAddressList(endpoints);
65 
66  // Create listening socket.
67  {
68  Socket listen_socket = Socket::Create();
69  listen_socket.SetReuseAddr();
70 
71  SocketAddress& lsa = address_list[my_rank_];
72 
73  if (!listen_socket.bind(lsa))
74  throw Exception("Could not bind listen socket to "
75  + lsa.ToStringHostPort(), errno);
76 
77  if (!listen_socket.listen())
78  throw Exception("Could not listen on socket "
79  + lsa.ToStringHostPort(), errno);

87  for (uint32_t g = 0; g < group_count_; g++) {
88  for (size_t id = my_rank_ + 1; id < address_list.size(); ++id) {
89  AsyncConnect(g, id, address_list[id]);

94  dispatcher_.AddRead(listener_,
95  [=]() {
96  return OnIncomingConnection(listener_);

100  while (!IsInitializationFinished())

103  dispatcher_.Dispatch();
104  }
105

107  listener_.Close();
108 
109  LOG << "Client " << my_rank_ << " done";

112  // output list of file descriptors connected to partners
113  for (size_t i = 0; i != address_list.size(); ++i) {
114  if (i == my_rank_) continue;
115  LOG << "Group " << j
116  << " link " << my_rank_ << " -> " << i << " = fd "
117  << groups_[j]->tcp_connection(i).GetSocket().fd();
118  }
119  }

224  /*!
225  * Starts connecting to the net connection specified. Starts connecting to
226  * the endpoint specified by the parameters. This method executes
227  * asynchronously.
228  *
229  * \param nc The connection to connect.
230  * \param address The address of the endpoint to connect to.
231  */
232  void AsyncConnect(net::Connection& nc, const SocketAddress& address) {
233  assert(dynamic_cast<Connection*>(&nc));
234  Connection& tcp = static_cast<Connection&>(nc);
235 
236  // Start asynchronous connect.
237  tcp.GetSocket().SetNonBlocking(true);
238  int res = tcp.GetSocket().connect(address);
239 
240  tcp.set_state(ConnectionState::Connecting);
241 
242  if (res == 0) {
243  LOG << "Early connect success. This should not happen.";
244  // connect() already successful? this should not be.
245  OnConnected(tcp, address);
246  }
247  else if (errno == EINPROGRESS) {
248  // connect is in progress, will wait for completion.
249  dispatcher_.AddWrite(tcp, [this, &address, &tcp]() {
250  return OnConnected(tcp, address);
251  });
252  }
253  else if (errno == ECONNREFUSED) {
254  LOG << "Early connect refused.";
255  // connect() already refused connection?
256  OnConnected(tcp, address, errno);
257  }
258  else {
259  // Failed to even try the connection - this might be a permanent
260  // error.
261  tcp.set_state(ConnectionState::Invalid);
262 
263  throw Exception("Error starting async connect client "
264  + std::to_string(tcp.peer_id()) + " via "
265  + address.ToStringHostPort(), errno);
266  }
267  }

312  //! calculate the next timeout on connect() errors
313  size_t NextConnectTimeout(size_t group, size_t id,
314  const SocketAddress& address) {
315  GroupNodeIdPair gnip(group, id);
316  auto it = timeouts_.find(gnip);
317  if (it == timeouts_.end()) {
318  it = timeouts_.insert(std::make_pair(gnip, initial_timeout_)).first;
319  }
320  else {
321  // exponential backoff of reconnects.
322  it->second = 2 * it->second;
323 
324  if (it->second >= final_timeout_) {
325  throw Exception("Timeout error connecting to client "
326  + std::to_string(id) + " via "
327  + address.ToStringHostPort());
328  }
329  }
330  return it->second;
331  }

333  /*!
334  * Called when a connection initiated by us succeeds on a network level.
335  * The Thrill welcome messages still have to be exchanged.
336  *
337  * \param conn The connection that was connected successfully.
338  *
339  * \param address The associated address. This parameter is needed in case
340  * we need to reconnect.
341  *
342  * \param _err An errno state if called synchronously after a connect().
343  *
344  * \return A bool indicating wether this callback should stay registered.
345  */
346  bool OnConnected(net::Connection& conn, const SocketAddress& address,
347  int _err = 0) {
348  assert(dynamic_cast<Connection*>(&conn));
349  Connection& tcp = static_cast<Connection&>(conn);
350 
351  // First, check if everything went well.
352  int err = _err ? _err : tcp.GetSocket().GetError();
353 
354  if (tcp.state() != ConnectionState::Connecting) {
355  LOG << "Client " << my_rank_
356  << " expected connection state " << ConnectionState::Connecting
357  << " but got " << tcp.state();
358  die("FAULTY STATE DETECTED");
359  }
360 
361  if (err == ECONNREFUSED || err == ETIMEDOUT) {
362 
363  // Connection refused. The other workers might not be online yet.
364 
365  size_t next_timeout = NextConnectTimeout(
366  tcp.group_id(), tcp.peer_id(), address);
367 
368  LOG << "Connect to " << address.ToStringHostPort()
369  << " fd=" << tcp.GetSocket().fd()
370  << " timed out or refused with error " << err << "."
371  << " Attempting reconnect in " << next_timeout << "msec";
372 
373  dispatcher_.AddTimer(
374  std::chrono::milliseconds(next_timeout),
375  [&]() {
376  // Construct a new connection since the socket might not be
377  // reusable.
378  AsyncConnect(tcp.group_id(), tcp.peer_id(), address);
379  return false;
380  });
381 
382  return false;
383  }
384  else if (err != 0) {
385  // Other failure. Fail hard.
386  tcp.set_state(ConnectionState::Invalid);
387 
388  throw Exception("Error connecting asynchronously to client "
389  + std::to_string(tcp.peer_id()) + " via "
390  + address.ToStringHostPort(), err);
391  }
392 
393  die_unless(tcp.GetSocket().IsValid());
394 
395  tcp.set_state(ConnectionState::TransportConnected);
396 
397  LOG << "OnConnected() " << my_rank_ << " connected"
398  << " fd=" << tcp.GetSocket().fd()
399  << " to=" << tcp.GetSocket().GetPeerAddress()
400  << " err=" << err
401  << " group=" << tcp.group_id();
402 
403  // send welcome message
404  const WelcomeMsg hello = { thrill_sign, tcp.group_id(), my_rank_ };
405 
406  dispatcher_.AsyncWriteCopy(
407  tcp, &hello, sizeof(hello),
408  AsyncWriteCallback::make<
409  Construction, &Construction::OnHelloSent>(this));
410 
411  LOG << "Client " << my_rank_ << " sent active hello to "
412  << "client " << tcp.peer_id() << " group id " << tcp.group_id();
413 
414  dispatcher_.AsyncRead(
415  tcp, sizeof(hello),
416  AsyncReadBufferCallback::make<
417  Construction, &Construction::OnIncomingWelcome>(this));
418 
419  return false;
420  }

433

441  // We already know those values since we connected actively. So, check
442  // for any errors.

444  LOG << "FAULTY ID DETECTED";

452  die_unequal(tcp.group_id(), msg->group_id);

527  dispatcher_.AsyncRead(

529  AsyncReadBufferCallback::make<

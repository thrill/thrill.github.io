536 ATTRIBUTE_NO_SANITIZE
537 static void * preinit_malloc(size_t size) noexcept {
538 
539  size_t aligned_size = size + (init_alignment - size % init_alignment);
540 
541 #if defined(_MSC_VER) || USE_ATOMICS
542  size_t offset = (init_heap_use += (padding + aligned_size));
543 #else
544  size_t offset = __sync_fetch_and_add(&init_heap_use, padding + aligned_size);
545 #endif
546 
547  if (offset > INIT_HEAP_SIZE) {
548  fprintf(stderr, PPREFIX "init heap full !!!\n");
549  exit(EXIT_FAILURE);
550  }
551 
552  char* ret = init_heap + offset;
553 
554  //! prepend allocation size and check sentinel
555  *reinterpret_cast<size_t*>(ret) = aligned_size;
556  *reinterpret_cast<size_t*>(ret + padding - sizeof(size_t)) = sentinel;
557 
558  inc_count(aligned_size);
559 
560  if (log_operations_init_heap) {
561  fprintf(stderr, PPREFIX "malloc(%zu / %zu) = %p on init heap\n",
562  size, aligned_size, static_cast<void*>(ret + padding));
563  }
564 
565  return ret + padding;
566 }

569 static void * preinit_realloc(void* ptr, size_t size) {
570 
571  if (log_operations_init_heap) {
572  fprintf(stderr, PPREFIX "realloc(%p) = on init heap\n", ptr);
573  }
574

576

642 ATTRIBUTE_NO_SANITIZE
643 static void * preinit_malloc(size_t size) noexcept {
644 
645  size_t aligned_size = size + (init_alignment - size % init_alignment);
646 
647 #if defined(_MSC_VER) || USE_ATOMICS
648  size_t offset = (init_heap_use += (padding + aligned_size));
649 #else
650  size_t offset = __sync_fetch_and_add(&init_heap_use, padding + aligned_size);
651 #endif
652 
653  if (offset > INIT_HEAP_SIZE) {
654  fprintf(stderr, PPREFIX "init heap full !!!\n");
655  exit(EXIT_FAILURE);
656  }
657 
658  char* ret = init_heap + offset;
659 
660  //! prepend allocation size and check sentinel
661  *reinterpret_cast<size_t*>(ret) = aligned_size;
662  *reinterpret_cast<size_t*>(ret + padding - sizeof(size_t)) = sentinel;
663 
664  inc_count(aligned_size);
665 
666  if (log_operations_init_heap) {
667  fprintf(stderr, PPREFIX "malloc(%zu / %zu) = %p on init heap\n",
668  size, aligned_size, static_cast<void*>(ret + padding));
669  }
670 
671  return ret + padding;
672 }

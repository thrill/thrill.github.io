45 DynBlockSource BlockQueue::GetBlockSource(bool consume, size_t local_worker_id) {
46  if (consume && !read_closed_) {
47  // set to consume, and BlockQueue has not been read.
48  sLOG << "BlockQueue::GetBlockSource() consume, from queue.";
49  return ConstructDynBlockSource<ConsumeBlockQueueSource>(
50  *this, local_worker_id);
51  }
52  else if (consume && read_closed_) {
53  // consume the File, BlockQueue was already read.
54  sLOG << "BlockQueue::GetBlockSource() consume, from cache:"
55  << file_.num_items();
56  return ConstructDynBlockSource<ConsumeFileBlockSource>(
57  &file_, local_worker_id);
58  }
59  else if (!consume && !read_closed_) {
60  // non-consumer but the BlockQueue has not been read.
61  sLOG << "BlockQueue::GetBlockSource() non-consume, from queue.";
62  return ConstructDynBlockSource<CacheBlockQueueSource>(
63  this, local_worker_id);
64  }
65  else if (!consume && read_closed_) {
66  // non-consumer: reread the file that was cached.
67  sLOG << "BlockQueue::GetBlockSource() non-consume, from cache:"
68  << file_.num_items();
69  return ConstructDynBlockSource<KeepFileBlockSource>(
70  file_, local_worker_id);
71  }
72  else {
73  // impossible
74  abort();
75  }
76 }

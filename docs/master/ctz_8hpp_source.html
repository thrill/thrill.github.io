28 //! ctz (count trailing zeros) - generic implementation
29 template <typename Integral>
30 static inline unsigned ctz_template(Integral x) {
31  if (x == 0) return 8 * sizeof(x);
32  unsigned r = 0;
33  while ((x & static_cast<Integral>(1)) == 0)
34  x >>= 1, ++r;
35  return r;
36 }

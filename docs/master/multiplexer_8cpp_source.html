31 
32 /*!
33  * A Repository holds obects that are shared among workers. Each object is
34  * addressd via and Id. Workers can allocate new Id independetly but
35  * deterministically (the repository will issue the same id sequence to all
36  * workers). Objects are created inplace via argument forwarding.
37  */
38 template <typename Object>
39 class Repository
40 {
41 public:
42  using Id = size_t;
43  using ObjectPtr = tlx::CountingPtr<Object>;
44 
45  //! construct with initial ids 0.
46  explicit Repository(size_t num_workers_per_node)
47  : next_id_(num_workers_per_node, 0) { }
48 
49  //! Alllocates the next data target.
50  //! Calls to this method alter the internal state -> order of calls is
51  //! important and must be deterministic
52  size_t AllocateId(size_t local_worker_id) {
53  assert(local_worker_id < next_id_.size());
54  return next_id_[local_worker_id]++;
55  }
56 
57  //! Get object with given id, if it does not exist, create it.
58  //! \param object_id of the object
59  //! \param construction parameters forwards to constructor
60  template <typename Subclass = Object, typename... Types>
61  tlx::CountingPtr<Subclass>
62  GetOrCreate(Id object_id, Types&& ... construction) {
63  auto it = map_.find(object_id);
64 
65  if (it != map_.end()) {
66  die_unless(dynamic_cast<Subclass*>(it->second.get()));
67  return tlx::CountingPtr<Subclass>(
68  dynamic_cast<Subclass*>(it->second.get()));
69  }
70 
71  // construct new object
72  tlx::CountingPtr<Subclass> value = tlx::make_counting<Subclass>(
73  std::forward<Types>(construction) ...);
74 
75  map_.insert(std::make_pair(object_id, ObjectPtr(value)));
76  return value;
77  }
78 
79  template <typename Subclass = Object>
80  tlx::CountingPtr<Subclass> GetOrDie(Id object_id) {
81  auto it = map_.find(object_id);
82 
83  if (it != map_.end()) {
84  die_unless(dynamic_cast<Subclass*>(it->second.get()));
85  return tlx::CountingPtr<Subclass>(
86  dynamic_cast<Subclass*>(it->second.get()));
87  }
88 
89  die("object " + std::to_string(object_id) + " not in repository");
90  }
91 
92  //! return mutable reference to map of objects.
93  std::map<Id, ObjectPtr>& map() { return map_; }
94 
95 private:
96  //! Next ID to generate, one for each local worker.
97  std::vector<size_t> next_id_;
98 
99  //! map containing value items
100  std::map<Id, ObjectPtr> map_;
101 };

225 void Multiplexer::OnMultiplexerHeader(Connection& s, net::Buffer&& buffer) {
226 
227  // received invalid Buffer: the connection has closed?
228  if (!buffer.IsValid()) return;
229 
230  net::BufferReader br(buffer);
231  StreamMultiplexerHeader header = StreamMultiplexerHeader::Parse(br);
232 
233  LOG << "OnMultiplexerHeader() header"
234  << " magic=" << unsigned(header.magic)
235  << " size=" << header.size
236  << " num_items=" << header.num_items
237  << " first_item=" << header.first_item
238  << " typecode_verify=" << header.typecode_verify
239  << " stream_id=" << header.stream_id;
240 
241  // received stream id
242  StreamId id = header.stream_id;
243  size_t local_worker = header.receiver_local_worker;
244 
245  // round of allocation size to next power of two
246  size_t alloc_size = header.size;
247  if (alloc_size < THRILL_DEFAULT_ALIGN) alloc_size = THRILL_DEFAULT_ALIGN;
248  alloc_size = tlx::round_up_to_power_of_two(alloc_size);
249 
250  if (header.magic == MagicByte::CatStreamBlock)
251  {
252  CatStreamPtr stream = GetOrCreateCatStream(
253  id, local_worker, /* dia_id (unknown at this time) */ 0);
254  stream->rx_net_bytes_ += buffer.size();
255 
256  if (header.IsEnd()) {
257  sLOG << "end of stream on" << s << "in CatStream" << id
258  << "from worker" << header.sender_worker;
259 
260  stream->OnCloseStream(header.sender_worker);
261 
262  AsyncReadMultiplexerHeader(s);
263  }
264  else {
265  sLOG << "stream header from" << s << "on CatStream" << id
266  << "from worker" << header.sender_worker
267  << "for local_worker" << local_worker;
268 
269  PinnedByteBlockPtr bytes = block_pool_.AllocateByteBlock(
270  alloc_size, local_worker);
271 
272  dispatcher_.AsyncRead(
273  s, header.size, std::move(bytes),
274  [this, header, stream](Connection& s, PinnedByteBlockPtr&& bytes) {
275  OnCatStreamBlock(s, header, stream, std::move(bytes));
276  });
277  }
278  }
279  else if (header.magic == MagicByte::MixStreamBlock)
280  {
281  MixStreamPtr stream = GetOrCreateMixStream(
282  id, local_worker, /* dia_id (unknown at this time) */ 0);
283  stream->rx_net_bytes_ += buffer.size();
284 
285  if (header.IsEnd()) {
286  sLOG << "end of stream on" << s << "in MixStream" << id
287  << "from worker" << header.sender_worker;
288 
289  stream->OnCloseStream(header.sender_worker);
290 
291  AsyncReadMultiplexerHeader(s);
292  }
293  else {
294  sLOG << "stream header from" << s << "on MixStream" << id
295  << "from worker" << header.sender_worker
296  << "for local_worker" << local_worker;
297 
298  PinnedByteBlockPtr bytes = block_pool_.AllocateByteBlock(
299  alloc_size, local_worker);
300 
301  dispatcher_.AsyncRead(
302  s, header.size, std::move(bytes),
303  [this, header, stream](Connection& s, PinnedByteBlockPtr&& bytes) mutable {
304  OnMixStreamBlock(s, header, stream, std::move(bytes));
305  });
306  }
307  }
308  else {
309  die("Invalid magic byte in MultiplexerHeader");
310  }
311 }

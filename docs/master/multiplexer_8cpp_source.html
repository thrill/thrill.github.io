30 /*!
31  * A Repository holds obects that are shared among workers. Each object is
32  * addressd via and Id. Workers can allocate new Id independetly but
33  * deterministically (the repository will issue the same id sequence to all
34  * workers). Objects are created inplace via argument forwarding.
35  */
36 template <typename Object>
37 class Repository
38 {
39 public:
40  using Id = size_t;
41  using ObjectPtr = common::CountingPtr<Object>;
42 
43  //! construct with initial ids 0.
44  explicit Repository(size_t num_workers_per_node)
45  : next_id_(num_workers_per_node, 0) { }
46 
47  //! Alllocates the next data target.
48  //! Calls to this method alter the internal state -> order of calls is
49  //! important and must be deterministic
50  size_t AllocateId(size_t local_worker_id) {
51  assert(local_worker_id < next_id_.size());
52  return next_id_[local_worker_id]++;
53  }
54 
55  //! Get object with given id, if it does not exist, create it.
56  //! \param object_id of the object
57  //! \param construction parameters forwards to constructor
58  template <typename Subclass = Object, typename ... Types>
59  common::CountingPtr<Subclass>
60  GetOrCreate(Id object_id, Types&& ... construction) {
61  auto it = map_.find(object_id);
62 
63  if (it != map_.end()) {
64  die_unless(dynamic_cast<Subclass*>(it->second.get()));
65  return common::CountingPtr<Subclass>(
66  dynamic_cast<Subclass*>(it->second.get()));
67  }
68 
69  // construct new object
70  common::CountingPtr<Subclass> value = common::MakeCounting<Subclass>(
71  std::forward<Types>(construction) ...);
72 
73  map_.insert(std::make_pair(object_id, ObjectPtr(value)));
74  return value;
75  }
76 
77  template <typename Subclass = Object>
78  common::CountingPtr<Subclass> GetOrDie(Id object_id) {
79  auto it = map_.find(object_id);
80 
81  if (it != map_.end()) {
82  die_unless(dynamic_cast<Subclass*>(it->second.get()));
83  return common::CountingPtr<Subclass>(
84  dynamic_cast<Subclass*>(it->second.get()));
85  }
86 
87  die("object " + std::to_string(object_id) + " not in repository");
88  }
89 
90  //! return mutable reference to map of objects.
91  std::map<Id, ObjectPtr>& map() { return map_; }
92 
93 private:
94  //! Next ID to generate, one for each local worker.
95  std::vector<size_t> next_id_;
96 
97  //! map containing value items
98  std::map<Id, ObjectPtr> map_;
99 };

112 Multiplexer::Multiplexer(mem::Manager& mem_manager,
113  BlockPool& block_pool,
114  size_t workers_per_host, net::Group& group)
115  : mem_manager_(mem_manager),
116  block_pool_(block_pool),
117  dispatcher_(
118  mem_manager, group,
119  "host " + mem::to_string(group.my_host_rank()) + " multiplexer"),
120  group_(group),
121  workers_per_host_(workers_per_host),
122  d_(std::make_unique<Data>(workers_per_host)) {
123  for (size_t id = 0; id < group_.num_hosts(); id++) {
124  if (id == group_.my_host_rank()) continue;
125  AsyncReadMultiplexerHeader(group_.connection(id));
126  }
127  (void)mem_manager_; // silence unused variable warning.
128 }

223 void Multiplexer::OnMultiplexerHeader(Connection& s, net::Buffer&& buffer) {
224 
225  // received invalid Buffer: the connection has closed?
226  if (!buffer.IsValid()) return;
227 
228  net::BufferReader br(buffer);
229  StreamMultiplexerHeader header = StreamMultiplexerHeader::Parse(br);
230 
231  LOG << "OnMultiplexerHeader() header"
232  << " magic=" << unsigned(header.magic)
233  << " size=" << header.size
234  << " num_items=" << header.num_items
235  << " first_item=" << header.first_item
236  << " typecode_verify=" << header.typecode_verify
237  << " stream_id=" << header.stream_id;
238 
239  // received stream id
240  StreamId id = header.stream_id;
241  size_t local_worker = header.receiver_local_worker;
242 
243  // round of allocation size to next power of two
244  size_t alloc_size = header.size;
245  if (alloc_size < THRILL_DEFAULT_ALIGN) alloc_size = THRILL_DEFAULT_ALIGN;
246  alloc_size = common::RoundUpToPowerOfTwo(alloc_size);
247 
248  if (header.magic == MagicByte::CatStreamBlock)
249  {
250  CatStreamPtr stream = GetOrCreateCatStream(
251  id, local_worker, /* dia_id (unknown at this time) */ 0);
252  stream->rx_net_bytes_ += buffer.size();
253 
254  if (header.IsEnd()) {
255  sLOG << "end of stream on" << s << "in CatStream" << id
256  << "from worker" << header.sender_worker;
257 
258  stream->OnCloseStream(header.sender_worker);
259 
260  AsyncReadMultiplexerHeader(s);
261  }
262  else {
263  sLOG << "stream header from" << s << "on CatStream" << id
264  << "from worker" << header.sender_worker
265  << "for local_worker" << local_worker;
266 
267  PinnedByteBlockPtr bytes = block_pool_.AllocateByteBlock(
268  alloc_size, local_worker);
269 
270  dispatcher_.AsyncRead(
271  s, header.size, std::move(bytes),
272  [this, header, stream](Connection& s, PinnedByteBlockPtr&& bytes) {
273  OnCatStreamBlock(s, header, stream, std::move(bytes));
274  });
275  }
276  }
277  else if (header.magic == MagicByte::MixStreamBlock)
278  {
279  MixStreamPtr stream = GetOrCreateMixStream(
280  id, local_worker, /* dia_id (unknown at this time) */ 0);
281  stream->rx_net_bytes_ += buffer.size();
282 
283  if (header.IsEnd()) {
284  sLOG << "end of stream on" << s << "in MixStream" << id
285  << "from worker" << header.sender_worker;
286 
287  stream->OnCloseStream(header.sender_worker);
288 
289  AsyncReadMultiplexerHeader(s);
290  }
291  else {
292  sLOG << "stream header from" << s << "on MixStream" << id
293  << "from worker" << header.sender_worker
294  << "for local_worker" << local_worker;
295 
296  PinnedByteBlockPtr bytes = block_pool_.AllocateByteBlock(
297  alloc_size, local_worker);
298 
299  dispatcher_.AsyncRead(
300  s, header.size, std::move(bytes),
301  [this, header, stream](Connection& s, PinnedByteBlockPtr&& bytes) mutable {
302  OnMixStreamBlock(s, header, stream, std::move(bytes));
303  });
304  }
305  }
306  else {
307  die("Invalid magic byte in MultiplexerHeader");
308  }
309 }

311 void Multiplexer::OnCatStreamBlock(
312  Connection& s, const StreamMultiplexerHeader& header,
313  const CatStreamPtr& stream, PinnedByteBlockPtr&& bytes) {
314 
315  sLOG << "Multiplexer::OnCatStreamBlock()"
316  << "got block on" << s
317  << "in CatStream" << header.stream_id
318  << "from worker" << header.sender_worker;
319 
320  stream->OnStreamBlock(
321  header.sender_worker,
322  PinnedBlock(std::move(bytes), 0, header.size,
323  header.first_item, header.num_items,
324  header.typecode_verify));
325 
326  if (header.is_last_block)
327  stream->OnCloseStream(header.sender_worker);
328 
329  AsyncReadMultiplexerHeader(s);
330 }

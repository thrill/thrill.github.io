80  /*!
81  * Scatters a File to many worker: elements from [offset[0],offset[1]) are
82  * sent to the first worker, elements from [offset[1], offset[2]) are sent
83  * to the second worker, ..., elements from [offset[my_rank -
84  * 1],offset[my_rank]) are copied locally, ..., elements from
85  * [offset[num_workers - 1], offset[num_workers]) are sent to the last
86  * worker.
87  *
88  * The number of given offsets must be equal to the
89  * net::Group::num_hosts() * workers_per_host_ + 1.
90  *
91  * /param source File containing the data to be scattered.
92  *
93  * /param offsets - as described above. offsets.size must be equal to
94  * num_workers + 1
95  */
96  template <typename ItemType>
97  void Scatter(File& source, const std::vector<size_t>& offsets,
98  bool consume = false) {
99  tx_timespan_.StartEventually();
100 
101  assert(offsets.size() == num_workers() + 1);
102 
103  File::Reader reader = source.GetReader(consume);
104  size_t current = 0;
105 
106  {
107  // discard first items in Reader
108  size_t limit = offsets[0];
109 #if 0
110  for ( ; current < limit; ++current) {
111  assert(reader.HasNext());
112  // discard one item (with deserialization)
113  reader.template Next<ItemType>();
114  }
115 #else
116  if (current != limit) {
117  reader.template GetItemBatch<ItemType>(limit - current);
118  current = limit;
119  }
120 #endif
121  }
122 
123  std::vector<Writer> writers = GetWriters();
124 
125  for (size_t worker = 0; worker < num_workers(); ++worker) {
126  // write [current,limit) to this worker

128  assert(current <= limit);
129 #if 0
130  for ( ; current < limit; ++current) {
131  assert(reader.HasNext());
132  // move over one item (with deserialization and serialization)
133  writers[worker](reader.template Next<ItemType>());
134  }
135 #else
136  if (current != limit) {
137  writers[worker].AppendBlocks(
138  reader.template GetItemBatch<ItemType>(limit - current));
139  current = limit;
140  }
141 #endif

143  }

146  }

228  for (size_t i = 0; i < workers_per_host; i++)

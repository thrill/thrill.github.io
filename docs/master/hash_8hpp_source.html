61 //! This is the Hash128to64 function from Google's cityhash (available under the
62 //! MIT License).
63 static inline uint64_t Hash128to64(const uint64_t upper, const uint64_t lower) {
64  // Murmur-inspired hashing.
65  const uint64_t k = 0x9DDFEA08EB382D69ull;
66  uint64_t a = (lower ^ upper) * k;
67  a ^= (a >> 47);
68  uint64_t b = (upper ^ a) * k;
69  b ^= (b >> 47);
70  b *= k;
71  return b;
72 }

74 /*!
75  * Returns a uint32_t hash of a uint64_t.
76  *
77  * This comes from http://www.concentric.net/~ttwang/tech/inthash.htm
78  *
79  * This hash gives no guarantees on the cryptographic suitability nor the
80  * quality of randomness produced, and the mapping may change in the future.
81  */
82 static inline uint32_t Hash64to32(uint64_t key) {
83  key = (~key) + (key << 18);
84  key = key ^ (key >> 31);
85  key = key * 21;
86  key = key ^ (key >> 11);
87  key = key + (key << 6);
88  key = key ^ (key >> 22);
89  return (uint32_t)key;
90 }

112 /**
113  * A CRC32C hasher using SSE4.2 intrinsics.
114  *
115  * Note that you need to provide specializations of HashDataSwitch if you want
116  * to hash types with heap storage.
117  */
118 template <typename ValueType>
119 struct HashCrc32Sse42 {
120  // Hash data with Intel's CRC32C instructions
121  // Copyright 2008,2009,2010 Massachusetts Institute of Technology.
122  // For constant sizes, this is neatly optimized away at higher optimization
123  // levels - only a mov (for initialization) and crc32 instructions remain
124  uint32_t hash_bytes(const void* data, size_t length, uint32_t crc = 0xffffffff) const {
125  const char* p_buf = (const char*)data;
126  // The 64-bit crc32 instruction returns a 64-bit value (even though a
127  // CRC32 hash has - well - 32 bits. Whatever.
128  uint64_t crc_carry = crc;
129  for (size_t i = 0; i < length / sizeof(uint64_t); i++) {
130  crc_carry = _mm_crc32_u64(crc_carry, *(const uint64_t*)p_buf);
131  p_buf += sizeof(uint64_t);
132  }
133  crc = (uint32_t)crc_carry; // discard the rest
134  length &= 7; // remaining length
135 
136  // ugly switch statement, faster than a loop-based version
137  switch (length) {
138  case 7:
139  crc = _mm_crc32_u8(crc, *p_buf++);
140  case 6:
141  crc = _mm_crc32_u16(crc, *(const uint16_t*)p_buf);
142  p_buf += 2;
143  // case 5 is below: 4 + 1
144  case 4:
145  crc = _mm_crc32_u32(crc, *(const uint32_t*)p_buf);
146  break;
147  case 3:
148  crc = _mm_crc32_u8(crc, *p_buf++);
149  case 2:
150  crc = _mm_crc32_u16(crc, *(const uint16_t*)p_buf);
151  break;
152  case 5:
153  crc = _mm_crc32_u32(crc, *(const uint32_t*)p_buf);
154  p_buf += 4;
155  case 1:
156  crc = _mm_crc32_u8(crc, *p_buf);
157  break;
158  case 0:
159  break;
160  default: // wat
161  assert(false);
162  }
163  return crc;
164  }

203 /*!
204  * Tabulation Hashing, see https://en.wikipedia.org/wiki/Tabulation_hashing
205  *
206  * Keeps a table with size * 256 entries of type hash_t, filled with random
207  * values. Elements are hashed by treating them as a vector of 'size' bytes,
208  * and XOR'ing the values in the data[i]-th position of the i-th table, with i
209  * ranging from 0 to size - 1.
210  */
211 
212 template <size_t size, typename hash_t = uint32_t,
213  typename prng_t = std::mt19937>
214 class TabulationHashing
215 {
216 public:
217  using hash_type = hash_t; // make public
218  using prng_type = prng_t;
219  using Subtable = std::array<hash_type, 256>;
220  using Table = std::array<Subtable, size>;
221 
222  explicit TabulationHashing(size_t seed = 0) { init(seed); }
223 
224  //! (re-)initialize the table by filling it with random values
225  void init(const size_t seed) {
226  prng_t rng { seed };
227  for (size_t i = 0; i < size; ++i) {
228  for (size_t j = 0; j < 256; ++j) {
229  table_[i][j] = rng();
230  }
231  }
232  }
233 
234  //! Hash an element
235  template <typename T>
236  hash_type operator () (const T& x) const {
237  static_assert(sizeof(T) == size, "Size mismatch with operand type");
238 
239  hash_t hash = 0;
240  const uint8_t* ptr = reinterpret_cast<const uint8_t*>(&x);
241  for (size_t i = 0; i < size; ++i) {
242  hash ^= table_[i][*(ptr + i)];
243  }
244  return hash;
245  }
246 
247 protected:
248  Table table_;
249 };

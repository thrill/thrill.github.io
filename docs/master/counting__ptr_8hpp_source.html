33 /*!
34  * High-performance smart pointer used as a wrapping reference counting pointer.
35  *
36  * This smart pointer class requires two functions in the template type: void
37  * inc_reference() and void dec_reference(). These must increment and decrement
38  * a reference count inside the templated object. When initialized, the type
39  * must have reference count zero. Each new object referencing the data calls
40  * inc_reference() and each destroying holder calls del_reference(). When the
41  * data object determines that it's internal count is zero, then it must destroy
42  * itself.
43  *
44  * Accompanying the CountingPtr is a class ReferenceCounter, from which reference
45  * counted classes may be derive from. The class ReferenceCounter implement all
46  * methods required for reference counting.
47  *
48  * The whole method is more similar to boost's instrusive_ptr, but also yields
49  * something resembling std::shared_ptr. However, compared to std::shared_ptr,
50  * this class only contains a single pointer, while shared_ptr contains two
51  * which are only related if constructed with std::make_shared.
52  *
53  * Another advantage with this method is that no kludges like
54  * std::enable_shared_from_this are needed.
55  */
56 template <typename Type, typename Deleter = CountingPtrDefaultDeleter>
57 class CountingPtr
58 {
59 public:
60  //! contained type.

71  //! decrement reference count of current object and maybe delete it.
72  void dec_reference() noexcept {
73  if (ptr_ && ptr_->dec_reference())

261 /*!
262  * Provides reference counting abilities for use with CountingPtr.
263  *
264  * Use as superclass of the actual object, this adds a reference_count_
265  * value. Then either use CountingPtr as pointer to manage references and
266  * deletion, or just do normal new and delete.
267  */
268 class ReferenceCounter
269 {
270 private:
271  //! the reference count is kept mutable for CountingPtr<const Type> to
272  //! change the reference count.
273  mutable std::atomic<size_t> reference_count_;
274 
275 public:
276  //! new objects have zero reference count
277  ReferenceCounter() noexcept
278  : reference_count_(0) { }
279 
280  //! coping still creates a new object with zero reference count
281  ReferenceCounter(const ReferenceCounter&) noexcept
282  : reference_count_(0) { }
283 
284  //! assignment operator, leaves pointers unchanged
285  ReferenceCounter& operator = (const ReferenceCounter&) noexcept {
286  // changing the contents leaves pointers unchanged
287  return *this;
288  }
289 
290  ~ReferenceCounter()
291  { assert(reference_count_ == 0); }
292 
293 public:
294  //! Call whenever setting a pointer to the object.
295  void inc_reference() const noexcept
296  { ++reference_count_; }
297 
298  /*!
299  * Call whenever resetting (i.e. overwriting) a pointer to the object.
300  * IMPORTANT: In case of self-assignment, call AFTER inc_reference().
301  *
302  * \return if the object has to be deleted (i.e. if it's reference count
303  * dropped to zero)
304  */
305  bool dec_reference() const noexcept {
306  assert(reference_count_ > 0);
307  return (--reference_count_ == 0);
308  }
309 
310  //! Test if the ReferenceCounter is referenced by only one CountingPtr.
311  bool unique() const noexcept
312  { return (reference_count_ == 1); }
313 
314  //! Return the number of references to this object (for debugging)
315  size_t reference_count() const noexcept
316  { return reference_count_; }
317 };

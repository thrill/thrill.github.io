41 /*!
42  * High-performance smart pointer used as a wrapping reference counting pointer.
43  *
44  * This smart pointer class requires two functions in the template type: void
45  * IncReference() and void DecReference(). These must increment and decrement a
46  * reference count inside the templated object. When initialized, the type must
47  * have reference count zero. Each new object referencing the data calls
48  * IncReference() and each destroying holder calls del_reference(). When the
49  * data object determines that it's internal count is zero, then it must destroy
50  * itself.
51  *
52  * Accompanying the CountingPtr is a class ReferenceCount, from which reference
53  * counted classes may be derive from. The class ReferenceCount implement all
54  * methods required for reference counting.
55  *
56  * The whole method is more similar to boost's instrusive_ptr, but also yields
57  * something resembling std::shared_ptr. However, compared to std::shared_ptr,
58  * this class only contains a single pointer, while shared_ptr contains two
59  * which are only related if constructed with std::make_shared.
60  */
61 template <typename Type,
62  typename Deleter = DefaultCountingPtrDeleter>
63 class CountingPtr
64 {
65 public:
66  //! contained type.
67  using element_type = Type;
68 
69 private:
70  //! the pointer to the currently referenced object.
71  Type* ptr_;
72 
73  //! increment reference count for current object.
74  void IncReference() noexcept
75  { IncReference(ptr_); }
76 
77  //! increment reference count of other object.
78  void IncReference(Type* o) noexcept
79  { if (o) o->IncReference(); }
80 
81  //! decrement reference count of current object and maybe delete it.
82  void DecReference() noexcept
83  { if (ptr_ && ptr_->DecReference()) Deleter()(ptr_); }
84 
85 public:
86  //! all CountingPtr are friends such that they may steal pointers.
87  template <typename Other, typename OtherDeleter>
88  friend class CountingPtr;
89 
90  //! default constructor: contains a nullptr pointer.
91  CountingPtr() noexcept
92  : ptr_(nullptr) { }

95  CountingPtr(std::nullptr_t) noexcept // NOLINT
96  : ptr_(nullptr) { }

99  explicit CountingPtr(Type* ptr) noexcept

103  CountingPtr(const CountingPtr& other) noexcept

257 /*!
258  * Provides reference counting abilities for use with CountingPtr.
259  *
260  * Use as superclass of the actual object, this adds a reference_count
261  * value. Then either use CountingPtr as pointer to manage references and
262  * deletion, or just do normal new and delete.
263  */
264 class ReferenceCount
265 {
266 private:
267  //! the reference count is kept mutable for CountingPtr<const Type> to
268  //! change the reference count.
269  mutable std::atomic<size_t> reference_count_;
270 
271 public:
272  //! new objects have zero reference count
273  ReferenceCount() noexcept
274  : reference_count_(0) { }
275 
276  //! coping still creates a new object with zero reference count
277  ReferenceCount(const ReferenceCount&) noexcept
278  : reference_count_(0) { }
279 
280  //! assignment operator, leaves pointers unchanged
281  ReferenceCount& operator = (const ReferenceCount&) noexcept
282  { return *this; } // changing the contents leaves pointers unchanged
283 
284  ~ReferenceCount()
285  { assert(reference_count_ == 0); }
286 
287 public:
288  //! Call whenever setting a pointer to the object
289  void IncReference() const noexcept
290  { ++reference_count_; }
291 
292  /*!
293  * Call whenever resetting (i.e. overwriting) a pointer to the object.
294  * IMPORTANT: In case of self-assignment, call AFTER IncReference().
295  *
296  * \return if the object has to be deleted (i.e. if it's reference count
297  * dropped to zero)
298  */
299  bool DecReference() const noexcept
300  { assert(reference_count_ > 0); return (--reference_count_ == 0); }
301 
302  //! Test if the ReferenceCount is referenced by only one CountingPtr.
303  bool unique() const noexcept
304  { return (reference_count_ == 1); }
305 
306  //! Return the number of references to this object (for debugging)
307  size_t reference_count() const noexcept
308  { return reference_count_; }
309 };

80 void Manager::RunTask(const std::chrono::steady_clock::time_point& tp) {
81 
82  common::JsonLine line = logger_.line();
83  line << "class" << "NetManager"
84  << "event" << "profile";
85 
86  double elapsed = static_cast<double>(
87  std::chrono::duration_cast<std::chrono::microseconds>(
88  tp - tp_last_).count()) / 1e6;
89 
90  size_t total_tx = 0, total_rx = 0;
91  size_t prev_total_tx = 0, prev_total_rx = 0;
92 
93  for (size_t g = 0; g < kGroupCount; ++g) {
94  Group& group = *groups_[g];
95 
96  size_t group_tx = 0, group_rx = 0;
97  size_t prev_group_tx = 0, prev_group_rx = 0;
98  std::vector<size_t> tx_per_host(group.num_hosts());
99  std::vector<size_t> rx_per_host(group.num_hosts());
100 
101  for (size_t h = 0; h < group.num_hosts(); ++h) {
102  if (h == group.my_host_rank()) continue;
103 
104  Connection& conn = group.connection(h);
105 
106  size_t tx = conn.tx_bytes_.load(std::memory_order_relaxed);
107  size_t rx = conn.rx_bytes_.load(std::memory_order_relaxed);
108  size_t prev_tx = conn.prev_tx_bytes_;
109  size_t prev_rx = conn.prev_rx_bytes_;
110 
111  group_tx += tx;
112  prev_group_tx += prev_tx;
113  group.connection(h).prev_tx_bytes_ = tx;
114 
115  group_rx += rx;
116  prev_group_rx += prev_rx;
117  group.connection(h).prev_rx_bytes_ = rx;
118 
119  tx_per_host[h] = tx;
120  rx_per_host[h] = rx;
121  }
122 
123  line.sub(g == 0 ? "flow" : g == 1 ? "data" : "???")
124  << "tx_bytes" << group_tx
125  << "rx_bytes" << group_rx
126  << "tx_speed"
127  << static_cast<double>(group_tx - prev_group_tx) / elapsed
128  << "rx_speed"
129  << static_cast<double>(group_rx - prev_group_rx) / elapsed
130  << "tx_per_host" << tx_per_host
131  << "rx_per_host" << rx_per_host;
132 
133  total_tx += group_tx;
134  total_rx += group_rx;
135  prev_total_tx += prev_group_tx;
136  prev_total_rx += prev_group_rx;
137 
138  tp_last_ = tp;
139  }
140 
141  // write out totals
142  line
143  << "tx_bytes" << total_tx
144  << "rx_bytes" << total_rx
145  << "tx_speed"
146  << static_cast<double>(total_tx - prev_total_tx) / elapsed
147  << "rx_speed"
148  << static_cast<double>(total_rx - prev_total_rx) / elapsed;
149 }

49 void ProfileThread::Worker() {
50  std::unique_lock<std::timed_mutex> lock(mutex_);
51 
52  steady_clock::time_point tm = steady_clock::now();
53 
54  while (!terminate_)
55  {
56  if (tasks_.empty()) {
57  cv_.wait(mutex_, [this]() { return !tasks_.empty(); });
58  continue;
59  }
60 
61  while (tasks_.top().next_timeout <= tm) {
62  const Timer& top = tasks_.top();
63  top.task->RunTask(tm);
64 
65  // requeue timeout event again.
66  tasks_.emplace(top.next_timeout + top.period,
67  top.period, top.task, top.own_task);
68  tasks_.pop();
69  }
70 
71  tm = tasks_.top().next_timeout;
72  cv_.wait_until(mutex_, tm);
73  tm = steady_clock::now();
74  }
75 }

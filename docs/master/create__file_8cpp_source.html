46 FileBasePtr CreateFile(DiskConfig& config, int mode, int disk_allocator_id) {
47  // apply disk_config settings to open mode
48 
49  mode &= ~(FileBase::DIRECT | FileBase::REQUIRE_DIRECT); // clear DIRECT and REQUIRE_DIRECT
50 
51  switch (config.direct) {
52  case DiskConfig::DIRECT_OFF:
53  break;
54  case DiskConfig::DIRECT_TRY:
55  mode |= FileBase::DIRECT;
56  break;
57  case DiskConfig::DIRECT_ON:
58  mode |= FileBase::DIRECT | FileBase::REQUIRE_DIRECT;
59  break;
60  }
61 
62  // automatically enumerate disks as separate device ids
63 
64  if (config.device_id == FileBase::DEFAULT_DEVICE_ID)
65  {
66  config.device_id = Config::GetInstance()->get_next_device_id();
67  }
68  else
69  {
70  Config::GetInstance()->update_max_device_id(config.device_id);
71  }
72 
73  // *** Select fileio Implementation
74 
75  if (config.io_impl == "syscall")
76  {
77  UfsFileBase* result =
78  new SyscallFile(config.path, mode, config.queue, disk_allocator_id,
79  config.device_id);
80  result->lock();
81 
82  // if marked as device but file is not -> throw!
83  if (config.raw_device && !result->is_device())
84  {
85  delete result;
86  THRILL_THROWS(IoError, "Disk " << config.path << " was expected to "
87  "be a raw block device, but it is a normal file!");
88  }
89 
90  // if is raw_device -> get size and remove some flags.
91  if (result->is_device())
92  {
93  config.raw_device = true;
94  config.size = result->size();
95  config.autogrow =
96  config.delete_on_exit = config.unlink_on_open = false;
97  }
98 
99  if (config.unlink_on_open)
100  result->unlink();
101 
102  return FileBasePtr(result);
103  }
104  else if (config.io_impl == "memory")
105  {
106  MemoryFile* result =
107  new MemoryFile(config.queue, disk_allocator_id, config.device_id);
108  result->lock();
109  return FileBasePtr(result);
110  }
111 #if THRILL_HAVE_LINUXAIO_FILE
112  // linuxaio can have the desired queue length, specified as queue_length=?
113  else if (config.io_impl == "linuxaio")
114  {
115  // linuxaio_queue is a singleton.
116  config.queue = FileBase::DEFAULT_LINUXAIO_QUEUE;
117

145  }

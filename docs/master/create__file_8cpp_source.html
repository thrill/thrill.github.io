46 FileBasePtr CreateFile(DiskConfig& cfg, int mode, int disk_allocator_id) {
47  // apply disk_config settings to open mode
48 
49  mode &= ~(FileBase::DIRECT | FileBase::REQUIRE_DIRECT); // clear DIRECT and REQUIRE_DIRECT
50 
51  switch (cfg.direct) {
52  case DiskConfig::DIRECT_OFF:
53  break;
54  case DiskConfig::DIRECT_TRY:
55  mode |= FileBase::DIRECT;
56  break;
57  case DiskConfig::DIRECT_ON:
58  mode |= FileBase::DIRECT | FileBase::REQUIRE_DIRECT;
59  break;
60  }
61 
62  // automatically enumerate disks as separate device ids
63 
64  if (cfg.device_id == FileBase::DEFAULT_DEVICE_ID)
65  {
66  cfg.device_id = Config::GetInstance()->get_next_device_id();
67  }
68  else
69  {
70  Config::GetInstance()->update_max_device_id(cfg.device_id);
71  }
72 
73  // *** Select fileio Implementation
74 
75  if (cfg.io_impl == "syscall")
76  {
77  UfsFileBase* result =
78  new SyscallFile(cfg.path, mode, cfg.queue, disk_allocator_id,
79  cfg.device_id);
80  result->lock();
81 
82  // if marked as device but file is not -> throw!
83  if (cfg.raw_device && !result->is_device())
84  {
85  delete result;
86  THRILL_THROWS(IoError, "Disk " << cfg.path << " was expected to be "
87  "a raw block device, but it is a normal file!");
88  }
89 
90  // if is raw_device -> get size and remove some flags.
91  if (result->is_device())
92  {
93  cfg.raw_device = true;
94  cfg.size = result->size();
95  cfg.autogrow = cfg.delete_on_exit = cfg.unlink_on_open = false;
96  }
97 
98  if (cfg.unlink_on_open)
99  result->unlink();
100 
101  return FileBasePtr(result);
102  }
103  else if (cfg.io_impl == "memory")
104  {
105  MemoryFile* result = new MemoryFile(cfg.queue, disk_allocator_id, cfg.device_id);
106  result->lock();
107  return FileBasePtr(result);
108  }
109 #if THRILL_HAVE_LINUXAIO_FILE
110  // linuxaio can have the desired queue length, specified as queue_length=?
111  else if (cfg.io_impl == "linuxaio")
112  {
113  // linuxaio_queue is a singleton.
114  cfg.queue = FileBase::DEFAULT_LINUXAIO_QUEUE;
115 
116  UfsFileBase* result =
117  new LinuxaioFile(cfg.path, mode, cfg.queue, disk_allocator_id,
118  cfg.device_id, cfg.queue_length);

121 
122  // if marked as device but file is not -> throw!

127  "a raw block device, but it is a normal file!");
128  }
129

136  }
137

139  result->unlink();
140 
141  return FileBasePtr(result);
142  }
143 #endif
144 #if THRILL_HAVE_MMAP_FILE
145  else if (cfg.io_impl == "mmap")

29 void Percentiles(api::Context& ctx, const std::string& input_path) {

37  .FlatMap<std::pair<time_t, double> >(
38  [&splitted](const std::string& input, auto emit) {
39  common::SplitRef(input, ',', splitted);
40  if (splitted[0] != "time") {
41  time_t timet = std::stod(splitted[0]);
42  struct tm* tmstr = std::localtime(&timet);
43  double temp = std::stod(splitted[1]);
44  size_t time = 24 * tmstr->tm_yday + tmstr->tm_hour;
45  emit(std::make_pair(time, temp));
46  }
47  })

50  auto median_fn =
51  [](auto& r, std::size_t) {
52  std::vector<double> all;
53  size_t time = 0;
54  while (r.HasNext()) {
55  auto next = r.Next();
56  all.push_back(next.second);
57  time = next.first;
58  }
59  std::sort(std::begin(all), std::end(all));
60  return std::make_pair(time, all[all.size() / 2 - 1]);
61  };
62 
63  auto time_keyfn =
64  [](std::pair<time_t, double> input) {
65  return input.first;
66  };
67 
68  // group by to compute median
69  ctx.net.Barrier();
70  thrill::common::StatsTimerStart timer;
71  temps.GroupByKey<std::pair<size_t, double> >(
72  LocationDetectionFlag<use_detection>(),
73  time_keyfn, median_fn).Size();
74 
75  ctx.net.Barrier();

78  if (ctx.my_rank() == 0) {

84  << " machines=" << ctx.num_hosts();
85  }
86  else {

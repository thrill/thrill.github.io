101  void PreOp(const ValueType& input) {
102  unsorted_writer_.Put(input);
103  // In this stage we do not know how many elements are there in total.
104  // Therefore we draw samples based on current number of elements and
105  // randomly replace older samples when we have too many.
106  if (--sample_interval_ == 0) {
107  if (samples_.size() < wanted_sample_size()) {
108  while (samples_.size() < wanted_sample_size())
109  samples_.emplace_back(SampleIndexPair(input, local_items_));
110  }
111  else {
112  samples_[rng_() % samples_.size()] =
113  SampleIndexPair(input, local_items_);
114  }
115  sample_interval_ = std::max(
116  size_t(1), (local_items_ + 1) / wanted_sample_size());
117  LOG0 << "SortNode::PreOp() sample_interval_=" << sample_interval_;
118  }
119  local_items_++;
120  }

188  //! calculate maximum merging degree from available memory and the number of
189  //! files. additionally calculate the prefetch size of each File.
190  std::pair<size_t, size_t> MaxMergeDegreePrefetch() {
191  size_t avail_blocks = DIABase::mem_limit_ / data::default_block_size;
192  if (files_.size() >= avail_blocks) {
193  // more files than blocks available -> partial merge of avail_blocks
194  // Files with prefetch = 0, which is one read Block per File.
195  return std::make_pair(avail_blocks, 0u);
196  }
197  else {
198  // less files than available Blocks -> split blocks equally among
199  // Files.
200  return std::make_pair(
201  files_.size(),
202  std::min<size_t>(16u, (avail_blocks / files_.size()) - 1));

206  void PushData(bool consume) final {
207  Timer timer_pushdata;
208  timer_pushdata.Start();
209 
210  size_t local_size = 0;
211  if (files_.size() == 0) {
212  // nothing to push
213  }
214  else if (files_.size() == 1) {
215  local_size = files_[0].num_items();
216  this->PushFile(files_[0], consume);
217  }
218  else {
219  size_t merge_degree, prefetch;
220 
221  // merge batches of files if necessary
222  while (files_.size() > MaxMergeDegreePrefetch().first)
223  {
224  std::tie(merge_degree, prefetch) = MaxMergeDegreePrefetch();
225 
226  sLOG1 << "Partial multi-way-merge of"
227  << merge_degree << "files with prefetch" << prefetch;
228 
229  // create merger for first merge_degree_ Files
230  std::vector<data::File::ConsumeReader> seq;
231  seq.reserve(merge_degree);
232 
233  for (size_t t = 0; t < merge_degree; ++t)
234  seq.emplace_back(files_[t].GetConsumeReader(0));
235 
236  StartPrefetch(seq, prefetch);
237 
238  auto puller = core::make_multiway_merge_tree<ValueType>(
239  seq.begin(), seq.end(), compare_function_);
240 
241  // create new File for merged items
242  files_.emplace_back(context_.GetFile(this));
243  auto writer = files_.back().GetWriter();
244 
245  while (puller.HasNext()) {
246  writer.Put(puller.Next());
247  }
248  writer.Close();
249 
250  // this clear is important to release references to the files.
251  seq.clear();
252 
253  // remove merged files
254  files_.erase(files_.begin(), files_.begin() + merge_degree);
255  }
256 
257  std::tie(merge_degree, prefetch) = MaxMergeDegreePrefetch();
258 
259  sLOG1 << "Start multi-way-merge of" << files_.size() << "files"
260  << "with prefetch" << prefetch;
261 
262  // construct output merger of remaining Files
263  std::vector<data::File::Reader> seq;
264  seq.reserve(files_.size());
265 
266  for (size_t t = 0; t < files_.size(); ++t)
267  seq.emplace_back(files_[t].GetReader(consume, 0));
268 
269  StartPrefetch(seq, prefetch);
270 
271  auto puller = core::make_multiway_merge_tree<ValueType>(
272  seq.begin(), seq.end(), compare_function_);
273 
274  while (puller.HasNext()) {
275  this->PushItem(puller.Next());
276  local_size++;
277  }
278  }
279 
280  timer_pushdata.Stop();
281 
282  if (stats_enabled) {
283  context_.PrintCollectiveMeanStdev(
284  "Sort() timer_pushdata", timer_pushdata.SecondsDouble());
285 
286  context_.PrintCollectiveMeanStdev("Sort() local_size", local_size);
287  }

358  void FindAndSendSplitters(
359  std::vector<SampleIndexPair>& splitters, size_t sample_size,
360  data::MixStreamPtr& sample_stream,
361  std::vector<data::MixStream::Writer>& sample_writers) {
362 
363  // Get samples from other workers
364  size_t num_total_workers = context_.num_workers();
365 
366  std::vector<SampleIndexPair> samples;
367  samples.reserve(sample_size * num_total_workers);
368 
369  auto reader = sample_stream->GetMixReader(/* consume */ true);
370 
371  while (reader.HasNext()) {
372  samples.push_back(reader.template Next<SampleIndexPair>());
373  }
374  if (samples.size() == 0) return;
375 
376  // Find splitters
377  std::sort(samples.begin(), samples.end(),
378  [this](
379  const SampleIndexPair& a, const SampleIndexPair& b) {
380  return LessSampleIndex(a, b);
381  });
382 
383  size_t splitting_size = samples.size() / num_total_workers;
384 
385  // Send splitters to other workers
386  for (size_t i = 1; i < num_total_workers; ++i) {
387  splitters.push_back(samples[i * splitting_size]);
388  for (size_t j = 1; j < num_total_workers; j++) {
389  sample_writers[j].Put(splitters.back());
390  }
391  }
392 
393  for (size_t j = 1; j < num_total_workers; ++j)
394  sample_writers[j].Close();
395  }

420  void recurse(const SampleIndexPair* lo, const SampleIndexPair* hi,
421  unsigned int treeidx) {
422  // pick middle element as splitter
423  const SampleIndexPair* mid = lo + (ssize_t)(hi - lo) / 2;
424  assert(mid < samples_ + ssplitter_);
425  tree_[treeidx] = mid->first;
426 
427  if (2 * treeidx < ssplitter_)
428  {
429  const SampleIndexPair* midlo = mid, * midhi = mid + 1;
430  recurse(lo, midlo, 2 * treeidx + 0);
431  recurse(midhi, hi, 2 * treeidx + 1);
432  }
433  }
434  };

436  bool LessSampleIndex(const SampleIndexPair& a, const SampleIndexPair& b) {
437  return compare_function_(a.first, b.first) || (
438  !compare_function_(b.first, a.first) && a.second < b.second);
439  }

441  bool EqualSampleGreaterIndex(const SampleIndexPair& a, const SampleIndexPair& b) {
442  return !compare_function_(a.first, b.first) && a.second >= b.second;

451  void TransmitItems(
452  // Tree of splitters, sizeof |splitter|
453  const ValueType* const tree,
454  // Number of buckets: k = 2^{log_k}
455  size_t k,
456  size_t log_k,
457  // Number of actual workers to send to
458  size_t actual_k,
459  const SampleIndexPair* const sorted_splitters,
460  size_t prefix_items,
461  data::MixStreamPtr& data_stream) {
462 
463  data::File::ConsumeReader unsorted_reader =
464  unsorted_file_.GetConsumeReader();
465 
466  std::vector<data::MixStream::Writer> data_writers =
467  data_stream->GetWriters();
468 
469  // enlarge emitters array to next power of two to have direct access,
470  // because we fill the splitter set up with sentinels == last splitter,
471  // hence all items land in the last bucket.
472  assert(data_writers.size() == actual_k);
473  assert(actual_k <= k);
474 
475  while (data_writers.size() < k)
476  data_writers.emplace_back(nullptr);
477 
478  std::swap(data_writers[actual_k - 1], data_writers[k - 1]);
479 
480  // classify all items (take two at once) and immediately transmit them.
481 
482  const size_t stepsize = 2;
483 
484  size_t i = prefix_items;
485  for ( ; i < prefix_items + RoundDown(local_items_, stepsize); i += stepsize)
486  {
487  // take two items
488  size_t j0 = 1;
489  ValueType el0 = unsorted_reader.Next<ValueType>();
490 
491  size_t j1 = 1;
492  ValueType el1 = unsorted_reader.Next<ValueType>();
493 
494  // run items down the tree
495  for (size_t l = 0; l < log_k; l++)
496  {
497  j0 = 2 * j0 + (compare_function_(el0, tree[j0]) ? 0 : 1);
498  j1 = 2 * j1 + (compare_function_(el1, tree[j1]) ? 0 : 1);
499  }
500 
501  size_t b0 = j0 - k;
502  size_t b1 = j1 - k;
503 
504  while (b0 && EqualSampleGreaterIndex(
505  sorted_splitters[b0 - 1], SampleIndexPair(el0, i + 0))) {
506  b0--;
507 
508  // LOG0 << "el0 equal match b0 " << b0
509  // << " prefix_items " << prefix_items
510  // << " lhs.first = " << sorted_splitters[b0 - 1].first
511  // << " lhs.second = " << sorted_splitters[b0 - 1].second
512  // << " rhs.first = " << el0
513  // << " rhs.second = " << i;
514  }
515 
516  while (b1 && EqualSampleGreaterIndex(
517  sorted_splitters[b1 - 1], SampleIndexPair(el1, i + 1))) {
518  b1--;
519  }
520 
521  assert(data_writers[b0].IsValid());
522  assert(data_writers[b1].IsValid());
523 
524  data_writers[b0].Put(el0);
525  data_writers[b1].Put(el1);
526  }
527 
528  // last iteration of loop if we have an odd number of items.
529  for ( ; i < prefix_items + local_items_; i++)
530  {
531  size_t j0 = 1;
532  ValueType el0 = unsorted_reader.Next<ValueType>();
533 
534  // run item down the tree
535  for (size_t l = 0; l < log_k; l++)
536  {
537  j0 = 2 * j0 + (compare_function_(el0, tree[j0]) ? 0 : 1);
538  }
539 
540  size_t b0 = j0 - k;
541 
542  while (b0 && EqualSampleGreaterIndex(
543  sorted_splitters[b0 - 1], SampleIndexPair(el0, i))) {
544  b0--;
545  }
546 
547  assert(data_writers[b0].IsValid());
548  data_writers[b0].Put(el0);
549  }
550 
551  // close writers and flush data
552  for (size_t j = 0; j < data_writers.size(); j++)
553  data_writers[j].Close();
554  }

685  void ReceiveItems(data::MixStreamPtr& data_stream) {
686 
687  auto reader = data_stream->GetMixReader(/* consume */ true);
688 
689  LOG0 << "Writing files";
690 
691  // M/2 such that the other half is used to prepare the next bulk
692  size_t capacity = DIABase::mem_limit_ / sizeof(ValueType) / 2;
693  std::vector<ValueType> vec;
694  vec.reserve(capacity);
695 
696  while (reader.HasNext()) {
697  if (!mem::memory_exceeded && vec.size() < capacity) {
698  vec.push_back(reader.template Next<ValueType>());
699  }
700  else {
701  SortAndWriteToFile(vec);
702  }
703  }
704 
705  if (vec.size())

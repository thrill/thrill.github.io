33 UfsFileBase::UfsFileBase(
34  const std::string& filename,
35  int mode)
36  : file_des_(-1), mode_(mode), path_(filename) {
37  int flags = 0;
38 
39  if (mode & RDONLY)
40  {
41  flags |= O_RDONLY;
42  }
43 
44  if (mode & WRONLY)
45  {
46  flags |= O_WRONLY;
47  }
48 
49  if (mode & RDWR)
50  {
51  flags |= O_RDWR;
52  }
53 
54  if (mode & CREAT)
55  {
56  flags |= O_CREAT;
57  }
58 
59  if (mode & TRUNC)
60  {
61  flags |= O_TRUNC;
62  }
63 
64  if ((mode & DIRECT) || (mode & REQUIRE_DIRECT))
65  {
66 #if __APPLE__
67  // no additional open flags are required for Mac OS X
68 #else
69  flags |= O_DIRECT;
70 #endif
71  }
72 
73  if (mode & SYNC)
74  {
75  flags |= O_RSYNC;
76  flags |= O_DSYNC;
77  flags |= O_SYNC;
78  }
79 
80 #if THRILL_WINDOWS
81  flags |= O_BINARY; // the default in MS is TEXT mode
82 #endif
83 
84 #if THRILL_WINDOWS || defined(__MINGW32__)
85  const int perms = S_IREAD | S_IWRITE;
86 #else
87  const int perms = S_IREAD | S_IWRITE | S_IRGRP | S_IWGRP;
88 #endif
89 
90  if ((file_des_ = ::open(filename.c_str(), flags, perms)) >= 0)
91  {
92  need_alignment_ = (mode & DIRECT) != 0;
93  _after_open();
94  return;
95  }
96 
97  if ((mode & DIRECT) && !(mode & REQUIRE_DIRECT) && errno == EINVAL)
98  {
99  LOG1 << "open() error on path=" << filename
100  << " flags=" << flags << ", retrying without O_DIRECT.";
101 
102  flags &= ~O_DIRECT;
103  mode_ &= ~DIRECT;
104 
105  if ((file_des_ = ::open(filename.c_str(), flags, perms)) >= 0)
106  {
107  _after_open();
108  return;
109  }
110  }
111 
112  THRILL_THROW_ERRNO(IoError, "open() rc=" << file_des_ << " path=" << filename << " flags=" << flags);
113 }

119 void UfsFileBase::_after_open() {
120  // stat file type
121 #if THRILL_WINDOWS || defined(__MINGW32__)
122  struct _stat64 st;
123  THRILL_THROW_ERRNO_NE_0(::_fstat64(file_des_, &st), IoError,
124  "_fstat64() path=" << path_ << " fd=" << file_des_);
125 #else
126  struct stat st;
127  THRILL_THROW_ERRNO_NE_0(::fstat(file_des_, &st), IoError,
128  "fstat() path=" << path_ << " fd=" << file_des_);
129 #endif
130  is_device_ = S_ISBLK(st.st_mode) ? true : false;
131 
132 #if __APPLE__
133  if (mode_ & REQUIRE_DIRECT) {
134  THRILL_THROW_ERRNO_NE_0(fcntl(file_des_, F_NOCACHE, 1), IoError,
135  "fcntl() path=" << path_ << " fd=" << file_des_);
136  THRILL_THROW_ERRNO_NE_0(fcntl(file_des_, F_RDAHEAD, 0), IoError,
137  "fcntl() path=" << path_ << " fd=" << file_des_);
138  }
139  else if (mode_ & DIRECT) {
140  if (fcntl(file_des_, F_NOCACHE, 1) != 0) {
141  LOG1 << "fcntl(fd,F_NOCACHE,1) failed on path=" << path_
142  << " fd=" << file_des_ << " : " << strerror(errno);
143  }
144  if (fcntl(file_des_, F_RDAHEAD, 0) != 0) {
145  LOG1 << "fcntl(fd,F_RDAHEAD,0) failed on path=" << path_
146  << " fd=" << file_des_ << " : " << strerror(errno);
147  }
148  }
149 #endif
150 
151  // successfully opened file descriptor
152  if (!(mode_ & NO_LOCK))
153  lock();
154 }

156 void UfsFileBase::close() {
157  std::unique_lock<std::mutex> fd_lock(fd_mutex_);
158 
159  if (file_des_ == -1)
160  return;
161

168 void UfsFileBase::lock() {
169 #if THRILL_WINDOWS || defined(__MINGW32__)
170  // not yet implemented
171 #else
172  std::unique_lock<std::mutex> fd_lock(fd_mutex_);
173  struct flock lock_struct;
174  lock_struct.l_type = (short)(mode_ & RDONLY ? F_RDLCK : F_RDLCK | F_WRLCK);
175  lock_struct.l_whence = SEEK_SET;
176  lock_struct.l_start = 0;
177  lock_struct.l_len = 0; // lock all bytes
178  if ((::fcntl(file_des_, F_SETLK, &lock_struct)) < 0)
179  THRILL_THROW_ERRNO(IoError, "fcntl(,F_SETLK,) path=" << path_ << " fd=" << file_des_);
180 #endif
181 }

183 FileBase::offset_type UfsFileBase::_size() {
184  // We use lseek SEEK_END to find the file size. This works for raw devices
185  // (where stat() returns zero), and we need not reset the position because
186  // serve() always lseek()s before read/write.
187 
188  off_t rc = ::lseek(file_des_, 0, SEEK_END);
189  if (rc < 0)
190  THRILL_THROW_ERRNO(IoError, "lseek(fd,0,SEEK_END) path=" << path_ << " fd=" << file_des_);
191 
192  // return value is already the total size
193  return rc;
194 }

206 void UfsFileBase::_set_size(offset_type newsize) {
207  offset_type cur_size = _size();
208 
209  if (!(mode_ & RDONLY) && !is_device_)
210  {
211 #if THRILL_WINDOWS || defined(__MINGW32__)
212  HANDLE hfile = (HANDLE)::_get_osfhandle(file_des_);
213  THRILL_THROW_ERRNO_NE_0((hfile == INVALID_HANDLE_VALUE), IoError,
214  "_get_osfhandle() path=" << path_ << " fd=" << file_des_);
215 
216  LARGE_INTEGER desired_pos;
217  desired_pos.QuadPart = newsize;
218 
219  if (!SetFilePointerEx(hfile, desired_pos, nullptr, FILE_BEGIN))
220  THRILL_THROW_WIN_LASTERROR(IoError,
221  "SetFilePointerEx in ufs_file_base::set_size(..) oldsize=" << cur_size <<
222  " newsize=" << newsize << " ");
223 
224  if (!SetEndOfFile(hfile))
225  THRILL_THROW_WIN_LASTERROR(IoError,
226  "SetEndOfFile oldsize=" << cur_size <<
227  " newsize=" << newsize << " ");
228 #else
229  THRILL_THROW_ERRNO_NE_0(::ftruncate(file_des_, newsize), IoError,
230  "ftruncate() path=" << path_ << " fd=" << file_des_);
231 #endif
232  }
233 
234 #if !THRILL_WINDOWS
235  if (newsize > cur_size)
236  THRILL_THROW_ERRNO_LT_0(::lseek(file_des_, newsize - 1, SEEK_SET), IoError,
237  "lseek() path=" << path_ << " fd=" << file_des_ << " pos=" << newsize - 1);
238 #endif
239 }

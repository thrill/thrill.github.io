180 template <typename BIDType, typename DiskAssignFunctor, typename OutputIterator>
181 void BlockManager::new_blocks_int(
182  const size_t nblocks, const DiskAssignFunctor& functor,
183  size_t offset, OutputIterator out) {
184 
185  std::unique_lock<std::mutex> lock(mutex_);
186 
187  OutputIterator it = out;
188  for (size_t i = 0; i != nblocks; ++i, ++it)
189  {
190  size_t disk_id;
191  FileBase* disk_file;
192  DiskAllocator* disk_alloc;
193 
194  for (size_t retry = 0; retry < 100; ++retry) {
195  // choose disk by calling DiskAssignFunctor
196  disk_id = functor(offset + i);
197 
198  disk_file = disk_files_[disk_id].get();
199  disk_alloc = disk_allocators_[disk_file->get_allocator_id()];
200 
201  // check if disk has enough free space
202  if (disk_alloc->free_bytes() >= static_cast<int64_t>(it->size))
203  break;
204  }
205 
206  // if no disk has free space, pick an arbitrary one after 100 rounds.
207 
208  it->storage = disk_file;
209  disk_alloc->NewBlocks(it, it + 1);
210  LOG0 << "BLC:new " << *it;
211 
212  total_allocation_ += it->size;
213  current_allocation_ += it->size;
214  }
215 
216  maximum_allocation_ = std::max(maximum_allocation_, current_allocation_);
217 }

219 template <size_t BlockSize>
220 void BlockManager::delete_block(const BID<BlockSize>& bid) {
221  if (!bid.valid()) {
222  // THRILL_MSG("Warning: invalid block to be deleted.");
223  return;
224  }
225  if (!bid.is_managed())
226  return; // self managed disk
227 
228  std::unique_lock<std::mutex> lock(mutex_);
229 
230  LOG0 << "BLC:delete " << bid;
231  assert(bid.storage->get_allocator_id() >= 0);
232  disk_allocators_[bid.storage->get_allocator_id()]->DeleteBlock(bid);
233  disk_files_[bid.storage->get_allocator_id()]->discard(bid.offset, bid.size);
234 
235  current_allocation_ -= BlockSize;
236 }

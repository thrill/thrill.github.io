69 bool RequestQueueImpl1Q::CancelRequest(Request* req) {
70  if (!req)
71  THRILL_THROW_INVALID_ARGUMENT("Empty request canceled disk_queue.");
72  if (thread_state_() != RUNNING)
73  THRILL_THROW_INVALID_ARGUMENT("Request canceled to not running queue.");
74  if (!dynamic_cast<ServingRequest*>(req))
75  LOG1 << "Incompatible request submitted to running queue.";
76 
77  bool was_still_in_queue = false;
78  {
79  std::unique_lock<std::mutex> lock(queue_mutex_);
80  Queue::iterator pos
81  = std::find(queue_.begin(), queue_.end(), req);
82 
83  if (pos != queue_.end())
84  {
85  queue_.erase(pos);
86  was_still_in_queue = true;
87  lock.unlock();
88  sem_.wait();
89  }
90  }
91 
92  return was_still_in_queue;
93 }

99 void* RequestQueueImpl1Q::worker(void* arg) {
100  RequestQueueImpl1Q* pthis = static_cast<RequestQueueImpl1Q*>(arg);
101  // pin I/O thread to last core
102  common::SetCpuAffinity(std::thread::hardware_concurrency() - 1);
103 
104  for ( ; ; )
105  {
106  pthis->sem_.wait();
107 
108  {
109  std::unique_lock<std::mutex> lock(pthis->queue_mutex_);
110  if (!pthis->queue_.empty())
111  {
112  RequestPtr req = pthis->queue_.front();
113  pthis->queue_.pop_front();
114 
115  lock.unlock();
116 
117  // assert(req->nref() > 1);
118  dynamic_cast<ServingRequest*>(req.get())->serve();
119  }
120  else
121  {
122  lock.unlock();
123 
124  pthis->sem_.signal();
125  }
126  }
127 
128  // terminate if it has been requested and queues are empty
129  if (pthis->thread_state_() == TERMINATING) {
130  if (pthis->sem_.wait() == 0)
131  break;
132  else
133  pthis->sem_.signal();
134  }
135  }
136 
137  pthis->thread_state_.set_to(TERMINATED);
138 
139 #if THRILL_STD_THREADS && THRILL_MSVC >= 1700
140  // Workaround for deadlock bug in Visual C++ Runtime 2012 and 2013, see
141  // request_queue_impl_worker.cpp. -tb
142  ExitThread(nullptr);
143 #else
144  return nullptr;
145 #endif
146 }

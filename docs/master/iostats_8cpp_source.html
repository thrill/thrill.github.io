31 void Stats::write_started(size_t size, double now) {

34  now = timestamp();
35 #else
36  tlx::unused(now);

40 
41  ++write_ops_;
42  write_volume_ += size;
43 #if THRILL_IO_STATS_TIMING
44  double diff = now - parallel_write_begin_;
45  write_time_ += static_cast<double>(acc_writes_) * diff;

47  parallel_write_time_ += (acc_writes_++) ? diff : 0.0;

195  double now = timestamp();
196  {

198 
199  double diff = now - parallel_wait_begin_;
200  io_wait_time_ += static_cast<double>(acc_waits_) * diff;
201  parallel_wait_begin_ = now;
202  parallel_wait_time_ += (acc_waits_--) ? diff : 0.0;
203 
204  if (wait_op == WaitOp::READ) {
205  double diff_read = now - parallel_wait_read_begin_;
206  read_wait_time_ += static_cast<double>(acc_wait_read_) * diff_read;
207  parallel_wait_read_begin_ = now;
208  parallel_wait_read_time_ += (acc_wait_read_--) ? diff_read : 0.0;
209  }
210  else /* if (wait_op == WaitOp::WRITE) */ {
211  double diff_write = now - parallel_wait_write_begin_;
212  write_wait_time_ += static_cast<double>(acc_wait_write_) * diff_write;
213  parallel_wait_write_begin_ = now;
214  parallel_wait_write_time_ += (acc_wait_write_--) ? diff_write : 0.0;
215  }
216 #if THRILL_WAIT_LOG_ENABLED
217  std::ofstream* waitlog = stxxl::logger::GetInstance()->waitlog_stream();
218  if (waitlog)
219  *waitlog << (now - last_reset_time_) << "\t"
220  << ((wait_op == WaitOp::READ) ? diff : 0.0) << "\t"
221  << ((wait_op != WaitOp::READ) ? diff : 0.0) << "\t"
222  << read_wait_time_ << "\t" << write_wait_time_ << std::endl << std::flush;

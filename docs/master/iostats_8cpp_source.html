167 void Stats::wait_started(WaitOp wait_op) {
168  double now = timestamp();
169  {
170  std::unique_lock<std::mutex> wait_lock(wait_mutex_);
171 
172  double diff = now - parallel_wait_begin_;
173  io_wait_time_ += static_cast<double>(acc_waits_) * diff;
174  parallel_wait_begin_ = now;
175  parallel_wait_time_ += (acc_waits_++) ? diff : 0.0;
176 
177  if (wait_op == WaitOp::READ) {
178  diff = now - parallel_wait_read_begin_;
179  read_wait_time_ += static_cast<double>(acc_wait_read_) * diff;
180  parallel_wait_read_begin_ = now;
181  parallel_wait_read_time_ += (acc_wait_read_++) ? diff : 0.0;
182  }
183  else /* if (wait_op == WaitOp::WRITE) */ {
184  // wait_any() is only used from write_pool and buffered_writer, so account WAIT_OP_ANY for WaitOp::WRITE, too
185  diff = now - parallel_wait_write_begin_;
186  write_wait_time_ += static_cast<double>(acc_wait_write_) * diff;
187  parallel_wait_write_begin_ = now;

193 void Stats::wait_finished(WaitOp wait_op) {
194  double now = timestamp();
195  {
196  std::unique_lock<std::mutex> wait_lock(wait_mutex_);
197 
198  double diff = now - parallel_wait_begin_;
199  io_wait_time_ += static_cast<double>(acc_waits_) * diff;
200  parallel_wait_begin_ = now;
201  parallel_wait_time_ += (acc_waits_--) ? diff : 0.0;
202 
203  if (wait_op == WaitOp::READ) {
204  double diff_read = now - parallel_wait_read_begin_;
205  read_wait_time_ += static_cast<double>(acc_wait_read_) * diff_read;
206  parallel_wait_read_begin_ = now;
207  parallel_wait_read_time_ += (acc_wait_read_--) ? diff_read : 0.0;
208  }
209  else /* if (wait_op == WaitOp::WRITE) */ {
210  double diff_write = now - parallel_wait_write_begin_;
211  write_wait_time_ += static_cast<double>(acc_wait_write_) * diff_write;
212  parallel_wait_write_begin_ = now;
213  parallel_wait_write_time_ += (acc_wait_write_--) ? diff_write : 0.0;
214  }
215 #if THRILL_WAIT_LOG_ENABLED
216  std::ofstream* waitlog = stxxl::logger::GetInstance()->waitlog_stream();
217  if (waitlog)
218  *waitlog << (now - last_reset_time_) << "\t"
219  << ((wait_op == WaitOp::READ) ? diff : 0.0) << "\t"
220  << ((wait_op != WaitOp::READ) ? diff : 0.0) << "\t"
221  << read_wait_time_ << "\t" << write_wait_time_ << std::endl << std::flush;
222 #endif
223  }
224 }

227 std::ostream& operator << (std::ostream& o, const StatsData& s) {
228 #define hr common::FormatIecUnits
229  o << "Thrill I/O statistics" << std::endl;
230  o << " total number of reads : "
231  << hr(s.read_ops()) << std::endl;
232  o << " average block size (read) : "
233  << hr(s.read_ops() ? s.read_volume() / s.read_ops() : 0) << std::endl;
234  o << " number of bytes read from disks : "
235  << hr(s.read_volume()) << std::endl;
236 #if THRILL_IO_STATS_TIMING
237  o << " time spent in serving all read requests : "

240  << std::endl;
241  o << " time spent in reading (parallel read time) : "
242  << s.parallel_read_time() << " s"
243  << " @ " << (static_cast<double>(s.read_volume()) / 1048576.0 / s.parallel_read_time()) << " MiB/s"
244  << std::endl;
245  if (s.cached_read_ops()) {
246  o << " total number of cached reads : "
247  << hr(s.cached_read_ops()) << std::endl;
248  o << " average block size (cached read) : "
249  << hr(s.cached_read_volume() / s.cached_read_ops()) << std::endl;
250  o << " number of bytes read from cache : "
251  << hr(s.cached_read_volume()) << std::endl;
252  }
253  if (s.cached_write_ops()) {
254  o << " total number of cached writes : "
255  << hr(s.cached_write_ops()) << std::endl;
256  o << " average block size (cached write) : "
257  << hr(s.cached_write_volume() / s.cached_write_ops()) << std::endl;
258  o << " number of bytes written to cache : "
259  << hr(s.cached_write_volume()) << std::endl;
260  }
261 #endif
262  o << " total number of writes : "
263  << hr(s.write_ops()) << std::endl;
264  o << " average block size (write) : "
265  << hr(s.write_ops() ? s.write_volume() / s.write_ops() : 0) << std::endl;
266  o << " number of bytes written to disks : "
267  << hr(s.write_volume()) << std::endl;
268 #if THRILL_IO_STATS_TIMING

272  << std::endl;

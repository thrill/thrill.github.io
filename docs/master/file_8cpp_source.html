130 PinnedBlock KeepFileBlockSource::NextBlock() {
131 
132  if (current_block_ >= file_.num_blocks() && fetching_blocks_.empty())
133  return PinnedBlock();
134 
135  if (num_prefetch_ == 0)

137  // operate without prefetching

146  fetching_blocks_.emplace_back(

157 //! Determine current unpinned Block to deliver via NextBlock()
158 Block KeepFileBlockSource::NextUnpinnedBlock() {
159  if (current_block_ == first_block_) {
160  // construct first block differently, in case we want to shorten it.
161  Block b = file_.block(current_block_++);
162  if (first_item_ != keep_first_item)
163  b.set_begin(first_item_);
164  return b;
165  }
166  else {
167  return file_.block(current_block_++);

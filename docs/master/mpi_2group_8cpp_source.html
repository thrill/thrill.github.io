371 /*!
372  * Construct Group which connects to peers using MPI. As the MPI environment
373  * already defines the connections, no hosts or parameters can be
374  * given. Constructs group_count mpi::Group objects at once. Within each Group
375  * this host has its MPI rank.
376  *
377  * Returns true if this Thrill host participates in the Group.
378  */
379 bool Construct(size_t group_size,
380  std::unique_ptr<Group>* groups, size_t group_count) {
381  std::unique_lock<std::mutex> lock(g_mutex);
382 
383  Initialize();
384 
385  int my_rank;
386  int r = MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);
387  if (r != MPI_SUCCESS)
388  throw Exception("Error during MPI_Comm_rank()", r);
389 
390  int num_mpi_hosts;
391  r = MPI_Comm_size(MPI_COMM_WORLD, &num_mpi_hosts);
392  if (r != MPI_SUCCESS)
393  throw Exception("Error during MPI_Comm_size()", r);
394 
395  if (group_size > static_cast<size_t>(num_mpi_hosts))
396  throw Exception("mpi::Construct(): fewer MPI processes than hosts requested.");
397 
398  for (size_t i = 0; i < group_count; i++) {
399  groups[i] = std::make_unique<Group>(my_rank, i, group_size);
400  }
401 
402  return (static_cast<size_t>(my_rank) < group_size);
403 }

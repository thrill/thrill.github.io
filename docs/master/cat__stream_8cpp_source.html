199 void CatStream::Close() {
200  if (is_closed_) return;
201  is_closed_ = true;
202 
203  sLOG << "CatStream" << id() << "close"
204  << "host" << my_host_rank()
205  << "local_worker_id_" << local_worker_id_;
206 
207  // close all sinks, this should emit sentinel to all other worker.
208  for (size_t i = 0; i < sinks_.size(); ++i) {
209  if (sinks_[i].closed()) continue;
210  sLOG << "CatStream" << id() << "close"
211  << "unopened sink" << i;
212  sinks_[i].Close();
213  }
214 
215  // close loop-back queue from this worker to itself
216  auto my_global_worker_id = my_worker_rank();
217  if (!queues_[my_global_worker_id].write_closed())
218  queues_[my_global_worker_id].Close();
219 
220  // wait for close packets to arrive
221  for (size_t i = 0; i < queues_.size() - workers_per_host(); ++i)
222  sem_closing_blocks_.wait();
223 
224  {
225  std::unique_lock<std::mutex> lock(multiplexer_.mutex_);
226  multiplexer_.active_streams_--;
227  }
228 
229  tx_lifetime_.StopEventually();
230  tx_timespan_.StopEventually();
231  OnAllClosed("CatStream");
232 
233  LOG << "CatStream::Close() finished"
234  << " id_=" << id_
235  << " local_worker_id_=" << local_worker_id_;
236 }

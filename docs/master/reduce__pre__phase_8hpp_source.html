42 //! Emitter implementation to plug into a reduce hash table for
43 //! collecting/flushing items while reducing. Items flushed in the pre-phase are
44 //! transmitted via a network Channel.
45 template <typename TableItem, bool VolatileKey>
46 class ReducePrePhaseEmitter
47 {
48  static constexpr bool debug = false;
49 
50 public:
51  explicit ReducePrePhaseEmitter(std::vector<data::DynBlockWriter>& writer)
52  : writer_(writer),
53  stats_(writer.size(), 0) { }
54 
55  //! output an element into a partition, template specialized for robust and
56  //! non-robust keys
57  void Emit(const size_t& partition_id, const TableItem& p) {
58  assert(partition_id < writer_.size());
59  stats_[partition_id]++;
60  writer_[partition_id].Put(p);
61  }
62 
63  void Flush(size_t partition_id) {
64  assert(partition_id < writer_.size());
65  writer_[partition_id].Flush();
66  }
67 
68  void CloseAll() {
69  sLOG << "emit stats:";
70  size_t i = 0;
71  for (data::DynBlockWriter& e : writer_) {
72  e.Close();
73  sLOG << "emitter" << i << "pushed" << stats_[i++];
74  }
75  }
76 
77 public:
78  //! Set of emitters, one per partition.
79  std::vector<data::DynBlockWriter>& writer_;
80 
81  //! Emitter stats.
82  std::vector<size_t> stats_;
83 };

277  void FlushPartition(size_t partition_id, bool consume, bool grow) {
278  Super::table_.FlushPartitionEmit(
279  partition_id, consume, grow,
280  [this](const size_t& partition_id, const TableItem& ti) {
281  Key key = Super::MakeTableItem::GetKey(
282  ti, Super::table_.key_extractor());
283  if (!non_duplicates_[hash_function_(key) % max_hash_]) {
284 
285  duplicated_elements_++;
286  Super::emit_.Emit(partition_id, ti);
287  }
288  else {
289  non_duplicate_elements_++;
290  Super::emit_.Emit(Super::table_.ctx().my_rank(), ti);
291  }
292  });
293 
294  if (Super::table_.has_spilled_data_on_partition(partition_id)) {
295  data::File::Reader reader =
296  Super::table_.partition_files()[partition_id].GetReader(true);
297  while (reader.HasNext()) {
298  TableItem ti = reader.Next<TableItem>();
299  Key key = Super::MakeTableItem::GetKey(
300  ti, Super::table_.key_extractor());
301  if (!non_duplicates_[hash_function_(key) % max_hash_]) {
302 
303  duplicated_elements_++;
304  Super::emit_.Emit(partition_id, ti);
305  }
306  else {
307  non_duplicate_elements_++;
308  Super::emit_.Emit(Super::table_.ctx().my_rank(), ti);
309  }
310  }
311  }
312 
313  // flush elements pushed into emitter
314  Super::emit_.Flush(partition_id);
315  Super::emit_.Flush(Super::table_.ctx().my_rank());
316  }

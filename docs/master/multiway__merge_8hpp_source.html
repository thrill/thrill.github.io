25 template <typename ValueType, typename ReaderIterator, typename Comparator>
26 class MultiwayMergeTree
27 {
28 public:
29  using Reader = typename std::iterator_traits<ReaderIterator>::value_type;
30 
31  using LoserTreeType = typename LoserTreeTraits<
32  /* stable */ false, ValueType, Comparator>::Type;
33 
34  MultiwayMergeTree(ReaderIterator readers_begin, ReaderIterator readers_end,
35  const Comparator& comp)
36  : readers_(readers_begin),
37  num_inputs_(static_cast<unsigned>(readers_end - readers_begin)),
38  remaining_inputs_(num_inputs_),
39  lt_(static_cast<unsigned>(num_inputs_), comp),
40  current_(num_inputs_) {
41 
42  for (unsigned t = 0; t < num_inputs_; ++t)
43  {
44  if (THRILL_LIKELY(readers_[t].HasNext())) {
45  current_[t].first = true;
46  current_[t].second = readers_[t].template Next<ValueType>();
47  lt_.insert_start(&current_[t].second, t, false);
48  }
49  else {
50  current_[t].first = false;
51  lt_.insert_start(nullptr, t, true);
52  assert(remaining_inputs_ > 0);
53  --remaining_inputs_;
54  }
55  }
56 
57  lt_.init();
58  }
59 
60  bool HasNext() const {
61  return (remaining_inputs_ != 0);
62  }
63 
64  std::pair<ValueType, unsigned> NextWithSource() {
65  // take next smallest element out
66  unsigned top = lt_.min_source();
67  ValueType res = std::move(current_[top].second);
68 
69  if (THRILL_LIKELY(readers_[top].HasNext())) {
70  current_[top].first = true;
71  current_[top].second = readers_[top].template Next<ValueType>();
72  lt_.delete_min_insert(&current_[top].second, false);
73  }
74  else {
75  current_[top].first = false;
76  lt_.delete_min_insert(nullptr, true);
77  assert(remaining_inputs_ > 0);
78  --remaining_inputs_;
79  }

115 /*!
116  * Sequential multi-way merging switch for a file writer as output
117  *
118  * The decision if based on the branching factor and runtime settings.
119  *
120  * \param seqs_begin Begin iterator of iterator pair input sequence.
121  * \param seqs_end End iterator of iterator pair input sequence.
122  * \param comp Comparator.
123  * \tparam Stable Stable merging incurs a performance penalty.
124  * \tparam Sentinels The sequences have a sentinel element.
125  * \return End iterator of output sequence.
126  */
127 template <typename ValueType, typename ReaderIterator,
128  typename Comparator = std::less<ValueType> >
129 auto make_multiway_merge_tree(
130  ReaderIterator seqs_begin, ReaderIterator seqs_end,
131  const Comparator& comp = Comparator()) {
132 
133  assert(seqs_end - seqs_begin >= 1);
134  return MultiwayMergeTree<ValueType, ReaderIterator, Comparator>(
135  seqs_begin, seqs_end, comp);
136 }

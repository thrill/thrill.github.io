168  //! InputLineIterator gives you access to lines of a file
169  class InputLineIteratorUncompressed : public InputLineIterator
170  {
171  public:
172  //! Creates an instance of iterator that reads file line based
173  InputLineIteratorUncompressed(const vfs::FileList& files,
174  ReadLinesNode& node, bool local_storage)
175  : InputLineIterator(files, node) {
176 
177  // Go to start of 'local part'.
178  if (local_storage) {
179  my_range_ = node_.context_.CalculateLocalRangeOnHost(
180  files.total_size);
181  }
182  else {
183  my_range_ = node_.context_.CalculateLocalRange(
184  files.total_size);
185  }
186 
187  assert(my_range_.begin <= my_range_.end);
188  if (my_range_.begin == my_range_.end) return;
189 
190  file_nr_ = 0;
191 
192  while (files_[file_nr_].size_inc_psum() <= my_range_.begin) {
193  file_nr_++;
194  }
195 
196  offset_ = my_range_.begin - files_.size_ex_psum(file_nr_);
197 
198  sLOG << "ReadLines: opening file" << file_nr_
199  << "my_range_" << my_range_ << "offset_" << offset_;
200 
201  // find offset in current file:
202  // offset = start - sum of previous file sizes
203  stream_ = vfs::OpenReadStream(
204  files_[file_nr_].path, common::Range(offset_, 0));
205 
206  buffer_.Reserve(read_size);
207  ReadBlock(stream_, buffer_);
208 
209  if (offset_ != 0) {
210  bool found_n = false;
211 
212  // find next newline, discard all previous data as previous
213  // worker already covers it
214  while (!found_n) {
215  while (current_ < buffer_.end()) {
216  if (THRILL_UNLIKELY(*current_++ == '\n')) {
217  found_n = true;
218  break;
219  }
220  }
221  // no newline found: read new data into buffer_builder
222  if (!found_n) {
223  offset_ += buffer_.size();
224  if (!ReadBlock(stream_, buffer_)) {
225  // EOF = newline per definition
226  found_n = true;
227  }
228  }
229  }
230  }
231  data_.reserve(4 * 1024);
232  }
233 
234  //! returns the next element if one exists
235  //!
236  //! does no checks whether a next element exists!
237  const std::string& Next() {
238  total_elements_++;
239  data_.clear();
240  while (true) {
241  while (THRILL_LIKELY(current_ < buffer_.end())) {
242  if (THRILL_UNLIKELY(*current_ == '\n')) {
243  current_++;
244  return data_;
245  }
246  else {
247  data_.push_back(*current_++);
248  }
249  }
250  offset_ += buffer_.size();
251  if (!ReadBlock(stream_, buffer_)) {
252  LOG << "ReadLines: opening next file";
253 
254  stream_->close();
255  file_nr_++;
256  offset_ = 0;
257 
258  if (file_nr_ < files_.size()) {
259  stream_ = vfs::OpenReadStream(
260  files_[file_nr_].path, common::Range(offset_, 0));
261  offset_ += buffer_.size();
262  ReadBlock(stream_, buffer_);
263  }
264  else {
265  current_ = buffer_.begin() +
266  files_[file_nr_ - 1].size;
267  }
268 
269  if (data_.length()) {
270  return data_;
271  }
272  }
273  }
274  }
275 
276  //! returns true, if an element is available in local part
277  bool HasNext() {
278  size_t pos = current_ - buffer_.begin();
279  assert(current_ >= buffer_.begin());
280  size_t global_index = offset_ + pos + files_.size_ex_psum(file_nr_);
281  return global_index < my_range_.end ||
282  (global_index == my_range_.end &&
283  files_[file_nr_].size > offset_ + pos);
284  }
285 
286  private:
287  //! Offset of current block in stream_.
288  size_t offset_ = 0;
289  //! File handle to files_[file_nr_]
290  vfs::ReadStreamPtr stream_;
291  };

293  //! InputLineIterator gives you access to lines of a file
294  class InputLineIteratorCompressed : public InputLineIterator
295  {
296  public:
297  //! Creates an instance of iterator that reads file line based
298  InputLineIteratorCompressed(const vfs::FileList& files,
299  ReadLinesNode& node, bool local_storage)
300  : InputLineIterator(files, node) {
301 
302  // Go to start of 'local part'.
303  if (local_storage) {
304  my_range_ = node_.context_.CalculateLocalRangeOnHost(
305  files.total_size);
306  }
307  else {
308  my_range_ = node_.context_.CalculateLocalRange(
309  files.total_size);
310  }
311 
312  file_nr_ = 0;
313 
314  while (file_nr_ < files_.size() &&
315  (files_[file_nr_].size_inc_psum() +
316  files_[file_nr_].size_ex_psum) / 2 <= my_range_.begin) {
317  ++file_nr_;
318  }
319 
320  if (file_nr_ == files_.size()) {
321  LOG << "Start behind last file, not reading anything!";
322  return;
323  }
324 
325  for (size_t i = file_nr_; i < files_.size(); i++) {
326  if ((files_[i].size_inc_psum() + files_[i].size_ex_psum) / 2
327  == my_range_.end) {
328  break;
329  }
330  if ((files_[i].size_inc_psum() + files_[i].size_ex_psum) / 2
331  > my_range_.end) {
332  my_range_.end = files_.size_ex_psum(i);
333  break;
334  }
335  }
336 
337  if (my_range_.begin >= my_range_.end) {
338  // No local files, set buffer size to 2, so HasNext() does not try to read
339  LOG << "ReadLines: my_range " << my_range_;
340  buffer_.Reserve(2);
341  buffer_.set_size(2);
342  current_ = buffer_.begin();
343  return;
344  }
345 
346  sLOG << "ReadLines: opening compressed file" << file_nr_
347  << "my_range" << my_range_;
348 
349  stream_ = vfs::OpenReadStream(files_[file_nr_].path);
350 
351  buffer_.Reserve(read_size);
352  ReadBlock(stream_, buffer_);
353  data_.reserve(4 * 1024);
354  }
355 
356  //! returns the next element if one exists
357  //!
358  //! does no checks whether a next element exists!
359  const std::string& Next() {
360  total_elements_++;
361  data_.clear();
362  while (true) {
363  while (current_ < buffer_.end()) {
364  if (THRILL_UNLIKELY(*current_ == '\n')) {
365  current_++;
366  return data_;
367  }
368  else {
369  data_.push_back(*current_++);
370  }
371  }
372 
373  if (!ReadBlock(stream_, buffer_)) {
374  LOG << "ReadLines: opening new compressed file!";
375  stream_->close();
376  file_nr_++;
377 
378  if (file_nr_ < files_.size()) {
379  stream_ = vfs::OpenReadStream(files_[file_nr_].path);
380  ReadBlock(stream_, buffer_);
381  }
382  else {
383  LOG << "ReadLines: reached last file";
384  current_ = buffer_.begin();
385  }
386 
387  if (data_.length()) {
388  LOG << "ReadLines: end - returning string of length"
389  << data_.length();
390  return data_;
391  }
392  }
393  }
394  }
395 
396  //! returns true, if an element is available in local part
397  bool HasNext() {
398  if (files_.size_ex_psum(file_nr_) >= my_range_.end) {
399  return false;
400  }
401 
402  // if block is fully read, read next block. needs to be done here
403  // as HasNext() has to know if file is finished
404  // v-- no new line at end || v-- newline at end of file
405  if (current_ >= buffer_.end() || (current_ + 1 >= buffer_.end() && *current_ == '\n')) {
406  LOG << "ReadLines: new buffer in HasNext()";
407  ReadBlock(stream_, buffer_);
408  if (buffer_.size() > 1 || (buffer_.size() == 1 && buffer_[0] != '\n')) {
409  return true;
410  }
411  else {
412  LOG << "ReadLines: opening new file in HasNext()";
413  // already at last file
414  if (file_nr_ >= files_.size() - 1) {
415  return false;
416  }
417  stream_->close();
418  // if (this worker reads at least one more file)
419  if (my_range_.end > files_[file_nr_].size_inc_psum()) {
420  file_nr_++;
421  stream_ = vfs::OpenReadStream(files_[file_nr_].path);
422  ReadBlock(stream_, buffer_);
423  return true;
424  }
425  else {
426  return false;
427  }
428  }
429  }
430  else {
431  return true;
432  }
433  }
434 
435  private:
436  //! File handle to files_[file_nr_]
437  vfs::ReadStreamPtr stream_;
438  };

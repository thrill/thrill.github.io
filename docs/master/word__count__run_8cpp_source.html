160 int main(int argc, char* argv[]) {
161 
162  common::CmdlineParser clp;
163 
164  std::string output;
165  clp.AddString('o', "output", output,
166  "output file pattern");
167 
168  std::vector<std::string> input;
169  clp.AddParamStringlist("input", input,
170  "input file pattern(s)");
171 
172  bool generate = false;
173  clp.AddFlag('g', "generate", generate,
174  "generate random words, first file pattern "
175  "specifies approximately how many.");
176 
177  bool hash_words = false;
178  clp.AddFlag('H', "hash_words", hash_words,
179  "explicitly calculate hash values for words "
180  "to accelerate reduction.");
181 
182  if (!clp.Process(argc, argv)) {
183  return -1;
184  }
185 
186  clp.PrintResult();
187 
188  return api::Run(
189  [&](api::Context& ctx) {
190  if (generate) {
191  size_t num_words;
192  if (!common::from_str<size_t>(input[0], num_words))
193  die("For generated word data, set input to the number of words.");
194 
195  if (hash_words)
196  RunHashWordCountGenerated(ctx, num_words, output);
197  else
198  RunWordCountGenerated(ctx, num_words, output);
199  }
200  else {
201  if (hash_words)
202  RunHashWordCount(ctx, input, output);
203  else
204  RunWordCount(ctx, input, output);
205  }
206  });
207 }

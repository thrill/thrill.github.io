102  template <bool DoCache>

104  Table& table, std::vector<RangeFilePair>& remaining_files,
105  bool consume, data::File::Writer* writer = nullptr) {
106 
107  std::vector<data::File>& files = table.partition_files();
108 
109  size_t id = 0;

111  {
112  // get the actual reader from the file
113  data::File& file = files[id];
114 
115  if (file.num_items() > 0) {
116  // if items have been spilled, switch to second loop, which
117  // stores items for a second reduce
118  break;
119  }
120  else {
121  // calculate key range for the file
122  common::Range file_range = table.key_range(id);
123 
124  sLOG << "partition" << id << "range" << file_range
125  << "contains" << table.items_per_partition(id)
126  << "fully reduced items";

151  neutral_element_, table_.key_extractor());
152  emitter_.Emit(kv);
153  if (DoCache) writer->Put(kv);

155  }
156  }
157 
158  for ( ; id < files.size(); ++id)
159  {
160  // get the actual reader from the file
161  data::File& file = files[id];

163  // calculate key range for the file

171  << "contains" << file.num_items()

175  RangeFilePair(file_range, std::move(file)));
176  }
177  else {

179  // memory due to a second reduce, which is necessary.
180  sLOG << "partition" << id << "range" << file_range
181  << "contains" << table.items_per_partition(id)
182  << "fully reduced items";

188  file_range.Swap();

193  }

196  //! Flushes all items in the whole table. Since we have to flush recursively
197  //! such that the order of all indexes remains correct, we use an imaginary
198  //! deque of remaining files. In each iteration, the first remaining file is
199  //! further reduced and replaced by more files if necessary. Since the deque
200  //! is only extended in the front, we use a vector in reverse order.
201  template <bool DoCache>
202  void Flush(bool consume, data::File::Writer* writer = nullptr) {
203  LOG << "Flushing items";
204 
205  // list of remaining files, containing only partially reduced item pairs
206  // or items. in reverse order.
207  std::vector<RangeFilePair> remaining_files;
208 
209  // read primary hash table, since ReduceByHash delivers items in any
210  // order, we can just emit items from fully reduced partitions.
211  FlushTableInto<DoCache>(table_, remaining_files, consume, writer);
212 
213  if (remaining_files.size() == 0) {
214  LOG << "Flushed items directly.";
215  return;
216  }
217 
218  table_.Dispose();
219 
220  assert(consume && "Items were spilled hence Flushing must consume");
221 
222  // reverse order in remaining files
223  std::reverse(remaining_files.begin(), remaining_files.end());
224 
225  // if partially reduce files remain, create new hash tables to process
226  // them iteratively.
227 
228  Table subtable(
229  table_.ctx(), table_.dia_id(),
230  table_.key_extractor(), table_.reduce_function(), emitter_,
231  /* num_partitions */ 32, config_, false,
232  table_.index_function(),
233  table_.key_equal_function());
234 
235  subtable.Initialize(table_.limit_memory_bytes());
236 
237  size_t iteration = 1;
238 
239  sLOG << "ReduceToIndexPostPhase: re-reducing items from"
240  << remaining_files.size() << "spilled files";
241 
242  while (remaining_files.size())
243  {
244  sLOG << "ReduceToIndexPostPhase: re-reducing items from"
245  << remaining_files.size() << "remaining files"
246  << "iteration" << iteration;
247 
248  std::vector<RangeFilePair> next_remaining_files;
249 
250  size_t num_subfile = 0;
251 
252  // take last remaining file and reduce or output it.
253  RangeFilePair pair = std::move(remaining_files.back());
254  remaining_files.pop_back();
255 
256  common::Range range = pair.first;
257  data::File file = std::move(pair.second);
258 
259  assert(!range.IsEmpty());
260 
261  if (!range.IsValid())

263  range.Swap();
264 
265  // directly emit all items from the fully reduced file
266  sLOG << "emitting subfile" << num_subfile++
267  << "range" << range;
268 
269  data::File::ConsumeReader reader = file.GetConsumeReader();
270 
271  size_t index = range.begin;
272 
273  while (reader.HasNext()) {
274  TableItem p = reader.Next<TableItem>();
275 
276  for ( ; index < key(p); ++index) {
277  TableItem kv = MakeTableItem::Make(
278  neutral_element_, table_.key_extractor());
279  emitter_.Emit(kv);
280  if (DoCache) writer->Put(kv);
281 
282  // sLOG << "emit hole" << index << "-" << neutral_element_;
283  }
284 
285  // sLOG << "emit" << p.first << "-" << p.second;
286  emitter_.Emit(p);
287  if (DoCache) writer->Put(p);
288  ++index;
289  }
290 
291  for ( ; index < range.end; ++index) {
292  TableItem kv = MakeTableItem::Make(
293  neutral_element_, table_.key_extractor());
294  emitter_.Emit(kv);
295  if (DoCache) writer->Put(kv);
296  }
297  }
298  else
299  {
300  // change table's range
301  subtable.index_function().set_range(range);
302 
303  // insert all items from the partially reduced file
304  sLOG << "re-reducing subfile" << num_subfile++
305  << "range" << range;
306 
307  data::File::ConsumeReader reader = file.GetConsumeReader();
308 
309  while (reader.HasNext()) {
310  subtable.Insert(reader.Next<TableItem>());
311  }
312 
313  // after insertion, flush fully reduced partitions and save
314  // remaining files for next iteration.
315 
316  FlushTableInto<DoCache>(
317  subtable, next_remaining_files, /* consume */ true, writer);
318 
319  for (auto it = next_remaining_files.rbegin();
320  it != next_remaining_files.rend(); ++it)
321  {
322  remaining_files.emplace_back(std::move(*it));
323  }
324 
325  ++iteration;
326  }
327  }

329  LOG << "Flushed items";

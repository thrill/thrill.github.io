39 /*!
40  * A DIANode which performs a ReduceToIndex operation. ReduceToIndex groups the
41  * elements in a DIA by their key and reduces every key bucket to a single
42  * element each. The ReduceToIndexNode stores the key_extractor and the
43  * reduce_function UDFs. The chainable LOps ahead of the Reduce operation are
44  * stored in the Stack. The ReduceToIndexNode has the type ValueType, which is
45  * the result type of the reduce_function. The key type is an unsigned integer
46  * and the output DIA will have element with key K at index K.
47  *
48  * \tparam ParentType Input type of the Reduce operation
49  * \tparam ValueType Output type of the Reduce operation
50  * \tparam ParentStack Function stack, which contains the chained lambdas between the last and this DIANode.
51  * \tparam KeyExtractor Type of the key_extractor function.
52  * \tparam ReduceFunction Type of the reduce_function
53  *
54  * \ingroup api_layer
55  */
56 template <typename ValueType,
57  typename KeyExtractor, typename ReduceFunction,
58  typename ReduceConfig, bool VolatileKey>
59 class ReduceToIndexNode final : public DOpNode<ValueType>
60 {
61  static constexpr bool debug = false;
62 
63  using Super = DOpNode<ValueType>;
64  using Super::context_;
65 
66  using Key = typename common::FunctionTraits<KeyExtractor>::result_type;
67 
68  using TableItem =
69  typename std::conditional<
70  VolatileKey, std::pair<Key, ValueType>, ValueType>::type;
71 
72  static_assert(std::is_same<Key, size_t>::value,
73  "Key must be an unsigned integer");
74

76  static constexpr bool use_post_thread_ = ReduceConfig::use_post_thread_;

96  template <typename ParentDIA>
97  ReduceToIndexNode(const ParentDIA& parent,
98  const char* label,
99  const KeyExtractor& key_extractor,
100  const ReduceFunction& reduce_function,
101  size_t result_size,
102  const ValueType& neutral_element,
103  const ReduceConfig& config)
104  : Super(parent.ctx(), label, { parent.id() }, { parent.node() }),
105  mix_stream_(use_mix_stream_ ?
106  parent.ctx().GetNewMixStream(this) : nullptr),
107  cat_stream_(use_mix_stream_ ?
108  nullptr : parent.ctx().GetNewCatStream(this)),
109  emitters_(use_mix_stream_ ?
110  mix_stream_->GetWriters() : cat_stream_->GetWriters()),
111  result_size_(result_size),
112  pre_phase_(
113  context_, Super::id(), context_.num_workers(),
114  key_extractor, reduce_function, emitters_,
115  config, core::ReduceByIndex<Key>(0, result_size)),
116  post_phase_(
117  context_, Super::id(),
118  key_extractor, reduce_function, Emitter(this),
119  config, neutral_element)
120  {
121  // Hook PreOp: Locally hash elements of the current DIA onto buckets and
122  // reduce each bucket to a single value, afterwards send data to another
123  // worker given by the shuffle algorithm.
124  auto pre_op_fn = [this](const ValueType& input) {
125  return pre_phase_.Insert(input);
126  };
127 
128  // close the function stack with our pre op and register it at parent
129  // node for output
130  auto lop_chain = parent.stack().push(pre_op_fn).fold();
131  parent.node()->AddChild(this, lop_chain);
132  }

135  // request maximum RAM limit, the value is calculated by StageBuilder,
136  // and set as DIABase::mem_limit_.
137  return DIAMemUse::Max();
138  }

140  void StartPreOp(size_t /* id */) final {
141  if (!use_post_thread_) {
142  // use pre_phase without extra thread
143  pre_phase_.Initialize(DIABase::mem_limit_);
144 
145  // re-parameterize with resulting key range on this worker - this is
146  // only know after Initialize() of the pre_phase_.
147  post_phase_.SetRange(pre_phase_.key_range(context_.my_rank()));
148  }
149  else {
150  pre_phase_.Initialize(DIABase::mem_limit_ / 2);
151 
152  // re-parameterize with resulting key range on this worker - this is
153  // only know after Initialize() of the pre_phase_.
154  post_phase_.SetRange(pre_phase_.key_range(context_.my_rank()));
155  post_phase_.Initialize(DIABase::mem_limit_ / 2);
156 
157  // start additional thread to receive from the channel
158  thread_ = common::CreateThread([this] { ProcessChannel(); });
159  }
160  }
161 
162  void StopPreOp(size_t /* id */) final {
163  LOG << *this << " running StopPreOp";
164  // Flush hash table before the postOp
165  pre_phase_.FlushAll();
166  pre_phase_.CloseAll();
167  // waiting for the additional thread to finish the reduce
168  if (use_post_thread_) thread_.join();
169  use_mix_stream_ ? mix_stream_->Close() : cat_stream_->Close();
170  }
171 
172  void Execute() final { }
173 
174  DIAMemUse PushDataMemUse() final {
175  return DIAMemUse::Max();
176  }
177 
178  void PushData(bool consume) final {
179 
180  if (!use_post_thread_ && !reduced_) {
181  // not final reduced, and no additional thread, perform post reduce
182  post_phase_.Initialize(DIABase::mem_limit_);
183  ProcessChannel();
184 
185  reduced_ = true;
186  }
187  post_phase_.PushData(consume);
188  }
189 
190  //! process the inbound data in the post reduce phase
191  void ProcessChannel() {
192  if (use_mix_stream_)
193  {
194  auto reader = mix_stream_->GetMixReader(/* consume */ true);
195  sLOG << "reading data from" << mix_stream_->id()
196  << "to push into post table which flushes to" << this->id();
197  while (reader.HasNext()) {
198  post_phase_.Insert(reader.template Next<TableItem>());
199  }
200  }
201  else
202  {
203  auto reader = cat_stream_->GetCatReader(/* consume */ true);
204  sLOG << "reading data from" << cat_stream_->id()
205  << "to push into post table which flushes to" << this->id();
206  while (reader.HasNext()) {
207  post_phase_.Insert(reader.template Next<TableItem>());
208  }
209  }
210  }
211 
212  void Dispose() final {
213  post_phase_.Dispose();
214  }

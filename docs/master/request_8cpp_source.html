53 void Request::check_alignment() const {
54  if (offset_ % THRILL_DEFAULT_ALIGN != 0)
55  LOG1 << "Offset is not aligned: modulo "
56  << THRILL_DEFAULT_ALIGN << " = " << offset_ % THRILL_DEFAULT_ALIGN;
57 
58  if (bytes_ % THRILL_DEFAULT_ALIGN != 0)
59  LOG1 << "Size is not a multiple of "
60  << THRILL_DEFAULT_ALIGN << ", = " << bytes_ % THRILL_DEFAULT_ALIGN;
61 
62  if (uintptr_t(buffer_) % THRILL_DEFAULT_ALIGN != 0)
63  LOG1 << "Buffer is not aligned: modulo "
64  << THRILL_DEFAULT_ALIGN << " = " << size_t(buffer_) % THRILL_DEFAULT_ALIGN
65  << " (" << buffer_ << ")";

129 bool Request::cancel() {
130  LOG << "request::cancel() " << file_ << " " << buffer_ << " " << offset_;
131 
132  if (!file_) return false;
133 
134  if (DiskQueues::GetInstance()->CancelRequest(this, file_->get_queue_id()))
135  {
136  state_.set_to(DONE);
137  // user callback
138  if (on_complete_)
139  on_complete_(this, false);
140  file_.reset();
141  state_.set_to(READY2DIE);
142  return true;
143  }
144  return false;
145 }

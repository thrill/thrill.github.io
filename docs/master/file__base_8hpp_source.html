70  //! Definition of acceptable file open modes. Various open modes in a file
71  //! system must be converted to this set of acceptable modes.
72  enum OpenMode {
73  //! only reading of the file is allowed
74  RDONLY = 1,
75  //! only writing of the file is allowed
76  WRONLY = 2,
77  //! read and write of the file are allowed
78  RDWR = 4,
79  //! in case file does not exist no error occurs and file is newly
80  //! created
81  CREAT = 8,
82  //! I/Os proceed bypassing file system buffers, i.e. unbuffered I/O.
83  //! Tries to open with appropriate flags, if fails print warning and
84  //! open normally.
85  DIRECT = 16,
86  //! once file is opened its length becomes zero
87  TRUNC = 32,
88  //! open the file with O_SYNC | O_DSYNC | O_RSYNC flags set
89  SYNC = 64,
90  //! do not acquire an exclusive lock by default
91  NO_LOCK = 128,
92  //! implies DIRECT, fail if opening with DIRECT flag does not work.
93  REQUIRE_DIRECT = 256
94  };

98  bwt

100  return IndexChar { Index(i), c };
101  })
102  .Sort([](const IndexChar& a, const IndexChar& b) {
103  if (a.ch == b.ch)

105  return a.ch < b.ch;
106  });

175  })

179  if (index == 0)

206  });

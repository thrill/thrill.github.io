31 /*!
32  * Storage area allocated on the stack and usable by a StackAllocator.
33  */
34 template <size_t Size>
35 class StackArena
36 {
37  static constexpr size_t alignment = 16;
38 
39  //! stack memory area used for allocations.
40  alignas(alignment) char buf_[Size];
41 
42  //! pointer into free bytes in buf_
43  char* ptr_;
44 
45  //! debug method to check whether ptr_ is still in buf_.
46  bool pointer_in_buffer(char* p) noexcept
47  { return buf_ <= p && p <= buf_ + Size; }
48 
49 public:
50  //! default constructor: free pointer at the beginning.
51  StackArena() noexcept : ptr_(buf_) { }
52 
53  //! destructor clears ptr_ for debugging.
54  ~StackArena() { ptr_ = nullptr; }
55 
56  StackArena(const StackArena&) = delete;
57  StackArena& operator = (const StackArena&) = delete;
58 
59  char * allocate(size_t n) {
60  assert(pointer_in_buffer(ptr_) &&
61  "StackAllocator has outlived StackArena");
62 
63  // try to allocate from stack memory area
64  if (buf_ + Size >= ptr_ + n) {
65  char* r = ptr_;
66  ptr_ += n;
67  if (n % alignment != 0)
68  ptr_ += alignment - n % alignment;
69  return r;
70  }
71  // otherwise fallback to malloc()
72  return static_cast<char*>(malloc(n));
73  }
74 
75  void deallocate(char* p, size_t n) noexcept {
76  assert(pointer_in_buffer(ptr_) &&
77  "StackAllocator has outlived StackArena");
78  if (pointer_in_buffer(p)) {
79  // free memory area (only works for a stack-ordered
80  // allocations/deallocations).
81  if (p + n == ptr_)
82  ptr_ = p;
83  }
84  else {
85  free(p);
86  }
87  }
88 
89  //! size of memory area
90  static constexpr size_t size() { return Size; }
91 
92  //! return number of bytes used in StackArena
93  size_t used() const { return static_cast<size_t>(ptr_ - buf_); }
94 
95  //! reset memory area
96  void reset() { ptr_ = buf_; }
97 };

67 void MemoryFile::discard(offset_type offset, offset_type size) {
68  std::unique_lock<std::mutex> lock(mutex_);
69 #ifndef THRILL_MEMFILE_DONT_CLEAR_FREED_MEMORY
70  // overwrite the freed region with uninitialized memory
71  LOG << "discard at " << offset << " len " << size;
72  void* uninitialized = malloc(THRILL_DEFAULT_ALIGN);
73  while (size >= THRILL_DEFAULT_ALIGN) {
74  memcpy(ptr_ + offset, uninitialized, THRILL_DEFAULT_ALIGN);
75  offset += THRILL_DEFAULT_ALIGN;
76  size -= THRILL_DEFAULT_ALIGN;
77  }
78  assert(size <= std::numeric_limits<offset_type>::max());
79  if (size > 0)
80  memcpy(ptr_ + offset, uninitialized, (size_t)size);
81  free(uninitialized);
82 #else
83  tlx::unused(offset);
84  tlx::unused(size);
85 #endif
86 }

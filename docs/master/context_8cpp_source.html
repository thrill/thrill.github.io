119 //! Generic runner for backends supporting loopback tests.
120 template <typename NetGroup>
121 static inline void
122 RunLoopbackThreads(
123  const MemoryConfig& mem_config,
124  size_t num_hosts, size_t workers_per_host, size_t core_offset,
125  const std::function<void(Context&)>& job_startpoint) {
126 
127  MemoryConfig host_mem_config = mem_config.divide(num_hosts);
128  mem_config.print(workers_per_host);
129 
130  // construct a mock network of hosts
131  std::vector<std::unique_ptr<HostContext> > host_contexts =
132  ConstructLoopbackHostContexts<NetGroup>(
133  host_mem_config, num_hosts, workers_per_host);
134 
135  // launch thread for each of the workers on this host.
136  std::vector<std::thread> threads(num_hosts* workers_per_host);
137 
138  for (size_t host = 0; host < num_hosts; ++host) {
139  mem::by_string log_prefix = "host " + mem::to_string(host);
140  for (size_t worker = 0; worker < workers_per_host; ++worker) {
141  size_t id = host * workers_per_host + worker;
142  threads[id] = common::CreateThread(
143  [&host_contexts, &job_startpoint, host, worker, log_prefix] {
144  Context ctx(*host_contexts[host], worker);
145  common::NameThisThread(
146  log_prefix + " worker " + mem::to_string(worker));
147 
148  ctx.Launch(job_startpoint);
149  });
150  common::SetCpuAffinity(threads[id], core_offset + id);
151  }
152  }
153 
154  // join worker threads
155  for (size_t i = 0; i < num_hosts * workers_per_host; i++) {
156  threads[i].join();
157  }
158 }

313 template <typename NetGroup>
314 static inline
315 int RunBackendLoopback(
316  const char* backend, const std::function<void(Context&)>& job_startpoint) {
317 
318  char* endptr;
319 
320  // determine number of loopback hosts
321 
322  size_t num_hosts = 2;
323 
324  const char* env_local = getenv("THRILL_LOCAL");
325  if (env_local && *env_local) {
326  // parse envvar only if it exists.
327  num_hosts = std::strtoul(env_local, &endptr, 10);
328 
329  if (!endptr || *endptr != 0 || num_hosts == 0) {
330  std::cerr << "Thrill: environment variable"
331  << " THRILL_LOCAL=" << env_local
332  << " is not a valid number of local loopback hosts."
333  << std::endl;
334  return -1;
335  }
336  }
337 
338  const char* env_workers_per_host = getenv("THRILL_WORKERS_PER_HOST");
339 
340  // determine number of threads per loopback host
341 
342  size_t workers_per_host = 1;
343  if (env_workers_per_host && *env_workers_per_host) {
344  workers_per_host = std::strtoul(env_workers_per_host, &endptr, 10);
345  if (!endptr || *endptr != 0 || workers_per_host == 0) {
346  std::cerr << "Thrill: environment variable"
347  << " THRILL_WORKERS_PER_HOST=" << env_workers_per_host
348  << " is not a valid number of workers per host."
349  << std::endl;
350  return -1;
351  }
352  }
353  else {
354  // distribute threads into hosts
355  workers_per_host = std::thread::hardware_concurrency() / num_hosts;
356  }
357 
358  // core offset for pinning
359  const char* env_core_offset = getenv("THRILL_CORE_OFFSET");
360  size_t core_offset = 0;
361  if (env_core_offset && *env_core_offset) {
362  core_offset = std::strtoul(env_core_offset, &endptr, 10);
363 
364  size_t last_core = core_offset + num_hosts * workers_per_host;
365  if (!endptr || *endptr != 0 ||
366  last_core > std::thread::hardware_concurrency())
367  {
368  std::cerr << "Thrill: environment variable"
369  << " THRILL_CORE_OFFSET=" << env_core_offset
370  << " is not a valid number of cores to skip for pinning."
371  << std::endl;
372  return -1;
373  }
374  }
375 
376  // detect memory config
377 
378  MemoryConfig mem_config;
379  if (mem_config.setup_detect() < 0) return -1;
380  mem_config.print(workers_per_host);
381 
382  // okay, configuration is good.
383 
384  std::cerr << "Thrill: running locally with " << num_hosts
385  << " test hosts and " << workers_per_host << " workers per host"
386  << " in a local " << backend << " network." << std::endl;
387 
388  if (!Initialize()) return -1;
389 
390  RunLoopbackThreads<NetGroup>(
391  mem_config, num_hosts, workers_per_host, core_offset, job_startpoint);
392 
393  if (!Deinitialize()) return -1;
394 
395  return 0;
396 }

685  net::ib::Construct(num_hosts, groups.data(), kGroupCount);

688  { std::move(groups[0]), std::move(groups[1]) }

696  std::vector<std::thread> threads(workers_per_host);

698  for (size_t worker = 0; worker < workers_per_host; worker++) {
699  threads[worker] = common::CreateThread(

702  common::NameThisThread("host " + mem::to_string(ctx.host_rank())
703  + " worker " + mem::to_string(worker));

705  ctx.Launch(job_startpoint);
706  });
707  common::SetCpuAffinity(threads[worker], worker);
708  }
709

711  int global_result = 0;
712 
713  for (size_t i = 0; i < workers_per_host; i++) {
714  threads[i].join();
715  }
716 
717  if (!Deinitialize()) return -1;
718 
719  return global_result;

801 int Run(const std::function<void(Context&)>& job_startpoint) {
802 
803  if (RunCheckDieWithParent() < 0)
804  return -1;
805 
806  if (RunCheckUnlinkBinary() < 0)
807  return -1;
808 
809  // parse environment: THRILL_NET
810  const char* env_net = getenv("THRILL_NET");
811 
812  // if no backend configured: automatically select one.
813  if (env_net == nullptr || *env_net == 0) {
814  env_net = DetectNetBackend();
815  if (env_net == nullptr) return -1;
816  }
817 
818  // run with selected backend
819  if (strcmp(env_net, "mock") == 0) {
820  // mock network backend
821  return RunBackendLoopback<net::mock::Group>("mock", job_startpoint);
822  }
823 
824  if (strcmp(env_net, "local") == 0) {
825 #if THRILL_HAVE_NET_TCP
826  // tcp loopback network backend
827  return RunBackendLoopback<net::tcp::Group>("tcp", job_startpoint);
828 #else
829  return RunNotSupported(env_net);
830 #endif
831  }
832 
833  if (strcmp(env_net, "tcp") == 0) {
834 #if THRILL_HAVE_NET_TCP
835  // real tcp network backend
836  return RunBackendTcp(job_startpoint);
837 #else
838  return RunNotSupported(env_net);
839 #endif
840  }
841 
842  if (strcmp(env_net, "mpi") == 0) {
843 #if THRILL_HAVE_NET_MPI
844  // mpi network backend
845  return RunBackendMpi(job_startpoint);
846 #else
847  return RunNotSupported(env_net);
848 #endif
849  }
850 
851  if (strcmp(env_net, "ib") == 0) {
852 #if THRILL_HAVE_NET_IB
853  // ib/mpi network backend
854  return RunBackendIb(job_startpoint);
855 #else
856  return RunNotSupported(env_net);
857 #endif
858  }
859 
860  std::cerr << "Thrill: network backend " << env_net << " is unknown."
861  << std::endl;
862  return -1;
863 }

873 int MemoryConfig::setup_detect() {
874 
875  // determine amount of physical RAM or take user's limit
876 
877  const char* env_ram = getenv("THRILL_RAM");
878 
879  if (env_ram != nullptr && *env_ram != 0) {
880  uint64_t ram64;
881  if (!tlx::parse_si_iec_units(env_ram, &ram64)) {
882  std::cerr << "Thrill: environment variable"
883  << " THRILL_RAM=" << env_ram
884  << " is not a valid amount of RAM memory."
885  << std::endl;
886  return -1;
887  }
888  ram_ = static_cast<size_t>(ram64);
889  }
890  else {
891  // detect amount of physical memory on system
892 #if defined(_MSC_VER)
893  MEMORYSTATUSEX memstx;
894  memstx.dwLength = sizeof(memstx);
895  GlobalMemoryStatusEx(&memstx);
896 
897  ram_ = memstx.ullTotalPhys;
898 #elif __APPLE__
899  int mib[2];
900  int64_t physical_memory;
901  size_t length;
902 
903  // Get the physical memory size
904  mib[0] = CTL_HW;
905  mib[1] = HW_MEMSIZE;
906  length = sizeof(physical_memory);
907  sysctl(mib, 2, &physical_memory, &length, nullptr, 0);
908  ram_ = static_cast<size_t>(physical_memory);
909 #else
910  ram_ = sysconf(_SC_PHYS_PAGES) * static_cast<size_t>(sysconf(_SC_PAGESIZE));
911 #endif
912 
913 #if __linux__
914  // use getrlimit() to check user limit on address space
915  struct rlimit rl; // NOLINT
916  if (getrlimit(RLIMIT_AS, &rl) == 0) {
917  if (rl.rlim_cur != 0 && rl.rlim_cur * 3 / 4 < ram_) {
918  ram_ = rl.rlim_cur * 3 / 4;
919  }
920  }
921  else {
922  sLOG1 << "getrlimit(): " << strerror(errno);
923  }
924 #endif
925  }
926 
927  apply();
928 
929  return 0;
930 }

932 void MemoryConfig::apply() {
933  // divide up ram_
934 
935  ram_workers_ = ram_ / 3;
936  ram_block_pool_hard_ = ram_ / 3;
937  ram_block_pool_soft_ = ram_block_pool_hard_ * 9 / 10;
938  ram_floating_ = ram_ - ram_block_pool_hard_ - ram_workers_;
939 
940  // set memory limit, only BlockPool is excluded from malloc tracking, as
941  // only it uses bypassing allocators.
942  mem::set_memory_limit_indication(ram_floating_ + ram_workers_);
943 }

1104 void Context::Launch(const std::function<void(Context&)>& job_startpoint) {
1105  logger_ << "class" << "Context"
1106  << "event" << "job-start";
1107 
1108  common::StatsTimerStart overall_timer;
1109 
1110  try {
1111  job_startpoint(*this);
1112  }
1113  catch (std::exception& e) {
1114  LOG1 << "worker " << my_rank() << " threw " << typeid(e).name();
1115  LOG1 << " what(): " << e.what();
1116 
1117  logger_ << "class" << "Context"
1118  << "event" << "job-exception"
1119  << "exception" << typeid(e).name()
1120  << "what" << e.what();
1121  throw;
1122  }
1123 
1124  logger_ << "class" << "Context"
1125  << "event" << "job-done"
1126  << "elapsed" << overall_timer;
1127 
1128  // wait for all local workers to finish, prior to closing multiplexer's
1129  // streams.
1130  net.LocalBarrier();
1131 
1132  if (my_rank() == 0)
1133  multiplexer_.Close();
1134 
1135  net.LocalBarrier();
1136 
1137  overall_timer.Stop();
1138 
1139  // collect overall statistics
1140  OverallStats stats;
1141  stats.runtime = overall_timer.SecondsDouble();
1142 
1143  stats.max_block_bytes =
1144  local_worker_id_ == 0 ? block_pool().max_total_bytes() : 0;
1145 
1146  std::tie(stats.net_traffic_tx, stats.net_traffic_rx) =
1147  local_worker_id_ == 0 ? net_manager_.Traffic()
1148  : std::pair<size_t, size_t>(0, 0);
1149 
1150  if (local_host_id_ == 0 && local_worker_id_ == 0) {
1151  io::StatsData io_stats(*io::Stats::GetInstance());
1152  stats.io_volume = io_stats.read_volume() + io_stats.write_volume();
1153  stats.io_max_allocation =
1154  io::BlockManager::GetInstance()->maximum_allocation();
1155  }
1156  else {
1157  stats.io_volume = 0;
1158  stats.io_max_allocation = 0;
1159  }
1160 
1161  LOG0 << stats;
1162 
1163  stats = net.Reduce(stats);
1164 
1165  if (my_rank() == 0) {
1166  using tlx::format_iec_units;
1167 
1168  if (stats.net_traffic_rx != stats.net_traffic_tx)
1169  LOG1 << "Manager::Traffic() tx/rx asymmetry = "
1170  << tlx::abs_diff(stats.net_traffic_tx, stats.net_traffic_rx);
1171 
1172  if (mem_config().verbose_) {
1173  std::cerr
1174  << "Thrill:"
1175  << " ran " << stats.runtime << "s with max "
1176  << format_iec_units(stats.max_block_bytes) << "B in DIA Blocks, "
1177  << format_iec_units(stats.net_traffic_tx) << "B network traffic, "
1178  << format_iec_units(stats.io_volume) << "B disk I/O, and "
1179  << format_iec_units(stats.io_max_allocation) << "B max disk use."
1180  << std::endl;
1181  }
1182 
1183  logger_ << "class" << "Context"
1184  << "event" << "summary"
1185  << "runtime" << stats.runtime
1186  << "net_traffic" << stats.net_traffic_tx
1187  << "io_volume" << stats.io_volume
1188  << "io_max_allocation" << stats.io_max_allocation;
1189  }

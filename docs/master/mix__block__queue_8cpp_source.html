149  : mix_queue_(mix_queue),
150  reread_(mix_queue.read_closed()) {
151 
152  if (!reread_) {

154  available_at_.resize(mix_queue_.num_workers_, 0);

166  consume, local_worker_id));
167  }
168  // move BlockQueueSources into concatenation BlockSource, and to Reader.
169  cat_reader_ = CatBlockReader(CatBlockSource(std::move(result)));

177 bool MixBlockQueueReader::PullBlock() {
178  // no full item available: get next block from mix queue
179  while (available_ == 0) {
180 
181  MixBlockQueue::SrcBlockPair src_blk = mix_queue_.Pop();
182  LOG << "MixBlockQueueReader::PullBlock()"
183  << " still open_=" << open_
184  << " src=" << src_blk.src << " block=" << src_blk.block
185  << " selected_=" << selected_
186  << " available_=" << available_
187  << " available_at_[src]=" << available_at_[src_blk.src];
188 
189  assert(src_blk.src < readers_.size());
190 
191  if (src_blk.block.IsValid()) {
192  // block for this reader.
193  selected_ = src_blk.src;
194 
195  size_t num_items = src_blk.block.num_items();
196 
197  // save block with data for reader
198  mix_queue_.queues_[src_blk.src].AppendBlock(
199  std::move(src_blk.block), /* is_last_block */ false);
200 
201  // add available items: one less than in the blocks.
202  available_at_[src_blk.src] += num_items;
203  available_ = available_at_[src_blk.src] - 1;
204  available_at_[src_blk.src] -= available_;
205  }
206  else {
207  // close block received: maybe get last item
208  assert(open_ > 0);
209  --open_;
210 
211  // save block with data for reader
212  mix_queue_.queues_[src_blk.src].AppendBlock(
213  std::move(src_blk.block), /* is_last_block */ false);
214 
215  // check if we can still read the last item
216  if (available_at_[src_blk.src]) {
217  assert(available_at_[src_blk.src] == 1);
218  selected_ = src_blk.src;
219  available_ = available_at_[src_blk.src];
220  available_at_[src_blk.src] -= available_;
221  }
222  else if (open_ == 0) return false;
223  }
224 
225  LOG << "MixBlockQueueReader::PullBlock() afterwards"
226  << " selected_=" << selected_
227  << " available_=" << available_
228  << " available_at_[src]=" << available_at_[src_blk.src];
229  }
230  return true;
231 }

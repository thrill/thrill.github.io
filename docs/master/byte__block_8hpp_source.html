41 /*!
42  * A ByteBlock is the basic storage units of containers like File, BlockQueue,
43  * etc. It consists of a fixed number of bytes without any type and meta
44  * information. Conceptually a ByteBlock is written _once_ and can then be
45  * shared read-only between containers using CountingPtr<const ByteBlock>
46  * reference counting inside a Block, which adds meta information.
47  *
48  * ByteBlocks can be swapped to disk, which decreases their size to 0.
49  */

167 /*!
168  * A pinned / pin-counted pointer to a ByteBlock. By holding a pin, it is a
169  * guaranteed that the ByteBlock's underlying memory is loaded in RAM. Since
170  * pins are counted per thread, the PinnedByteBlockPtr is a counting pointer
171  * plus a thread id.
172  *
173  * Be careful to move PinnedByteBlockPtr as must as possible, since copying
174  * costs a pinning and an unpinning operation, whereas moving is free.
175  */
176 class PinnedByteBlockPtr : public ByteBlockPtr
177 {
178 public:
179  //! default ctor: contains a nullptr pointer.
180  PinnedByteBlockPtr() noexcept = default;
181 
182  //! copy-ctor: increment underlying's pin count
183  PinnedByteBlockPtr(const PinnedByteBlockPtr& pbb) noexcept
184  : ByteBlockPtr(pbb), local_worker_id_(pbb.local_worker_id_) {
185  if (valid()) get()->IncPinCount(local_worker_id_);
186  }
187 
188  //! move-ctor: move underlying's pin
189  PinnedByteBlockPtr(PinnedByteBlockPtr&& pbb) noexcept
190  : ByteBlockPtr(std::move(pbb)), local_worker_id_(pbb.local_worker_id_) {
191  assert(!pbb.valid());
192  }
193 
194  //! copy-assignment: transfer underlying's pin count
195  PinnedByteBlockPtr& operator = (PinnedByteBlockPtr& pbb) noexcept {
196  if (this == &pbb) return *this;
197  // first acquire other's pin count
198  if (pbb.valid()) pbb->IncPinCount(pbb.local_worker_id_);
199  // then release the current one
200  if (valid()) get()->DecPinCount(local_worker_id_);
201  // copy over information, keep pin
202  ByteBlockPtr::operator = (pbb);
203  local_worker_id_ = pbb.local_worker_id_;
204  return *this;
205  }
206 
207  //! move-assignment: move underlying's pin
208  PinnedByteBlockPtr& operator = (PinnedByteBlockPtr&& pbb) noexcept {
209  if (this == &pbb) return *this;
210  // release the current one
211  if (valid()) get()->DecPinCount(local_worker_id_);
212  // move over information, keep other's pin
213  ByteBlockPtr::operator = (std::move(pbb));
214  local_worker_id_ = pbb.local_worker_id_;
215  // invalidated other block
216  assert(!pbb.valid());
217  return *this;
218  }
219 
220  //! destructor: remove pin
221  ~PinnedByteBlockPtr() {
222  if (valid()) get()->DecPinCount(local_worker_id_);
223  }
224 
225  //! local worker id of holder of pin
226  size_t local_worker_id() const { return local_worker_id_; }
227 
228 private:
229  //! protected ctor for calling from Acquire().
230  PinnedByteBlockPtr(ByteBlock* ptr, size_t local_worker_id) noexcept
231  : ByteBlockPtr(ptr), local_worker_id_(local_worker_id) { }
232 
233  //! protected ctor for calling from Acquire().
234  PinnedByteBlockPtr(ByteBlockPtr&& ptr, size_t local_worker_id) noexcept
235  : ByteBlockPtr(std::move(ptr)), local_worker_id_(local_worker_id) { }
236 
237  //! local worker id of holder of pin
238  size_t local_worker_id_;
239 
240  //! for access to protected constructor to transfer pin
241  friend class PinnedBlock;
242  //! for access to protected constructor to AllocateByteBlock().
243  friend class BlockPool;
244 };

41 /*!
42  * A File is an ordered sequence of Block objects for storing items. By using
43  * the Block indirection, the File can be composed using existing Block objects
44  * (via reference counting), but only contain a subset of the items in those
45  * Blocks. This may be used for Zip() and Repartition().
46  *
47  * A File can be written using a BlockWriter instance, which is delivered by
48  * GetWriter(). Thereafter it can be read (multiple times) using a BlockReader,
49  * delivered by GetReader().
50  *
51  * Using a prefixsum over the number of items in a Block, one can seek to the
52  * block contained any item offset in log_2(Blocks) time, though seeking within
53  * the Block goes sequentially.
54  */

387 //! Get BlockReader seeked to the corresponding item index
388 template <typename ItemType>
389 typename File::KeepReader
390 File::GetReaderAt(size_t index, size_t prefetch) const {
391  static constexpr bool debug = false;
392 
393  // perform binary search for item block with largest exclusive size
394  // prefixsum less or equal to index.
395  auto it =
396  std::lower_bound(num_items_sum_.begin(), num_items_sum_.end(), index);
397 
398  if (it == num_items_sum_.end())
399  die("Access beyond end of File?");
400 
401  size_t begin_block = it - num_items_sum_.begin();
402 
403  sLOG << "File::GetReaderAt()"
404  << "item" << index << "in block" << begin_block
405  << "psum" << num_items_sum_[begin_block]
406  << "first_item" << blocks_[begin_block].first_item_absolute();
407 
408  // start Reader at given first valid item in located block
409  KeepReader fr(
410  KeepFileBlockSource(*this, local_worker_id_, prefetch,
411  begin_block,
412  blocks_[begin_block].first_item_absolute()));
413 
414  // skip over extra items in beginning of block
415  size_t items_before = it == num_items_sum_.begin() ? 0 : *(it - 1);
416 
417  sLOG << "File::GetReaderAt()"
418  << "items_before" << items_before << "index" << index
419  << "delta" << (index - items_before);
420  assert(items_before <= index);
421 
422  // use fixed_size information to accelerate jump.
423  if (Serialization<KeepReader, ItemType>::is_fixed_size)
424  {
425  // fetch a Block to get typecode_verify flag
426  fr.HasNext();
427 
428  const size_t skip_items = index - items_before;
429  const size_t bytes_per_item =
430  (fr.typecode_verify() ? sizeof(size_t) : 0)
431  + Serialization<KeepReader, ItemType>::fixed_size;
432 
433  fr.Skip(skip_items, skip_items * bytes_per_item);
434  }
435  else
436  {
437  for (size_t i = items_before; i < index; ++i) {
438  if (!fr.HasNext())
439  die("Underflow in GetItemRange()");
440  fr.template Next<ItemType>();
441  }
442  }
443 
444  sLOG << "File::GetReaderAt()"
445  << "after seek at" << fr.CopyBlock();
446 
447  return fr;
448 }

456 template <typename ItemType, typename CompareFunction>
457 size_t File::GetIndexOf(
458  const ItemType& item, size_t tie, size_t left, size_t right,
459  const CompareFunction& less) const {
460 
461  static constexpr bool debug = false;
462 
463  static_assert(
464  std::is_convertible<
465  bool, typename common::FunctionTraits<CompareFunction>::result_type
466  >::value,
467  "Comperator must return boolean.");
468 
469  LOG << "File::GetIndexOf() looking for item " << item << " tie " << tie
470  << " in range [" << left << "," << right << ") ="
471  << " size " << right - left;
472 
473  assert(left <= right);
474  assert(left <= num_items());
475  assert(right <= num_items());
476 
477  // Use a binary search to find the item.
478  while (left < right) {
479  size_t mid = (right + left) >> 1;
480  LOG << "left: " << left << "right: " << right << "mid: " << mid;
481  ItemType cur = GetItemAt<ItemType>(mid);
482  LOG << "Item at mid: " << cur;
483  if (less(item, cur) ||
484  (!less(item, cur) && !less(cur, item) && tie <= mid)) {
485  right = mid;
486  }
487  else {
488  left = mid + 1;
489  }
490  }
491 
492  LOG << "found insert position at: " << left;
493  return left;
494 }

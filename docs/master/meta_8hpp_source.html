28 // SFINAE check whether a class method exists.
29 
30 // based on http://stackoverflow.com/questions/257288/is-it-possible
31 // -to-write-a-c-template-to-check-for-a-functions-existence
32 
33 //! macro template for class method / attribute SFINAE test
34 #define THRILL_MAKE_METHOD_TEST(Method) \
35  template <typename Type> \
36  class has_method_ ## Method \
37  { \
38  template <typename C> \
39  static char test(decltype(&C::Method)); \
40  template <typename C> \
41  static int test(...); \
42  public: \
43  enum { value = sizeof(test<Type>(0)) == sizeof(char) }; \
44  };

60 // Tuple Applier: takes a std::tuple<> and applies a variadic template function
61 // to it. Hence, this expands the content of the tuple as the arguments.
62 
63 template <typename Functor, typename Tuple, std::size_t ... Is>
64 auto ApplyTupleImpl(Functor&& f, Tuple&& t, common::index_sequence<Is ...>) {
65  return std::forward<Functor>(f)(
66  std::get<Is>(std::forward<Tuple>(t)) ...);
67 }
68 
69 //! Call the functor f with the contents of t as arguments.
70 template <typename Functor, typename Tuple>
71 auto ApplyTuple(Functor&& f, Tuple&& t) {
72  using Indices = common::make_index_sequence<
73  std::tuple_size<std::decay_t<Tuple> >::value>;
74  return ApplyTupleImpl(std::forward<Functor>(f), std::forward<Tuple>(t),
75  Indices());
76 }

142 // Variadic Template Expander: run a generic templated functor (like a generic
143 // lambda) for each of the variadic template parameters. Called with
144 // func(IndexSaver<> index, Argument arg).

146 //! Helper for VariadicCallForeachIndex to save the index as a compile-time
147 //! index
148 template <size_t Index>
149 struct IndexSaver {
150  //! compile-time index
151  static constexpr size_t index = Index;
152 
153  //! implicit conversion to a run-time index.
154  operator size_t () const { return index; }
155 };
156 
157 //! helper for VariadicCallForeachIndex: base case
158 template <size_t Index, typename Functor, typename Arg>
159 void VariadicCallForeachIndexImpl(Functor&& f, Arg&& arg) {
160  std::forward<Functor>(f)(IndexSaver<Index>(), std::forward<Arg>(arg));
161 }

163 //! helper for VariadicCallForeachIndex: general recursive case
164 template <size_t Index, typename Functor, typename Arg, typename ... MoreArgs>
165 void VariadicCallForeachIndexImpl(
166  Functor&& f, Arg&& arg, MoreArgs&& ... rest) {
167  std::forward<Functor>(f)(IndexSaver<Index>(), std::forward<Arg>(arg));
168  VariadicCallForeachIndexImpl<Index + 1>(
169  std::forward<Functor>(f), std::forward<MoreArgs>(rest) ...);

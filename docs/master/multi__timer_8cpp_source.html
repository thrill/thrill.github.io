33 MultiTimer::Entry& MultiTimer::find_or_create(const char* name) {
34  uint32_t hash = hash_djb2(name);
35  for (size_t i = 0; i < timers_.size(); ++i) {
36  if (timers_[i].hash == hash && strcmp(timers_[i].name, name) == 0)
37  return timers_[i];
38  }
39  Entry new_entry;
40  new_entry.hash = hash;
41  new_entry.name = name;
42  new_entry.duration = std::chrono::duration<double>::zero();
43  timers_.emplace_back(new_entry);
44  return timers_.back();
45 }

47 void MultiTimer::start(const char* timer) {
48  tlx_die_unless(timer);
49  uint32_t hash = hash_djb2(timer);
50  if (hash == running_hash_ && strcmp(running_, timer) == 0) {
51  static bool warning_shown = false;
52  if (!warning_shown) {
53  TLX_LOG1 << "MultiTimer: trying to start timer "
54  << timer << " twice!";
55  TLX_LOG1 << "MultiTimer: multi-threading is not supported, "
56  << "use .add()";
57  warning_shown = true;
58  }
59  }
60  stop();
61  running_ = timer;
62  running_hash_ = hash;
63 }

65 void MultiTimer::stop() {
66  auto new_time_point = std::chrono::high_resolution_clock::now();
67  if (running_) {
68  Entry& e = find_or_create(running_);
69  e.duration += new_time_point - time_point_;
70  total_duration_ += new_time_point - time_point_;
71  }
72  time_point_ = new_time_point;
73  running_ = nullptr;
74  running_hash_ = 0;
75 }

108 MultiTimer& MultiTimer::add(const MultiTimer& b) {
109  std::unique_lock<std::mutex> lock(s_timer_add_mutex);
110  if (b.running_) {
111  TLX_LOG1 << "MultiTimer: trying to add running timer";

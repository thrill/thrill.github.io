150 /*!
151  * Simple structure that holds a all stream instances for the workers on the
152  * local host for a given stream id.
153  */
154 template <typename StreamData>
155 class StreamSet : public StreamSetBase
156 {
157 public:
158  using StreamDataPtr = tlx::CountingPtr<StreamData>;
159 
160  //! Creates a StreamSet with the given number of streams (num workers per
161  //! host).
162  StreamSet(Multiplexer& multiplexer, StreamId id,
163  size_t workers_per_host, size_t dia_id) {
164  for (size_t i = 0; i < workers_per_host; ++i) {
165  streams_.emplace_back(
166  tlx::make_counting<StreamData>(multiplexer, id, i, dia_id));
167  }
168  remaining_ = workers_per_host;
169  }
170 
171  //! Returns the stream that will be consumed by the worker with the given
172  //! local id
173  StreamDataPtr Peer(size_t local_worker_id) {
174  assert(local_worker_id < streams_.size());
175  return streams_[local_worker_id];
176  }
177 
178  //! Release local_worker_id, returns true when all individual streams are
179  //! done.
180  bool Release(size_t local_worker_id) {
181  assert(local_worker_id < streams_.size());
182  if (streams_[local_worker_id]) {
183  assert(remaining_ > 0);
184  streams_[local_worker_id].reset();
185  --remaining_;
186  }
187  return (remaining_ == 0);
188  }
189 
190  void Close() final {
191  for (StreamDataPtr& c : streams_)
192  c->Close();
193  }
194 
195 private:
196  //! 'owns' all streams belonging to one stream id for all local workers.
197  std::vector<StreamDataPtr> streams_;
198  //! countdown to destruction
199  size_t remaining_;
200 };

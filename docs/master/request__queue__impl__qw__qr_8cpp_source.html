87 bool RequestQueueImplQwQr::CancelRequest(Request* req) {
88  if (!req)
89  THRILL_THROW_INVALID_ARGUMENT("Empty request canceled disk_queue.");
90  if (thread_state_() != RUNNING)
91  THRILL_THROW_INVALID_ARGUMENT("Request canceled to not running queue.");
92  if (!dynamic_cast<ServingRequest*>(req))
93  LOG1 << "Incompatible request submitted to running queue.";
94 
95  bool was_still_in_queue = false;
96  if (req->type() == Request::READ)
97  {
98  std::unique_lock<std::mutex> lock(read_mutex_);
99  Queue::iterator pos
100  = std::find(read_queue_.begin(), read_queue_.end(), req);
101  if (pos != read_queue_.end())
102  {
103  read_queue_.erase(pos);
104  was_still_in_queue = true;
105  lock.unlock();
106  sem_.wait();
107  }
108  }
109  else
110  {
111  std::unique_lock<std::mutex> lock(write_mutex_);
112  Queue::iterator pos
113  = std::find(write_queue_.begin(), write_queue_.end(), req);
114  if (pos != write_queue_.end())
115  {
116  write_queue_.erase(pos);
117  was_still_in_queue = true;
118  lock.unlock();
119  sem_.wait();
120  }
121  }
122 
123  return was_still_in_queue;
124 }

130 void* RequestQueueImplQwQr::worker(void* arg) {
131  RequestQueueImplQwQr* pthis = static_cast<RequestQueueImplQwQr*>(arg);
132  // pin I/O thread to last core
133  common::SetCpuAffinity(std::thread::hardware_concurrency() - 1);
134 
135  bool write_phase = true;
136  for ( ; ; )
137  {
138  pthis->sem_.wait();
139 
140  if (write_phase)
141  {
142  std::unique_lock<std::mutex> write_lock(pthis->write_mutex_);
143  if (!pthis->write_queue_.empty())
144  {
145  RequestPtr req = pthis->write_queue_.front();
146  pthis->write_queue_.pop_front();
147 
148  write_lock.unlock();
149 
150  // assert(req->get_reference_count()) > 1);
151  dynamic_cast<ServingRequest*>(req.get())->serve();
152  }
153  else
154  {
155  write_lock.unlock();
156 
157  pthis->sem_.signal();
158 
159  if (pthis->priority_op_ == WRITE)
160  write_phase = false;
161  }
162 
163  if (pthis->priority_op_ == NONE || pthis->priority_op_ == READ)
164  write_phase = false;
165  }
166  else
167  {
168  std::unique_lock<std::mutex> read_lock(pthis->read_mutex_);
169 
170  if (!pthis->read_queue_.empty())
171  {
172  RequestPtr req = pthis->read_queue_.front();
173  pthis->read_queue_.pop_front();
174 
175  read_lock.unlock();
176 
177  LOG << "queue: before serve request has " << req->reference_count() << " references ";
178  // assert(req->get_reference_count() > 1);
179  dynamic_cast<ServingRequest*>(req.get())->serve();
180  LOG << "queue: after serve request has " << req->reference_count() << " references ";
181  }
182  else
183  {
184  read_lock.unlock();
185 
186  pthis->sem_.signal();
187 
188  if (pthis->priority_op_ == READ)
189  write_phase = true;
190  }
191 
192  if (pthis->priority_op_ == NONE || pthis->priority_op_ == WRITE)
193  write_phase = true;
194  }
195 
196  // terminate if it has been requested and queues are empty
197  if (pthis->thread_state_() == TERMINATING) {
198  if (pthis->sem_.wait() == 0)
199  break;
200  else
201  pthis->sem_.signal();
202  }
203  }
204 
205  pthis->thread_state_.set_to(TERMINATED);
206 
207 #if THRILL_STD_THREADS && THRILL_MSVC >= 1700
208  // Workaround for deadlock bug in Visual C++ Runtime 2012 and 2013, see
209  // request_queue_impl_worker.cpp. -tb
210  ExitThread(nullptr);
211 #else
212  return nullptr;
213 #endif
214 }

155 //! What happens in the dispatcher thread
156 void DispatcherThread::Work() {
157  common::NameThisThread(name_);
158  // pin DispatcherThread to last core
159  common::SetCpuAffinity(std::thread::hardware_concurrency() - 1);
160 
161  while (!terminate_ ||
162  dispatcher_->HasAsyncWrites() || jobqueue_.size_approx() != 0)
163  {
164  // process jobs in jobqueue_
165  {
166  Job job;
167  while (jobqueue_.try_dequeue(job))
168  job();
169  }
170 
171  // set busy flag, but check once again for jobs.
172  busy_ = true;
173  {
174  Job job;
175  if (jobqueue_.try_dequeue(job)) {
176  busy_ = false;
177  job();
178  continue;
179  }
180  }
181 
182  // run one dispatch
183  dispatcher_->Dispatch();
184 
185  busy_ = false;
186  }
187 }

66  {

79  // in: Length of array_of_requests (integer).

81  // in: Array of requests (array of handles).

83  // out: Number of completed requests (integer).
84  &out_count,
85  // out: Array of indices of operations that completed (array of

88  // out: Array of status objects for operations that completed (array

101  sLOG << "DispatchOne(): MPI_Testsome() out_count=" << out_count;
102 
103  // run through finished requests back to front, swap last entry into
104  // finished ones, such that preceding indexes remain valid.
105  for (int k = out_count - 1; k >= 0; --k) {
106  size_t p = mpi_async_out_[k];
107 
108  // TODO(tb): check for errors?
109 
110  // perform callback
111  mpi_async_[p]();
112 
113  // deleted the entry (no problem is k == len)
114  size_t back = mpi_async_.size() - 1;
115 
116  mpi_async_[p] = std::move(mpi_async_[back]);
117  mpi_async_requests_[p] = std::move(mpi_async_requests_[back]);
118  mpi_async_out_[p] = std::move(mpi_async_out_[back]);
119 
120  mpi_async_.resize(back);
121  mpi_async_requests_.resize(back);
122  mpi_async_out_.resize(back);
123  }
124  }
125  }
126

128  {
129  // use MPI_Iprobe() to check for a new message on this MPI tag.
130  int flag = 0;
131  MPI_Status status;
132 
133  {
134  // lock the GMLIM
135  std::unique_lock<std::mutex> lock(g_mutex);
136 
137  int r = MPI_Iprobe(MPI_ANY_SOURCE, group_tag_, MPI_COMM_WORLD,
138  &flag, &status);
139 
140  if (r != MPI_SUCCESS)
141  throw Exception("Error during MPI_Iprobe()", r);
142  }
143 
144  // check whether probe was successful
145  if (flag == 0) return;
146

149  assert(p >= 0 && static_cast<size_t>(p) < watch_.size());
150

152 
153  if (!w.active) {
154  sLOG << "Got Iprobe() for unwatched peer" << p;
155  return;
156  }
157 
158  sLOG << "Got iprobe for peer" << p;
159 
160  if (w.read_cb.size()) {
161  // run read callbacks until one returns true (in which case it wants
162  // to be called again), or the read_cb list is empty.
163  while (w.read_cb.size() && w.read_cb.front()() == false) {
164  w.read_cb.pop_front();
165  }
166 
167  if (w.read_cb.size() == 0) {
168  w.active = false;
169  watch_active_--;
170  }
171  }
172  else {

174  << p << " without a read handler.";
175  }

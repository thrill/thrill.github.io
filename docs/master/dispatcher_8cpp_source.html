81 void Dispatcher::DispatchOne(const std::chrono::milliseconds& /* timeout */) {
82 
83  // use MPI_Testsome() to check for finished writes
84  if (mpi_async_requests_.size())
85  {
86  // lock the GMLIM
87  std::unique_lock<std::mutex> lock(g_mutex);
88 
89  die_unless(mpi_async_.size() == mpi_async_requests_.size());
90  die_unless(mpi_async_.size() == mpi_async_out_.size());
91 
92 #if 1
93  int out_count;
94 
95  sLOG << "DispatchOne(): MPI_Testsome()"
96  << " mpi_async_requests_=" << mpi_async_requests_.size();
97 
98  int r = MPI_Testsome(
99  // in: Length of array_of_requests (integer).
100  static_cast<int>(mpi_async_requests_.size()),
101  // in: Array of requests (array of handles).
102  mpi_async_requests_.data(),
103  // out: Number of completed requests (integer).
104  &out_count,
105  // out: Array of indices of operations that completed (array of
106  // integers).
107  mpi_async_out_.data(),
108  // out: Array of status objects for operations that completed (array
109  // of status).
110  MPI_STATUSES_IGNORE);
111 
112  lock.unlock();
113 
114  if (r != MPI_SUCCESS)
115  throw Exception("Error during MPI_Testsome()", r);
116 
117  if (out_count == MPI_UNDEFINED) {
118  // nothing returned
119  }
120  else if (out_count > 0) {
121  sLOG << "DispatchOne(): MPI_Testsome() out_count=" << out_count;
122 
123  die_unless(std::is_sorted(mpi_async_out_.begin(),
124  mpi_async_out_.begin() + out_count));
125 
126  // rewrite the arrays, process and remove all finished requests.
127  {
128  size_t i = 0;
129  int k = 0;
130 
131  for (size_t j = 0; j < mpi_async_.size(); ++j)
132  {
133  if (k < out_count && mpi_async_out_[k] == static_cast<int>(j)) {
134 
135  sLOG << "Working #" << k
136  << "which is $" << mpi_async_out_[k];
137 
138  // perform callback
139  mpi_async_[j].DoCallback();
140 
141  // skip over finished request
142  ++k;
143  continue;
144  }
145  if (i != j) {
146  mpi_async_[i] = std::move(mpi_async_[j]);
147  mpi_async_requests_[i] = std::move(mpi_async_requests_[j]);
148  }
149  ++i;
150  }
151 
152  mpi_async_.resize(i);
153  mpi_async_requests_.resize(i);
154  mpi_async_out_.resize(i);
155  }
156  }
157 #else
158  int out_index = 0, out_flag = 0;
159  MPI_Status out_status;
160 
161  sLOG << "DispatchOne(): MPI_Testany()"
162  << " mpi_async_requests_=" << mpi_async_requests_.size();
163 
164  int r = MPI_Testany(
165  // in: Length of array_of_requests (integer).
166  static_cast<int>(mpi_async_requests_.size()),
167  // in: Array of requests (array of handles).
168  mpi_async_requests_.data(),
169  // out: Number of completed request (integer).
170  &out_index,
171  // out: True if one of the operations is complete (logical).
172  &out_flag,
173  // out: Status object (status).
174  &out_status /* MPI_STATUS_IGNORE */);
175 
176  if (r != MPI_SUCCESS)
177  throw Exception("Error during MPI_Testany()", r);
178 
179  if (out_flag == 0) {
180  // nothing returned
181  }
182  else {
183  int out_size;
184  MPI_Get_count(&out_status, MPI_BYTE, &out_size);
185 
186  LOG1 << "DispatchOne(): MPI_Testany() out_flag=" << out_flag
187  << " done #" << out_index
188  << " out_size=" << out_size
189  << " out_tag=" << out_status.MPI_TAG;
190 
191  // perform callback
192  mpi_async_[out_index].DoCallback(out_size);
193 
194  mpi_async_.erase(mpi_async_.begin() + out_index);
195  mpi_async_requests_.erase(mpi_async_requests_.begin() + out_index);
196  mpi_async_out_.erase(mpi_async_out_.begin() + out_index);
197  }
198 
199  lock.unlock();
200 #endif
201  }
202 
203  if (watch_active_)
204  {
205  // use MPI_Iprobe() to check for a new message on this MPI tag.
206  int flag = 0;
207  MPI_Status status;
208 
209  {
210  // lock the GMLIM
211  std::unique_lock<std::mutex> lock(g_mutex);
212 
213  int r = MPI_Iprobe(MPI_ANY_SOURCE, group_tag_, MPI_COMM_WORLD,
214  &flag, &status);
215 
216  if (r != MPI_SUCCESS)
217  throw Exception("Error during MPI_Iprobe()", r);
218  }
219 
220  // check whether probe was successful
221  if (flag == 0) return;
222 
223  // get the right watch
224  int p = status.MPI_SOURCE;
225  assert(p >= 0 && static_cast<size_t>(p) < watch_.size());
226 
227  Watch& w = watch_[p];
228 
229  if (!w.active) {
230  sLOG << "Got Iprobe() for unwatched peer" << p;
231  return;
232  }
233 
234  sLOG << "Got iprobe for peer" << p;
235 
236  if (w.read_cb.size()) {
237  // run read callbacks until one returns true (in which case it wants
238  // to be called again), or the read_cb list is empty.
239  while (w.read_cb.size() && w.read_cb.front()() == false) {
240  w.read_cb.pop_front();
241  }
242 
243  if (w.read_cb.size() == 0) {
244  w.active = false;
245  watch_active_--;
246  }
247  }
248  else {
249  LOG << "Dispatcher: got MPI_Iprobe() for peer "
250  << p << " without a read handler.";
251  }
252  }
253 }

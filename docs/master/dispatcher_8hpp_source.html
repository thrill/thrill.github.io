698  //! \name Dispatch
699  //! \{
700 
701  //! Dispatch one or more events
702  void Dispatch() {
703  // process timer events that lie in the past
704  steady_clock::time_point now = steady_clock::now();
705 
706  while (!terminate_ &&
707  !timer_pq_.empty() &&
708  timer_pq_.top().next_timeout <= now)
709  {
710  const Timer& top = timer_pq_.top();
711  if (top.cb()) {
712  // requeue timeout event again.
713  timer_pq_.emplace(top.next_timeout + top.timeout,
714  top.timeout, top.cb);
715  }
716  timer_pq_.pop();
717  }
718 
719  if (terminate_) return;
720 
721  // calculate time until next timer event
722  if (timer_pq_.empty()) {
723  LOG << "Dispatch(): empty timer queue - selecting for 10s";
724  DispatchOne(milliseconds(10000));
725  }
726  else {
727  auto diff = std::chrono::duration_cast<milliseconds>(
728  timer_pq_.top().next_timeout - now);
729 
730  if (diff < milliseconds(1)) diff = milliseconds(1);
731 
732  sLOG << "Dispatch(): waiting" << diff.count() << "ms";
733  DispatchOne(diff);
734  }
735 
736  // clean up finished AsyncRead/Writes
737  while (async_read_.size() && async_read_.front().IsDone()) {
738  async_read_.pop_front();
739  }
740  while (async_write_.size() && async_write_.front().IsDone()) {
741  async_write_.pop_front();
742  }
743 
744  while (async_read_block_.size() && async_read_block_.front().IsDone()) {
745  async_read_block_.pop_front();
746  }
747  while (async_write_block_.size() && async_write_block_.front().IsDone()) {
748  async_write_block_.pop_front();
749  }
750  }
751 
752  //! Loop over Dispatch() until terminate_ flag is set.
753  void Loop() {
754  while (!terminate_) {
755  Dispatch();
756  }
757  }

832  if (r <= 0) {
833  // these errors are acceptable: just redo the recv later.
834  if (errno == EINTR || errno == EAGAIN) return true;
835 
836  // signal artificial IsDone, for clean up.
837  size_ = buffer_.size();
838 
839  // these errors are end-of-file indications (both good and bad)
840  if (errno == 0 || errno == EPIPE || errno == ECONNRESET) {
841  if (callback_) callback_(*conn_, Buffer());
842  return false;
843  }
844  throw Exception("AsyncReadBuffer() error in recv() on "
845  "connection " + conn_->ToString(), errno);
846  }
847 
848  size_ += r;
849 
850  if (size_ == buffer_.size()) {
851  DoCallback();
852  return false;
853  }
854  else {
855  return true;
856  }
857  }
858 
859  bool IsDone() const { return size_ == buffer_.size(); }
860 
861  //! reference to Buffer
862  Buffer& buffer() { return buffer_; }
863 
864  void DoCallback() {
865  if (callback_) {
866  callback_(*conn_, std::move(buffer_));
867  callback_ = AsyncReadBufferCallback();
868  }
869  }

58 /*!
59  * Generate a Record in a similar way as the "binary" version of Hadoop's
60  * GenSort does. The underlying random generator is different.
61  */
62 class GenerateRecord
63 {
64 public:
65  Record operator () (size_t index) {
66  Record r;
67 
68  // generate random key record
69  for (size_t i = 0; i < 10; ++i)
70  r.key[i] = static_cast<unsigned char>(rng_());
71 
72  uint8_t* v = r.value;
73 
74  // add 2 bytes "break"
75  *v++ = 0x00;
76  *v++ = 0x11;
77 
78  // fill values with hexadecimal representation of the record number
79  static constexpr uint8_t hexdigits[16] = {
80  '0', '1', '2', '3', '4', '5', '6', '7',
81  '8', '9', 'A', 'B', 'C', 'D', 'E', 'F'
82  };
83  uint64_t rec = index;
84  for (size_t i = 0; i != 2 * sizeof(rec); ++i)
85  *v++ = hexdigits[(rec >> (4 * i)) & 0x0F];
86  for (size_t i = 0; i != 16; ++i)
87  *v++ = '0';
88 
89  // add 4 bytes "break"
90  *v++ = 0x88;
91  *v++ = 0x99;
92  *v++ = 0xAA;
93  *v++ = 0xBB;
94 
95  // add 48 byte filler based on index
96  for (size_t i = 0; i < 12; ++i) {
97  uint8_t f = hexdigits[((20 + rec) >> (4 * i)) & 0x0F];
98  *v++ = f;
99  *v++ = f;
100  *v++ = f;
101  *v++ = f;
102  }
103 
104  // add 4 bytes "break"
105  *v++ = 0xCC;
106  *v++ = 0xDD;
107  *v++ = 0xEE;
108  *v++ = 0xFF;
109 
110  assert(v == r.value + 90);
111 
112  return r;
113  }
114 
115 private:
116  std::default_random_engine rng_ { std::random_device { } () };
117 };

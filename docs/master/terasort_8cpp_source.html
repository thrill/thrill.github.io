55 /*!
56  * Generate a Record in a similar way as the "binary" version of Hadoop's
57  * GenSort does. The underlying random generator is different.
58  */
59 class GenerateRecord
60 {
61 public:
62  Record operator () (size_t index) {
63  Record r;
64 
65  // generate random key record
66  for (size_t i = 0; i < 10; ++i)
67  r.key[i] = static_cast<unsigned char>(rng_());
68 
69  uint8_t* v = r.value;
70 
71  // add 2 bytes "break"
72  *v++ = 0x00;
73  *v++ = 0x11;
74 
75  // fill values with hexadecimal representation of the record number
76  static constexpr uint8_t hexdigits[16] = {
77  '0', '1', '2', '3', '4', '5', '6', '7',
78  '8', '9', 'A', 'B', 'C', 'D', 'E', 'F'
79  };
80  uint64_t rec = index;
81  for (size_t i = 0; i != 2 * sizeof(rec); ++i)
82  *v++ = hexdigits[(rec >> (4 * i)) & 0x0F];
83  for (size_t i = 0; i != 16; ++i)
84  *v++ = '0';
85 
86  // add 4 bytes "break"
87  *v++ = 0x88;
88  *v++ = 0x99;
89  *v++ = 0xAA;
90  *v++ = 0xBB;
91 
92  // add 48 byte filler based on index
93  for (size_t i = 0; i < 12; ++i) {
94  uint8_t f = hexdigits[((20 + rec) >> (4 * i)) & 0x0F];
95  *v++ = f;
96  *v++ = f;
97  *v++ = f;
98  *v++ = f;
99  }
100 
101  // add 4 bytes "break"
102  *v++ = 0xCC;
103  *v++ = 0xDD;
104  *v++ = 0xEE;
105  *v++ = 0xFF;
106 
107  assert(v == r.value + 90);
108 
109  return r;
110  }
111 
112 private:
113  std::default_random_engine rng_ { std::random_device { } () };
114 };

26 //! calculate the log2 floor of an integer type (by repeated bit shifts)
27 template <typename IntegerType>
28 static inline unsigned int IntegerLog2Floor(IntegerType i) {
29  unsigned int p = 0;
30  while (i >= 256) i >>= 8, p += 8;

191 /*!
192  * Calculate a Perfect Matching (1-Factor) on a Complete Graph. Used by
193  * collective network algorithms.
194  *
195  * \param r round [0..n-1) of matching
196  * \param p rank of this processor 0..n-1
197  * \param n number of processors (graph size)
198  * \return peer processor in this round
199  */
200 static inline size_t CalcOneFactorPeer(size_t r, size_t p, size_t n) {
201  assert(r < CalcOneFactorSize(n));
202  assert(p < n);
203 
204  if (n % 2 == 0) {
205  // p is even
206  size_t idle = (r * n / 2) % (n - 1);
207  if (p == n - 1)
208  return idle;
209  else if (p == idle)
210  return n - 1;
211  else
212  return (r - p + n - 1) % (n - 1);
213  }
214  else {
215  // p is odd
216  return (r - p + n) % n;
217  }
218 }
